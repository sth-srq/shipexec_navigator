            function captureImage () {
                // vm.shipping.videoModal.stream.id = vm.shipping.videoModal.stream.selector
                var cameraStream = document.getElementById(vm.shipping.videoModal.stream.id);
                var track = cameraStream.srcObject.getVideoTracks()[0];
                var imageCapture = new ImageCapture(track);

                imageCapture.grabFrame().then(function(imageBitmap) {
                    const canvas = document.createElement('canvas');
                    const ctx = canvas.getContext('2d');

                    // Set canvas dimensions to match the imageBitmap
                    canvas.width = imageBitmap.width;
                    canvas.height = imageBitmap.height;

                    // Draw the imageBitmap onto the canvas
                    ctx.drawImage(imageBitmap, 0, 0);

                    // Get the image data as a Base64 data URL
                    // You can specify format and quality (e.g., 'image/jpeg', 0.8)
                    const base64DataUrl = canvas.toDataURL('image/png', 1.0);
                    // var newTab = window.open('about:blank','image from canvas');
                    // newTab.document.write("<img src='" + base64DataUrl + "' alt='from canvas'/>");

                    vm.loadValue = 'Label Capture';
                    vm.labelImage = base64DataUrl.replace(/^data:.+;base64,/, '');

                    vm.load();

            function videoStreamStop(isCanceled) {
                vm.labelImage = isCanceled ? null : vm.labelImage;

                var cameraStream = document.getElementById(vm.shipping.videoModal.stream.id);
                if (cameraStream.srcObject) {
                    var tracks = cameraStream.srcObject.getTracks();
                    tracks.forEach(track => track.stop()); // Stop each track in the stream
                    cameraStream.srcObject = null;
                }
            }

// videoStreamStart() // returns undefined
            function videoStreamStart () {
                if (navigator.mediaDevices && navigator.mediaDevices.getUserMedia) {
                    // Request access to the video stream
                    var constraints = {
                        audio: false,
                        video: {
                            zoom: true,
                            facingMode: 'environment',
                            // aspectRatio: { ideal: 9/16 }
                        }
                    };
                    //HD
                    // constraints.video.width = { ideal: 1920 };
                    // constraints.video.height = { ideal: 1080 };

                    //QHD
                    constraints.video.width = { ideal: 2560 };
                    constraints.video.height = { ideal: 1440 };

                    //4K UHD
                    // constraints.video.width = { ideal: 3840 };
                    // constraints.video.height = { ideal: 2160 }

                    navigator.mediaDevices.getUserMedia(constraints).then(function(stream) {
                        // Assign the stream to the video element's source object
                        var cameraStream = document.getElementById(vm.shipping.videoModal.stream.id);
                        cameraStream.srcObject = stream;

                        var [track] = stream.getVideoTracks();
                        var capabilities = track.getCapabilities();

                        vm.videoModal.zoom ??= {
                            min: capabilities.zoom.min,
                            max: capabilities.zoom.max,
                            current: track.getSettings().zoom
                        }
                        var zoomLevel = localStorage.getItem('zoomLevel') ?? getZoomLevel(track.getSettings().zoom);

                        track.applyConstraints({ advanced: [{ zoom: zoomLevel }] });
                        vm.videoModal.track = track;
                        // if (capabilities.zoom) {
                        //     track.applyConstraints({ advanced: [{ zoom: capabilities.zoom.max }] });
                        // }
                    }).catch(function(error) {
                        console.error("Error accessing the camera: ", error);
                    });
                } else {
                    // Fallback for older browsers
                    console.error("Your browser does not support the getUserMedia API.");
                }

                function getZoomLevel(currentZoom) {
                    localStorage.setItem('zoomLevel', currentZoom);
                    return currentZoom;
                }
            }
