                if (this.menu.guide.inputFile.selected) {
                    if (this.menu.guide.selectedDownloadFile.slice(-3) == this.menu.guide.inputFile.selected.name.slice(-3))
                        updateFile();
                } else {
                    var payload = { Action: 'downloadFile', Data: this.menu.guide.selectedDownloadFile };
                    var data = { UserContext: undefined, Data: JSON.stringify(payload) };
                    thinClient.httpClient('UserMethod', data).then(function (ret) {
                        if (ret.ErrorCode === 0) {
                            var data = JSON.parse(atob(ret.Data));

                            let linkData = `data:${data.fileType};base64,${data.encodedFile}`;
                            let downloadLink = document.createElement("a");
                            downloadLink.style.display = "none";
                            downloadLink.download = data.fileName;
                            downloadLink.href = linkData;
                            document.body.appendChild(downloadLink);
                            downloadLink.click();
                            document.body.removeChild(downloadLink);
                        }
                    });
                }

                function updateFile() {
                    var reader = new FileReader();
                    reader.onload = function () {
                        var request = {
                            Key: vm.shipping.menu.guide.selectedDownloadFile,
                            Value: JSON.stringify({
                                fileType: vm.shipping.menu.guide.inputFile.selected.type,
                                fileName: vm.shipping.menu.guide.inputFile.selected.name,
                                encodedFile: this.result.split(',')[1]
                            })
                        }
                        var payload = { Action: 'updateFile', Data: request };
                        var data = { UserContext: undefined, Data: JSON.stringify(payload) };
                        thinClient.httpClient('UserMethod', data).then(function (ret) {
                            if (ret.ErrorCode !== 0)
                                vm.notificationService.notifyDanger(ret.ErrorCode + " - " + ret.ErrorMessage);
                        }).catch(function (response) {
                            vm.notificationService.notifyDanger(vm.translations.Common.ErrorConnectingWCFService + ": " + response.status + " " + response.statusText);
                        })
                        vm.shipping.menu.guide.inputFile.selected = null;
                        $(vm.shipping.menu.guide.inputFile.selector).val(null);
                        vm.thinClient.loadData.show();

                    }
                    if (vm.shipping.menu.guide.inputFile.selected) reader.readAsDataURL(vm.shipping.menu.guide.inputFile.selected);
                }
            }

// SetShipNotificationAddressEmail(pkg, ngModelName, emailAddress, checked) // returns void
        function SetShipNotificationAddressEmail(pkg, ngModelName, emailAddress, checked) {
        if (checked) {
            if (pkg.ShipNotificationAddressEmail == undefined || pkg.ShipNotificationAddressEmail.trim() == '') {
                if (ngModelName == 'ShipFromNotificationEmail' && emailAddress.Origin) {
                    pkg.ShipNotificationAddressEmail = emailAddress.Origin;
                } else if (ngModelName == 'ShipToNotificationEmail' && emailAddress.Consignee) {
                    pkg.ShipNotificationAddressEmail = emailAddress.Consignee;
                }
            } else {
                if (ngModelName == 'ShipFromNotificationEmail' && emailAddress.Origin) {
                    var regexp = new RegExp(',?' + emailAddress.Origin, 'g');
                    if (!regexp.test(pkg.ShipNotificationAddressEmail)) {
                        pkg.ShipNotificationAddressEmail += ',' + emailAddress.Origin;
                    }
                } else if (ngModelName == 'ShipToNotificationEmail' && emailAddress.Consignee) {
                    pkg.ShipNotificationAddressEmail += ',' + emailAddress.Consignee;
                }
            }
        } else {
            var notificationAddressEmail = pkg.ShipNotificationAddressEmail;
            if (ngModelName == 'ShipToNotificationEmail' && emailAddress.Consignee) {
                var regexp = new RegExp(',?' + emailAddress.Consignee, 'g')
                notificationAddressEmail = notificationAddressEmail.replace(regexp, '');
            } else if (ngModelName == 'ShipFromNotificationEmail' && emailAddress.Origin) {
                var regexp = new RegExp(',?' + emailAddress.Origin, 'g')
                notificationAddressEmail = notificationAddressEmail.replace(regexp, '');
            }

            if (notificationAddressEmail.indexOf(',') == 0) {
                notificationAddressEmail = notificationAddressEmail.substring(1);
            }
            pkg.ShipNotificationAddressEmail = notificationAddressEmail;
        }
    }

// convertFileInputToBase64(inputId, leg) // returns Promise resolving upload payload
    this.convertFileInputToBase64 = function (inputId, leg) {
        return new Promise(function (resolve, reject) {

            var data = {};
            const fileInput = $('#' + inputId);

            // Do not continue if there are no files
            if (!fileInput ||
                fileInput.length === 0 ||
                !fileInput[0].files ||
                fileInput[0].files.length === 0) {
                return;
            }

            // Get basic file info
            var file = fileInput[0].files[0];
            var fileName = file.name;

            // Create reader for reading files
            var reader = new FileReader();
            reader.readAsDataURL(file);
            reader.addEventListener("load", function () {

                // Create and return payload for upload call
                data = {
                    Action: "PaperlessDocUpload",
                    FileName: fileName,
                    encodedString: reader.result.split(',')[1],
                    shipperName: that.currentShipment.PackageDefaults.Shipper,
                    leg: leg
                }

                resolve(data);

            }, false);

            reader.addEventListener("error", function (error) {
                reject(error);
            }, false);

        });

    }
