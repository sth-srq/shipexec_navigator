// SetCurrentUserContext(viewModel, bIncludeUser) // returns false or sets internal currentUserContext
      SetCurrentUserContext: function (viewModel, bIncludeUser)
      {
        /* If we only need the CompanyId, we can get that from the vmProfile. */
        /* If we need the UserId, we'll need to use a $.get call. */
        var localContext = {};

        try
        {
          /* The .UserInformation is not always populated in time and will throw an error so wrap it in a try/catch. */
          /* If this info is available, assign it to the local object and return out of the function. */
          if (viewModel.UserInformation)
          {
            localContext.CompanyId = viewModel.UserInformation.CompanyId;
            localContext.UserId = viewModel.UserInformation.UserId;
            currentUserContext = localContext;
            return false;
          }

          if (!currentUserContext.UserId && bIncludeUser === true)
          {
            $.ajax(
              {
                url: 'api/usercontext/GET',
                method: 'GET',
                async: false
              })
              .done(function (data)
              {
                localContext = data;
              })
              .fail(function (jqXHR, textStatus)
              {
                Logger.Log({ Source: 'SetCurrentUserContext()', Error: { name: 'AjaxError', message: jqXHR.ResponseText } });
              });
          }
          else if (!currentUserContext.CompanyId)
          {
            /* Just need the CompanyId, get it from the profile, but don't null out the UserId if it is present. */
            localContext.CompanyId = viewModel.profile.CompanyId;
            localContext.UserId = currentUserContext.UserId;
          }
        }
        catch (error)
        {
          Logger.Log({ Source: 'SetCurrentUserContext()', Error: error });
        }

        currentUserContext = localContext;

      }, /* end GetUserContext: function (vmProfile, bIncludeUser) */

      {

        /* If we need the UserId, we'll need to use a $.get call. */
        var localContext = {};

        {
          /* The .UserInformation is not always populated in time and will throw an error so wrap it in a try/catch. */

          if (viewModel.UserInformation.CompanyId)

            localContext.CompanyId = viewModel.UserInformation.CompanyId;
            localContext.UserId = viewModel.UserInformation.UserId;
            currentUserContext = localContext;
            return false;

          if (!currentUserContext.UserId && bIncludeUser === true)
          {
            $.ajax(
              {
                url: 'api/usercontext/GET',
                method: 'GET',
                async: false
              })
              .done(function (data)
              {
                localContext = data;

              .fail(function (jqXHR, textStatus)
              {
                Logger.Log({ Source: 'SetCurrentUserContext()', Error: { name: 'AjaxError', message: jqXHR.ResponseText } });
              });

            localContext.UserId = currentUserContext.UserId;
          }
        }
        catch (error)
        {
          Logger.Log({ Source: 'SetCurrentUserContext()', Error: error });
        }

// downloadFile(exportFileName, encodedString) // returns undefined
function downloadFile(exportFileName, encodedString) {
        var decodedResult = atob(encodedString); // decode a base-64 encoded string
        var csvFile = new Blob([decodedResult], { type: "text/csv" });
    
        var ie = navigator.userAgent.match(/MSIE\s([\d.]+)/),
            ie11 = navigator.userAgent.match(/Trident\/7.0/) && navigator.userAgent.match(/rv:11/),
            ieEdge = navigator.userAgent.match(/Edge/g),
            ieVer = (ie ? ie[1] : (ie11 ? 11 : (ieEdge ? 12 : -1)));
    
        exportFileName = exportFileName + '.csv';
    
        if (ieVer > -1) {
            window.navigator.msSaveBlob(csvFile, exportFileName);
    
        } else {
            var downloadLink = document.createElement("a");
            downloadLink.download = exportFileName;
            downloadLink.href = window.URL.createObjectURL(csvFile);
            downloadLink.style.display = "none";
            document.body.appendChild(downloadLink);
            downloadLink.click();
            document.body.removeChild(downloadLink);
        }
    }
