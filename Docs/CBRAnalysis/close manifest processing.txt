      SaveConsolidatedOrderNumbers: function ()
      {
        try
        {
          var ordersToQuery = '';

          $('#selectConsolidatedOrders > option.not-processed').each(function ()
          {
            ordersToQuery += $(this).text() + ',';
          });

          /* Done creating the list of orders to get from the server which is now held in the ordersToQuery variable. */

          /* Hide the div. */
          $('#divModalConsolidateShipments').hide();

          /* Make sure there are orders in the list. */
          if (ordersToQuery.length > 0)
          {
            //ConsolidatedOrders.GetOrderInformationFromCustomer(ordersToQuery);
            Tools.MakeUserMethodRequest('GetOrderInformation', ordersToQuery, true, this.ProcessDataFromCustomer);
          }
          else
          {
            Tools.ShowErrorAlert('ERROR: No unique orders to pull from the server.  All consolidated orders have already been added to this shipment.');
          }

        }
        catch (error)
        {
          Logger.Log({ Source: 'SaveConsolidatedOrderNumbers()', Error: error });
        }
      }

// ProcessDataFromCustomer(jsonData) // returns undefined
      ProcessDataFromCustomer: function (jsonData)
      {
        try
        {
          $(jsonData).each(function (a, b)
          {
            /* For testing, this is set to '!== true' so it will return order data.  This needs to be set to '=== true' for any other case. */
            if (ConsolidatedOrders.CheckForMatchingConsignee(b) == true)
            {
              var currentPackage = Tools.GetCurrentPackage();
              if (!currentPackage)
              {
                Tools.ShowErrorAlert('ERROR:  Could not add commodities to package for order ' + b.orderNumber + '.  See error console for details.');
                return false;
              }
              ConsolidatedOrders.UpdateReferencesWithOrderNumbers(b.orderNumber);
            }
            else
            {
              Tools.ShowErrorAlert('ERROR 500:  Could not add commodities for order ' + b.orderNumber + '.  Address mismatch.');
            }
          });
        }
        catch (error)
        { 
          console.log(error);
        }
      }

    function getFTPUploaded() {
        var selectedShipper = document.getElementById("Shipper").value;
		var shipper = '';
		debugger;
		if (selectedShipper !== "")
		{
			selectedShipper = JSON.parse(selectedShipper);
			shipper = selectedShipper.Name;
		}
        var data = { Action: 'ExportFTP', Shipper: shipper };

        $.post('api/ShippingService/UserMethod',
            {UserContext: userContext, Data: JSON.stringify(data) },
            function(response, status, xhr) {
            }
        ).done(function(response) {
            debugger;
            var resp = JSON.stringify(response);
            resp = JSON.parse(resp);
            debugger;
            if (resp.Data !== "" && resp.Data !== undefined)
            {
                alert('Lastest File: \n' + DecodeReturnString(resp.Data).substring(0,resp.Data.length -1));
            }
            else if (resp.ErrorMessage !== undefined)
            {
                if (resp.ErrorCode == '104')
                {
                    alert('Lastest File: \nNo files found!');
                }
                else
                {
                    alert(resp.ErrorMessage);
                }
            }
        });
    }

      SaveConsolidatedOrderNumbers: function ()
      {
        try
        {
          /* Append the shipper reference with each item in the select list in csv format. */
          /* Once this is done, change the SSBR on post load to loop through each option in the csv list. */
          var inputShipperReference = $('input[type=text][ng-model="vm.currentShipment.Packages[vm.packageIndex].ShipperReference"]');
          var shipperReferenceVal = $(inputShipperReference).val();

          /* If the user closes the dialog box, then opens it back up, we don't want to check the same orders more than once so only process what was entered into the dialog box this time. */
          var ordersToQuery = '';

          /* Loop through all the orders in the select box and determine if they are already in the list. */
          $('#selectConsolidatedOrders').find('option').each(function (i, v)
          {
            /* Remove the spaces before the commas for comparison. */
            var arrayOrders = shipperReferenceVal.split(",").map(function (item)
            {
              return item.trim();
            });

            /* If the current option value is NOT in the list within the Shipper Reference field, add it. */
            if ($.inArray(v.value.trim(), arrayOrders) == -1)
            {
              ordersToQuery += v.value.trim() + ',';
              //shipperReferenceVal += ',' + v.value;
            }

            /* Remove the option once it is added to the input box. */
            $(this).remove();

          }).promise().done(function ()
          {
            /* Done creating the list of orders to get from the server which is now held in the ordersToQuery variable. */

            /* Hide the div. */
            $('#divModalConsolidateShipments').hide();

            /* Make sure there are orders in the list. */
            if (ordersToQuery.length > 0)
            {
              PackVerify.GetOrderInformationFromCustomer(ordersToQuery);
            }
            else
            {
              PackVerify.ShowErrorAlert('ERROR: No unique orders to pull from the server.  All consolidated orders have already been added to this shipment.');
            }
          });
        }
        catch (error)
        {
          Logger.Log({ Source: 'SaveConsolidatedOrderNumbers()', Error: error });
        }
      }

  CloseContainer = (ContainerIndex) =>
  {
    let msg = {};

    try
    {
      let that = this;
      ContainerIndex = +ContainerIndex;

      this.UpdateContainerArrays('Close', ContainerIndex);

      const cont = $(this.H.PanelOpenContainersId).find(`.container-parent[container-index=${ContainerIndex}]`).first();

      //take all the .data() objects from all the list items
      const listData = $(cont).find('li.list-group-item').map(function(a, b) { return $(b).data() }).get();
      listData.map(f => f.PackedItems = f.PackedItems.filter(g => g.ContainerIndex == ContainerIndex));

      const hdr = $(cont).find('div.panel-body').first();
      const panel = $(this.H.PanelClosedContainersId);
      const icon1 = '<i class="glyphicon glyphicon-list-alt" event="ShowContainerDetails" data-toggle="tooltip" data-placement="top" title="Show box details."></i>';
      const icon2 = '<i class="glyphicon glyphicon-print" event="PrintPackingList" data-toggle="tooltip" data-placement="top" title="Print box packing list."></i>';
      const icons = `<span class="icon-row pull-right">${icon1}${icon2}</span>`;

      $(panel).find('.collapse-text').slideUp();

      $(cont).animate({ height: 0, opacity: 0 }, 300, function(e)
      {
        let newItem = $(cont).detach().data(listData);

        //after we hide the container remove the elements that aren't used in the closed containers panel
        $(newItem).find('ul.list-group, div.panel-body-sm, div.panel-menu, div.panel-icons-right').remove();

        $(hdr).removeAttr('event').find('.panel-text').first().css('width', '100%').append(icons);

        $(panel).append(newItem);

        $(newItem).animate({ 'height': '2.65em', 'opacity': 1 }, 300, function() { $(cont).find('[data-toggle="tooltip"]').tooltip(); });

        //expand a new active container if we just closed the active one
        if ($(that.H.PanelOpenContainersActive).length === 0)
        {
          let i = $.grep($('#divPanelOpenContainers .container-parent'), function(v, i) { return +$(v).attr('container-index') < ContainerIndex })[0];
          that.SetActiveContainer(+$(i).attr('container-index'));
        }

        msg.Action = 'CloseContainer';
        msg.ContainerIndex = ContainerIndex;
        that.UpdateToastNotification(msg);
      });
    }
    catch (error)
    {
      console.log(error);
      msg.Action = 'Error';
      msg.ErrorMessage = error.message;
      this.UpdateToastNotification(msg);
    }

  }

// InitCloseContainerDialog() // returns void
  InitCloseContainerDialog = () =>
  {
    let that = this;

    $(this.closeContainerDialog).delegate('button[op]', 'click', function(e)
    {
      const op = $(this).attr('op');

      switch (op)
      {
        case 'ok':
          that.OkCloseContainer();
          break;
        case 'cancel':
          break;
        default:
          break;
      }

      that.packVerify.Overlay.Hide();
      $(that.closeContainerDialog).fadeOut().find('input').val([]);
    });
  }

    }

    $(this.packVerify.H.DialogCloseContainerId).fadeOut();
  }
