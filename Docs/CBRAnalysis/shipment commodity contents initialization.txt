// function GetCommodityContentObject(data) // returns properly formatted commodity contents object with extra unique id and package index fields
    GetCommodityContentObject(data)
    {
      let cc = {};
      cc.Quantity = data.quantity;
      cc.ProductCode = data.skuId;
      cc.QuantityUnitMeasure = 'EA';
      cc.UnitValue = { Currency: 'USD', Amount: +data.unitPrice };
      cc.UnitWeight = { Units: 'LB', Amount: +data.weightLbs };
      cc.UniqueId = data.UniqueId;
      cc.PkgIndex = data.PkgIndex || 1;
      cc.PVTotalWeight = (+data.quantity * +data.weightLbs);
      cc.PVTotalValue = (+data.quantity * +data.unitPrice);

      return cc;
    }

// function MoveGoodToBox(li, destBox) // returns void
    MoveGoodToBox(li, destBox)
    {
      let vm = this.ClassViewModel;
      let that = this;
      let data = $(li).data();
      let destCC = this.packages[destBox - 1].CommodityContents || [];

      //find the box the item currently resides in
      //split this statement up over a few calls to make it easier to read
      let allGoods = this.packages.map(m => m.CommodityContents).flat();
      let item = allGoods.find(cc => cc.UniqueId == data.UniqueId);

      if (+item.PkgIndex == +destBox) { console.log('Item already exists in box.'); return false; }

      //remove the item from its current box
      this.packages[item.PkgIndex - 1].CommodityContents = this.packages[item.PkgIndex - 1].CommodityContents.filter(p => p.UniqueId != data.UniqueId);

      //add it to the new box
      item.PkgIndex = destBox;
      destCC.push(item);
      $(li).find('.pkg-index-text').fadeOut().promise().done(function(e) { $(this).text(destBox).fadeIn(); });

      //console.log(vm.currentShipment.Packages);
      $('li[heading="Goods"]:visible:first > a').trigger('click');
      $('#goods .btn-group').find('.btn:gt(0)').not('.active').first().trigger('click');

      //update the weights and values
      //this data will have changed by now so get a new reference to the object
      allGoods = this.packages.map(m => m.CommodityContents).flat();
      //the items array will have the CommodityContents object for every package in the shipment

      //sort by package index
      //update the weight and declared values for all the packages in the shipment
      allGoods = allGoods.sort((a, b) => { return a.PkgIndex - b.PkgIndex });
      $(allGoods).each(function(a, b)
      {
        let pkg = that.packages[b.PkgIndex - 1];

        if ((a == 0) || (b.PkgIndex != allGoods[a - 1].PkgIndex))
        {
          pkg.Weight.Amount = 0;
          pkg.DeclaredValueAmount.Amount = 0;
        }

        pkg.Weight.Amount += +b.PVTotalWeight;
        pkg.DeclaredValueAmount.Amount += +b.PVTotalValue;
      });
    }

    this.PreShip = function (shipmentRequest, userParams) {

        var shipDate = new Date(shipmentRequest.PackageDefaults.Shipdate.Year, shipmentRequest.PackageDefaults.Shipdate.Month - 1, shipmentRequest.PackageDefaults.Shipdate.Day);
        var today = new Date();

        if (shipDate.getDay() == 6) {
            throw ({ message: 'Saturday shipping is not allowed. Code[CBR]"', errorCode: '115' });
        }
        
        if (shipDate.getDay() == 7) {
            throw ({ message: 'Sunday shipping is not allowed. Code[CBR]"', errorCode: '115' });
        }
        

        if (shipmentRequest.Packages[0].ShipperReference === undefined || shipmentRequest.Packages[0].ShipperReference === null || shipmentRequest.Packages[0].ShipperReference === '') {
            throw ({ message: 'Study Reference is required.  Please select a valid Study Group in order to continue. [CBR ShipperReference]', errorCode: '115' });
        }
        if (shipmentRequest.Packages[0].ConsigneeReference === undefined || shipmentRequest.Packages[0].ConsigneeReference === null || shipmentRequest.Packages[0].ConsigneeReference === '') {
            throw ({ message: 'Study Reference is required.  Please select a valid Study Group in order to continue. [CBR ConsigneeReference]', errorCode: '115' });
        }

        var diw = shipmentRequest.Packages[0].DryIceWeight;
        var pkgw = shipmentRequest.Packages[0].Weight;

            throw ({ message: 'Shipping Error: Dry Ice Weight MUST be less than the package Weight.', errorCode: '115' });

// ApplyCommodityAssignments() // returns undefined
  ApplyCommodityAssignments()
  {
    let that = this;

    //every list item will have a pkg-index attr, get that highest value to determine how many pkgs we need
    //don't use pkg-status.length because that will include multiple commodities and the end result will be wrong
    let packed = $('#listItemContainer').find('li[pkg-status=packed]');

    //first get all the pkg-index attributes, then get the highest number and add one
    let pkgs = $(packed).map(function(x, value) { return +$(this).attr('pkg-index') }).toArray();
    let pkgCount = Math.max(...pkgs) + 1;

    while (this.packages.length < pkgCount)
    {
      //the vm.addPackage isn't firing, just click the button
      $('button[ng-click="vm.addPackage();"]').trigger('click').promise().done(function(e) { });
    }

    //loop the pkg count, get the commodities for that pkg, write the objects to the vm CommodityContents object
    for (let p = 0; p < pkgCount; p++)
    {
      let w = $.grep(this.working, function(item, i) { return item.packageIndex == p });
      this.packages[p].CommodityContents = w;
    }

    //click the General tab
    $('li[heading=General]:visible a').first().click();
    this.HideAssignGoodsPane();

    //after we are done, move back to the first package in the shipment
    while ($('#spanCurrentPackageIndex').text() !== '1')
    {
      $('button[ng-click="vm.previousPackage();"]').first().click();
    }

  }

// GetCommodityContentObject(data) // returns cc object
GetCommodityContentObject(data)
    {
      let cc = {};
      cc.Quantity = data.quantity;
      cc.ProductCode = data.skuId;
      cc.QuantityUnitMeasure = 'EA';
      cc.UnitValue = { Currency: 'USD', Amount: +data.unitPrice };
      cc.UnitWeight = { Units: 'LB', Amount: +data.weightLbs };
      cc.UniqueId = data.UniqueId;

// MoveGoodToBox(li, destBox) // returns false | void
MoveGoodToBox(li, destBox)
    {
      let vm = this.ClassViewModel;
      let that = this;
      let data = $(li).data();
      let destCC = this.packages[destBox - 1].CommodityContents || [];

      //find the box the item currently resides in
      //split this statement up over a few calls to make it easier to read
      let allGoods = this.packages.map(m => m.CommodityContents).flat();
      let item = allGoods.find(cc => cc.UniqueId == data.UniqueId);

      if (+item.PkgIndex == +destBox) { console.log('Item already exists in box.'); return false; }

      //remove the item from its current box
      this.packages[item.PkgIndex - 1].CommodityContents = this.packages[item.PkgIndex - 1].CommodityContents.filter(p => p.UniqueId != data.UniqueId);

      //add it to the new box
      item.PkgIndex = destBox;
      destCC.push(item);
      $(li).find('.pkg-index-text').fadeOut().promise().done(function(e) { $(this).text(destBox).fadeIn(); });

      //console.log(vm.currentShipment.Packages);
      $('li[heading="Goods"]:visible:first > a').trigger('click');
      $('#goods .btn-group').find('.btn:gt(0)').not('.active').first().trigger('click');

      //update the weights and values
      //this data will have changed by now so get a new reference to the object
      allGoods = this.packages.map(m => m.CommodityContents).flat();
      //the items array will have the CommodityContents object for every package in the shipment

      //sort by package index
      //update the weight and declared values for all the packages in the shipment
      allGoods = allGoods.sort((a, b) => { return a.PkgIndex - b.PkgIndex });
      $(allGoods).each(function(a, b)
      {
        let pkg = that.packages[b.PkgIndex - 1];

        if ((a == 0) || (b.PkgIndex != allGoods[a - 1].PkgIndex))
        {
          pkg.Weight.Amount = 0;
          pkg.DeclaredValueAmount.Amount = 0;
        }

        pkg.Weight.Amount += +b.PVTotalWeight;
        pkg.DeclaredValueAmount.Amount += +b.PVTotalValue;
      });
    }

// function ParseJson(result) // returns undefined
    function ParseJson(result) {
        //debugger;
        var jsonLoad = result; //jQuery.parseJSON(result);
       
        if (jsonLoad !== null && jsonLoad !== undefined)
        {
            $('#search_input').css('background-color', '#fffef0');
    
            //#region Build Drop-Down List
            var html = "";

            for (let i in jsonLoad.listData)
            {
                html += "<option value='" + jsonLoad.listData[i].glCode + "'>" + jsonLoad.listData[i].fullName + " " + jsonLoad.listData[i].glCode + "</option>";
            }
    
            if (jsonLoad.listData.length === 0)
            {
                alert("No records were returned from search")
            }
            else //More than one record returned
            {
                //$("#costcodes").remove();
                //$(html).insertAfter("#searchButton");
                $("#costcodes").append(html);
                $('#costcodes').change(function() {
                    populateShipExec();
                });
                $("#costcodes").show();
            }
    
        }
        else
        {
            alert("Error populating dropdown box --- " + jsonLoad.ErrorMessage);
        }
    
    }

            var pkg1MiscRef4 = shipmentRequest.Packages[0].MiscReference4;

                alert(errorMessage);

        }

        var thirdParty = vmTemp.currentShipment.Packages[vmTemp.packageIndex].MiscReference19;

// GetNewTableMarkup(OrderData) // returns array of table row elements
  GetNewTableMarkup = (OrderData) =>
  {
    let that = this, rows = [];
    const json = new JsonData();

    //loop through the data and create the object that will be attached to each row as a .data() object
    OrderData.forEach((item, index) =>
    {
      json.TrimAllKeysAndValues(OrderData);

      //setup the row data object
      let data = {};
      //the .Count var is used when the items get placed into containers, it represents the count of items in the parent container object
      data.Count = 0;
      data.ItemNumber = item.CSpartno;
      data.ItemDescription = item.CSdescript;
      data.HarmonizedCode = (item.CSHTScode || 'Not Available');
      data.HarmonizedDescription = (item.CSHTSDESC || 'Not Available');
      data.CountryOfOrigin = (item.CSCOO || 'Not Available');
      data.OrderNumber = item.CSsono;

    this.BuildShipment = function (ret) {
        $buildShipmentModal.modal('hide');

        var weightPerPackage;
        var multiPackage = vm.currentShipment.Packages.length > 1;

        if (ret.TotalWeight !== undefined && ret.TotalWeight !== 0 && ret.TotalWeight !== "")
            weightPerPackage = (ret.TotalWeight / ret.TotalPackages).toFixed(3);

        for (var i = 0; i < ret.TotalPackages; i++) {
            vm.packageIndex = vm.currentShipment.Packages.length - 1;
            if (i > 0 || multiPackage) {
                vm.addPackage();
                multiPackage = true;
            }

            if (weightPerPackage !== undefined)
                vm.currentShipment.Packages[vm.packageIndex].Weight.Amount = weightPerPackage;
            if (ret.WeightUnit !== undefined)
                vm.currentShipment.Packages[vm.packageIndex].Weight.Units = ret.WeightUnit;
            else
                vm.currentShipment.Packages[vm.packageIndex].Weight.Units = vm.translations.WeightUnit;
        }

        vm.totalPackages = vm.currentShipment.Packages.length;

        ret.TotalWeight = '';
        ret.TotalPackages = '';
    }
