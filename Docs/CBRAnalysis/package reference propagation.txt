        $(inputShipperReference).val(newUniqueString);
        $(inputMiscReference07).val(newUniqueString);

    };

        if(!referenceValue)
            referenceValue = "";
        return referenceValue;
    };

    this.CheckValidation = function(referenceValue, referenceList) {
        var readyToShip = false;
        for (var item in referenceList) {
            //alert(referenceList[item]);
            if (referenceList[item].Value == referenceValue) {
                readyToShip = true;
                break;
            }
        }
        if (readyToShip === false) throw { message: "Unable to validate shipment", errorCode: "001" };
    };

    this.GetCustom = function(fieldName, customDataType, shipmentRequest) {
        var returnValue = "";
        var customData = null;
        if (customDataType === "User") {
            if (shipmentRequest &&
                shipmentRequest.PackageDefaults &&
                shipmentRequest.PackageDefaults.OriginAddress &&
                shipmentRequest.PackageDefaults.OriginAddress.CustomData)
                customData = shipmentRequest.PackageDefaults.OriginAddress.CustomData;
        }
        if (customDataType === "To") {
            if (shipmentRequest &&
                shipmentRequest.PackageDefaults &&
                shipmentRequest.PackageDefaults.Consignee &&
                shipmentRequest.PackageDefaults.Consignee.CustomData)
                customData = shipmentRequest.PackageDefaults.Consignee.CustomData;
        }

        if (customData) {
            customData.forEach(function(customField) {
                if (fieldName === customField.Key) returnValue = customField.Value;
            });
        }
        return returnValue;

    };
}

// function GetCommodityContentObject(data) // returns cc object with package/commodity reference fields
    GetCommodityContentObject(data)
    {
      let cc = {};
      cc.Quantity = data.quantity;
      cc.ProductCode = data.skuId;
      cc.QuantityUnitMeasure = 'EA';
      cc.UnitValue = { Currency: 'USD', Amount: +data.unitPrice };
      cc.UnitWeight = { Units: 'LB', Amount: +data.weightLbs };
      cc.UniqueId = data.UniqueId;
      cc.PkgIndex = data.PkgIndex || 1;
      cc.PVTotalWeight = (+data.quantity * +data.weightLbs);
      cc.PVTotalValue = (+data.quantity * +data.unitPrice);

      return cc;
    }

// function MoveGoodToBox(li, destBox) // returns undefined or false
    MoveGoodToBox(li, destBox)
    {
      let vm = this.ClassViewModel;
      let that = this;
      let data = $(li).data();
      let destCC = this.packages[destBox - 1].CommodityContents || [];

      //find the box the item currently resides in
      //split this statement up over a few calls to make it easier to read
      let allGoods = this.packages.map(m => m.CommodityContents).flat();
      let item = allGoods.find(cc => cc.UniqueId == data.UniqueId);

      if (+item.PkgIndex == +destBox) { console.log('Item already exists in box.'); return false; }

      //remove the item from its current box
      this.packages[item.PkgIndex - 1].CommodityContents = this.packages[item.PkgIndex - 1].CommodityContents.filter(p => p.UniqueId != data.UniqueId);

      //add it to the new box
      item.PkgIndex = destBox;
      destCC.push(item);
      $(li).find('.pkg-index-text').fadeOut().promise().done(function(e) { $(this).text(destBox).fadeIn(); });

      //console.log(vm.currentShipment.Packages);
      $('li[heading="Goods"]:visible:first > a').trigger('click');
      $('#goods .btn-group').find('.btn:gt(0)').not('.active').first().trigger('click');

      //update the weights and values
      //this data will have changed by now so get a new reference to the object
      allGoods = this.packages.map(m => m.CommodityContents).flat();
      //the items array will have the CommodityContents object for every package in the shipment

      //sort by package index
      //update the weight and declared values for all the packages in the shipment
      allGoods = allGoods.sort((a, b) => { return a.PkgIndex - b.PkgIndex });
      $(allGoods).each(function(a, b)
      {
        let pkg = that.packages[b.PkgIndex - 1];

        if ((a == 0) || (b.PkgIndex != allGoods[a - 1].PkgIndex))
        {
          pkg.Weight.Amount = 0;
          pkg.DeclaredValueAmount.Amount = 0;
        }

        pkg.Weight.Amount += +b.PVTotalWeight;
        pkg.DeclaredValueAmount.Amount += +b.PVTotalValue;
      });
    }

function SetShipNotificationAddressEmail(pkg, ngModelName, emailAddress, checked) {
        if (checked) {
            if (pkg.ShipNotificationAddressEmail == undefined || pkg.ShipNotificationAddressEmail.trim() == '') {
                if (ngModelName == 'ShipFromNotificationEmail' && emailAddress.Origin) {
                    pkg.ShipNotificationAddressEmail = emailAddress.Origin;
                } else if (ngModelName == 'ShipToNotificationEmail' && emailAddress.Consignee) {
                    pkg.ShipNotificationAddressEmail = emailAddress.Consignee;
                }
            } else {
                if (ngModelName == 'ShipFromNotificationEmail' && emailAddress.Origin) {
                    var regexp = new RegExp(',?' + emailAddress.Origin, 'g');
                    if (!regexp.test(pkg.ShipNotificationAddressEmail)) {
                        pkg.ShipNotificationAddressEmail += ',' + emailAddress.Origin;
                    }
                } else if (ngModelName == 'ShipToNotificationEmail' && emailAddress.Consignee) {
                    pkg.ShipNotificationAddressEmail += ',' + emailAddress.Consignee;
                }
            }
        } else {
            var notificationAddressEmail = pkg.ShipNotificationAddressEmail;
            if (ngModelName == 'ShipToNotificationEmail' && emailAddress.Consignee) {
                var regexp = new RegExp(',?' + emailAddress.Consignee, 'g')
                notificationAddressEmail = notificationAddressEmail.replace(regexp, '');
            } else if (ngModelName == 'ShipFromNotificationEmail' && emailAddress.Origin) {
                var regexp = new RegExp(',?' + emailAddress.Origin, 'g')
                notificationAddressEmail = notificationAddressEmail.replace(regexp, '');
            }

            if (notificationAddressEmail.indexOf(',') == 0) {
                notificationAddressEmail = notificationAddressEmail.substring(1);
            }
            pkg.ShipNotificationAddressEmail = notificationAddressEmail;
        }
    }

// UpdateListItemPackageQuantities(li) // returns void
  UpdateListItemPackageQuantities(li)
  {
    this.UpdatePkgCountsInWorkingObject(li);

    let numOfItems = $(li).find('.num-of-items-in-pkg');
    let remItems = $(li).find('.goods-quantity > strong');

    if (!$(numOfItems).val()) { return; }

    let x = (+$(remItems).text() - $(numOfItems).val());

    //sanity check to make sure we didn't pack more items that we had
    if (x < 0)
    {
      $(remItems).addClass('bad-math');
      return;
    }

    $(remItems).removeClass('bad-math');
    $(numOfItems).attr('disabled', 'disabled');
    $(li).attr('pkg-status', 'packed');

    //move to the next list item and update then focus
    $(li).next().find('.goods-quantity > strong').text(x)

    //on the second-to-last list item, use the input value to auto-set the last item
    //to get the number of rows that have already been assigned, count the list items with the goods-key
    //that have disabled input boxes
    let rowsForThisGood = $('#listItemContainer').find('li[goods-key="' + $(li).attr('goods-key') + '"]');
    let rowsDone = $(rowsForThisGood).find('input:disabled:visible').length;

    if (rowsForThisGood.length == (rowsDone + 1))
    {
      let lastLi = $('#listItemContainer').find('li[goods-key="' + $(li).attr('goods-key') + '"]').last();

      $(lastLi).attr('pkg-status', 'packed');

      let lastVal = $(li).next().find('.goods-quantity > strong').text();
      $(lastLi).find('.num-of-items-in-pkg').val(lastVal).attr('disabled', 'disabled');

      $('#listItemContainer').find('.assign-to-pkg:visible').first().focus();

      this.UpdatePkgCountsInWorkingObject(lastLi);
    }
  }

// ApplyCommodityAssignments() // returns void
  ApplyCommodityAssignments()
  {
    let that = this;

    //every list item will have a pkg-index attr, get that highest value to determine how many pkgs we need
    //don't use pkg-status.length because that will include multiple commodities and the end result will be wrong
    let packed = $('#listItemContainer').find('li[pkg-status=packed]');

    //first get all the pkg-index attributes, then get the highest number and add one
    let pkgs = $(packed).map(function(x, value) { return +$(this).attr('pkg-index') }).toArray();
    let pkgCount = Math.max(...pkgs) + 1;

    while (this.packages.length < pkgCount)
    {
      //the vm.addPackage isn't firing, just click the button
      $('button[ng-click="vm.addPackage();"]').trigger('click').promise().done(function(e) { });
    }

    //loop the pkg count, get the commodities for that pkg, write the objects to the vm CommodityContents object
    for (let p = 0; p < pkgCount; p++)
    {
      let w = $.grep(this.working, function(item, i) { return item.packageIndex == p });
      this.packages[p].CommodityContents = w;
    }

    //click the General tab
    $('li[heading=General]:visible a').first().click();
    this.HideAssignGoodsPane();

    //after we are done, move back to the first package in the shipment
    while ($('#spanCurrentPackageIndex').text() !== '1')
    {
      $('button[ng-click="vm.previousPackage();"]').first().click();
    }

  }

  GetNewCommodityObject(obj)
  {
    let c = {};

    Object.keys(obj).forEach(key =>
    {
      if (key != '$$hashKey' && key != 'packageIndex')
      {
        let value = obj[key];
        if (value) c[key] = value;
      }
    });

    return c;
  }

this.GetCustom = function(fieldName, customDataType, shipmentRequest) {
    var returnValue = "";
    var customData = null;
    if (customDataType === "User") {
        if (shipmentRequest &&
            shipmentRequest.PackageDefaults &&
            shipmentRequest.PackageDefaults.OriginAddress &&
            shipmentRequest.PackageDefaults.OriginAddress.CustomData)
            customData = shipmentRequest.PackageDefaults.OriginAddress.CustomData;
    }

    if (customData) {
        customData.forEach(function(customField) {
            if (fieldName === customField.Key) returnValue = customField.Value;
        });
    }
    return returnValue;

// MoveGoodToBox(li, destBox) // returns boolean
MoveGoodToBox(li, destBox)
    {
      let vm = this.ClassViewModel;
      let that = this;
      let data = $(li).data();
      let destCC = this.packages[destBox - 1].CommodityContents || [];

      //find the box the item currently resides in
      //split this statement up over a few calls to make it easier to read
      let allGoods = this.packages.map(m => m.CommodityContents).flat();
      let item = allGoods.find(cc => cc.UniqueId == data.UniqueId);

      if (+item.PkgIndex == +destBox) { console.log('Item already exists in box.'); return false; }

      //remove the item from its current box
      this.packages[item.PkgIndex - 1].CommodityContents = this.packages[item.PkgIndex - 1].CommodityContents.filter(p => p.UniqueId != data.UniqueId);

      //add it to the new box
      item.PkgIndex = destBox;
      destCC.push(item);
      $(li).find('.pkg-index-text').fadeOut().promise().done(function(e) { $(this).text(destBox).fadeIn(); });

      //console.log(vm.currentShipment.Packages);
      $('li[heading="Goods"]:visible:first > a').trigger('click');
      $('#goods .btn-group').find('.btn:gt(0)').not('.active').first().trigger('click');

      //update the weights and values
      //this data will have changed by now so get a new reference to the object
      allGoods = this.packages.map(m => m.CommodityContents).flat();
      //the items array will have the CommodityContents object for every package in the shipment

      //sort by package index
      //update the weight and declared values for all the packages in the shipment
      allGoods = allGoods.sort((a, b) => { return a.PkgIndex - b.PkgIndex });
      $(allGoods).each(function(a, b)
      {
        let pkg = that.packages[b.PkgIndex - 1];

        if ((a == 0) || (b.PkgIndex != allGoods[a - 1].PkgIndex))
        {
          pkg.Weight.Amount = 0;
          pkg.DeclaredValueAmount.Amount = 0;
        }

        pkg.Weight.Amount += +b.PVTotalWeight;
        pkg.DeclaredValueAmount.Amount += +b.PVTotalValue;
      });
    }

// openFileDialog() // returns undefined
    function openFileDialog() {
        //debugger;
        var filesSelected = document.getElementById("fileLoader");

        if (filesSelected !== undefined) {
            const fileReader = new FileReader();

            if (filesSelected.files[0]) {
                fileReader.readAsDataURL(filesSelected.files[0]);
            }

            fileReader.onload = function (event) {
                var files = event.target.result;

                // Get base64 file data
                var contents = event.target.result.split(',')[1];

                // adding prefix to beginning of file data to send to user method
                var fileData = "file|" + contents;

                // $.post('api/ShippingService/UserMethod',
                // {
                //     UserContext: userContext,
                //     Data: JSON.stringify(fileData)

                // updated for 2.3
                $.post({
                    url: clientService.config.ShipExecServiceUrl + '/UserMethod',
                    data: {
                        UserContext: client.userContext,
                        Data: JSON.stringify(fileData)
                    },
                    headers: clientService.authorizationToken()
                }).done(function(result) {
                    //debugger;
                    // result.Data should be unencoded [Success] or [Error] string
                    if (result.ErrorCode === 0 && result.Data !== undefined && result.Data.indexOf('[Error]') == -1) {
                        alert(result.Data);
                    }
                    else {
                        // display error
                        alert(result.Data);
                    }
                }).fail(function (e) {
                    alert(e.message);
                });
            };
        }
    }

// GetKeyValue(key, arr) // returns string
function GetKeyValue(key, arr) {
        for (var index in arr) {
            if (arr[index].Key.toLowerCase() === key.toLowerCase()) return arr[index].Value;
        }
        return '';
    }

// function GetShipment(shipmentRequest) // returns Shipment object
function GetShipment(shipmentRequest) {

        class Shipment {
           constructor(shipmentRequest) {
              this.ShipmentRequest = shipmentRequest;
           }

           get UserRef1() { return this.GetCustom("Custom1", "User"); }
           get UserRef2() { return this.GetCustom("Custom2", "User"); }
           get UserRef3() { return this.GetCustom("Custom3", "User"); }
           get UserRef4() { return this.GetCustom("Custom4", "User"); }
           get UserRef5() { return this.GetCustom("Custom5", "User"); }
           get UserRef6() { return this.GetCustom("Custom6", "User"); }
           get UserRef7() { return this.GetCustom("Custom7", "User"); }
           get UserRef8() { return this.GetCustom("Custom8", "User"); }
           get UserRef9() { return this.GetCustom("Custom9", "User"); }
           get UserRef10() { return this.GetCustom("Custom10", "User"); }

           get ToRef1() { return this.GetCustom("Custom1", "To"); }
           get ToRef2() { return this.GetCustom("Custom2", "To"); }
           get ToRef3() { return this.GetCustom("Custom3", "To"); }
           get ToRef4() { return this.GetCustom("Custom4", "To"); }
           get ToRef5() { return this.GetCustom("Custom5", "To"); }
           get ToRef6() { return this.GetCustom("Custom6", "To"); }
           get ToRef7() { return this.GetCustom("Custom7", "To"); }
           get ToRef8() { return this.GetCustom("Custom8", "To"); }
           get ToRef9() { return this.GetCustom("Custom9", "To"); }
           get ToRef10() { return this.GetCustom("Custom10", "To"); }

           GetCustom(fieldName, customDataType) {
              var returnValue = "";
              var customData = null;
              if (customDataType === "User") customData = this.ShipmentRequest.PackageDefaults.OriginAddress.CustomData;
              if (customDataType === "To") customData = this.ShipmentRequest.PackageDefaults.Consignee.CustomData;

              if (customData) {
                 customData.forEach(function (customField) {
                    if (fieldName === customField.Key) returnValue = customField.Value;
                 })
              }
              return returnValue;
           }
        }

        return new Shipment(shipmentRequest);
     }
