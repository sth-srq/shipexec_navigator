// function history() // returns undefined
        function history() {
            vm.history = {
                searchButton: 'button:contains("Search")',
                reportSelector: '#reports'
            };

            $(document).on('click', vm.history.searchButton, selectReportChange);
            $(document).on('change', vm.history.reportSelector, selectReportChange);

            activate();

            function activate() {
                if (vm.profile.Name != vm.thinClient.adminProfileName) {
                    vm.hideButtonExport = true;

                    let reports = [ 'Default', 'Detailed Report', 'Void Report', 'Summary Report' ];
                    vm.reports = vm.reports.filter(function(report) {
                        report.TemplateHtml = report.TemplateHtml.replace(/<span.*glyphicon-search.*<\/span>/g, '');
                        return reports.includes(report.Name);
                    });
                }
            }

            function selectReportChange(event) {
                if (vm.report?.Name == 'Summary Report') {
                    vm.hideButtonExport = true;
                    vm.history.carrierAcceptTableContainer = {
                        'height': $('.panel-body').height() + 'px',
                        'overflow-y': 'auto'
                    };

                    let payload = {
                        Action: 'summaryReport',
                        Data: {
                            startDate: vm.dtstart.toLocaleDateString('en-CA'),
                            endDate: vm.dtend.toLocaleDateString('en-CA'),
                            sites: getSites()
                        }
                    };
                    let data = { UserContext: undefined, Data: JSON.stringify(payload) };
                    thinClient.httpClient('UserMethod', data).then(function (ret) {
                        debugger;
                        vm.history.carrierAcceptReport = JSON.parse(atob(ret.Data));
                        vm.thinClient.loadData.show();
                    });
                } else {
                    vm.hideButtonExport = false;
                }

                function getSites() {
                    let sites = vm.UserInformation.SiteId ? vm.sites.filter(function(site) { return site.Id == vm.UserInformation.SiteId;}) : vm.sites;
                    return sites.map(function(site) { return { name: site.Name }; });
                }
            }
        }

// ParseCutTicket(inVal) // returns undefined
      ParseCutTicket: function (inVal)
      {
        var splitArray = inVal.split(" ", 2);
        var itemNumber = splitArray[0];
        var itemQuantity = splitArray[1];
        var remainingQuantity, shippedQuantity;
        var matchingArray;

        try
        {
          /* Now find the table row that corresponds to the item number passed into the function. */
          var rowsArray = $('#tablePackVerifyOrderDetails').find('tr.not-packed, tr.not-complete');

          matchingArray = $.grep(rowsArray, function (pVal, pIndex)
          {
            remainingQuantity = parseFloat($(pVal).find('td:eq(3)').text());
            shippedQuantity = parseFloat($(pVal).find('td:eq(2)').text());
            return (($(pVal).text().split(" ")[0] == itemNumber) && (itemQuantity == remainingQuantity));
          })[0];

          /* The matchingArray will contain rows that match the item needed with the quantity needed. */
          /* If the array is empty, find rows that have enough material to use. */
          if ($.isEmptyObject(matchingArray))
          {
            matchingArray = $.grep(rowsArray, function (pVal, pIndex)
            {
              remainingQuantity = parseFloat($(pVal).find('td:eq(3)').text());
              shippedQuantity = parseFloat($(pVal).find('td:eq(2)').text());
              return (($(pVal).text().split(" ")[0] == itemNumber) && (itemQuantity < remainingQuantity));
            })[0];
          }

          /* If the matchingArray is still empty, then the user tried to scan something into the box that wasn't part of the order.  Raise an error. */
          if ($.isEmptyObject(matchingArray))
          {
            this.ShowErrorAlert('ERROR: Item ' + itemNumber + ' not found within currently loaded order(s).');
            return false;
          }

          $(matchingArray).data().ScanQuantity = itemQuantity;
          $(matchingArray).data().ItemNumber = itemNumber;

          this.AddRowItemToCurrentBox(matchingArray);
        }
        catch (error)
        {
          Logger.Log({ Source: 'ParseCutTicket()', Error: error });
        }
      }

      AddRowItemToCurrentBox: function (triggerElement)
      {
        try
        {
          var tableRow = $(triggerElement).closest('tr');
          var commodityContent = $(tableRow).data();

          /* No box type selected, paint the border red. */
          // if ($('#selectClonedBoxTypes').prop('selectedIndex') === 0)
          // {
          //   $('#selectClonedBoxTypes').css('border-color', 'crimson');
          // }

          /* This adds the commodities to the 'Goods' tab. */
          var packageIndex = $('span.current-container-index').text();
          packageIndex -= 1;
          this.AddCommoditiesToPackage(packageIndex, commodityContent);

          var newRow = $('<tr />').css('display', 'none');
          var itemDescText = $(tableRow).find('td:eq(0)').text();
          var itemNumber = itemDescText.split(' ')[0];
          var tableCurrent = $('div.table-box-contents table.table-custom:visible');

          var trData = {};
          trData.OrderNumber = commodityContent.OrderNumber;
          trData.ItemNumber = itemNumber;
          trData.ItemDescription = itemDescText;
          trData.StartVal = parseFloat($(tableRow).find('td:eq(1)').text());

          trData.RemainingVal = parseFloat(trData.StartVal - trData.ShippedVal);

this.PreSearchHistory = function(searchCriteria) {


		if (searchCriteria.WhereClauses !== null && searchCriteria.WhereClauses !== undefined) {
			if (searchCriteria.WhereClauses.length > 0) {
				var StartShipDate = null;
				var EndShipDate = null;
				var Shipper = "";
				for (var i = 0; i <= searchCriteria.WhereClauses.length - 1; i++) {
					if (searchCriteria.WhereClauses[i].FieldName == "Shipper") {
						Shipper = searchCriteria.WhereClauses[i].FieldValue;
					} else if (searchCriteria.WhereClauses[i].FieldName == "Shipdate" && StartShipDate == null) {
						StartShipDate = searchCriteria.WhereClauses[i].FieldValue;
					} else if (searchCriteria.WhereClauses[i].FieldName == "Shipdate" && StartShipDate != null) {
						EndShipDate = searchCriteria.WhereClauses[i].FieldValue;
					} else if (searchCriteria.WhereClauses[i].FieldName == "MiscReference1") {
						var miscReferenceObject = searchCriteria.WhereClauses[i];
						searchCriteria.WhereClauses.splice(i, 1);
						searchCriteria.WhereClauses.push({
							"FieldName": "ConsolidationTrackingNumber",
							"FieldValue": miscReferenceObject.FieldValue,
							"Operator": miscReferenceObject.Operator
						});
					}
				}



				var data = {
					UserContext: client.userContext,
					Data: "ShipDate=" + StartShipDate + ";" + "Shipper=" + Shipper
				}
				if (StartShipDate == EndShipDate) {
					sendSearchPackages("UserMethod", data, false)
				} else {
					var divResults = document.getElementById("divResults");
					divResults.innerHTML = "Select one day only to show package count!";
				}
			}
		}

	}
