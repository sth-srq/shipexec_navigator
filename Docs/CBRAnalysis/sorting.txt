      GetUniqueArrayFromString: function (inString, stringDelimiter = ',')
      {
        var trimArray = $.grep(inString.split(stringDelimiter), function (a)
        {
          return a.trim().length > 0;
        });

        /* Get rid of any duplicate array entries. */
        trimArray = trimArray.filter((e, i, a) => a.indexOf(e) === i);

        return trimArray;
      },

// GetUniqueCSVStringFromArray(inArray) // returns a CSV string with duplicates removed
      GetUniqueCSVStringFromArray: function (inArray)
      {
        /* Get rid of any duplicate array entries. */
        var uniqueArray = inArray.filter((e, i, a) => a.indexOf(e) === i);
        var newUniqueString = '';

        $(uniqueArray).each(function (a, b)
        {
          if (b.trim().length > 0)
          {
            newUniqueString += b.trim() + ',';
          }
        });

        return newUniqueString;
      },

                     return [{ name: '--- No Shipper Configured ---', disabled: true }];

                var customData = shipper.CustomData.filter(c => c.Key.toUpperCase() == "CUSTOMER");

                    if (customer == 'AMAZON')
                        return { name: "Amazon", value: customer, hasLoad: true, hasReturn: true, autoShip: true };
                    else if (customer == 'UPS')
                        return { name: "UPS", value: customer, hasCameraImage: true, hasFileImage: true, autoShip: false };

                    else if (customer == 'FANATICS')
                        return { name: "Fanatics", value: customer, hasReturn: true, hasCameraImage: true, autoShip: false };
                    else
                        return { name: `--- ${customer} ---`, disabled: true };
                });

                customers.sort(function(a, b) {
                    if (a.disabled && !b.disabled) return 1;
                    if (!a.disabled && b.disabled) return -1;
                    return a.name.localeCompare(b.name);
                });
                return [{ name: '--- Select Customer Option ---', selector: 'select[ng-model="vm.shipping.customer"]', disabled: true }].concat(customers);

MoveGoodToBox(li, destBox)
    {
      let vm = this.ClassViewModel;
      let that = this;
      let data = $(li).data();
      let destCC = this.packages[destBox - 1].CommodityContents || [];

      //find the box the item currently resides in
      //split this statement up over a few calls to make it easier to read
      let allGoods = this.packages.map(m => m.CommodityContents).flat();
      let item = allGoods.find(cc => cc.UniqueId == data.UniqueId);

      if (+item.PkgIndex == +destBox) { console.log('Item already exists in box.'); return false; }

      //remove the item from its current box
      this.packages[item.PkgIndex - 1].CommodityContents = this.packages[item.PkgIndex - 1].CommodityContents.filter(p => p.UniqueId != data.UniqueId);

      //add it to the new box
      item.PkgIndex = destBox;
      destCC.push(item);
      $(li).find('.pkg-index-text').fadeOut().promise().done(function(e) { $(this).text(destBox).fadeIn(); });

      //console.log(vm.currentShipment.Packages);
      $('li[heading="Goods"]:visible:first > a').trigger('click');
      $('#goods .btn-group').find('.btn:gt(0)').not('.active').first().trigger('click');

      //update the weights and values
      //this data will have changed by now so get a new reference to the object
      allGoods = this.packages.map(m => m.CommodityContents).flat();
      //the items array will have the CommodityContents object for every package in the shipment

      //sort by package index
      //update the weight and declared values for all the packages in the shipment
      allGoods = allGoods.sort((a, b) => { return a.PkgIndex - b.PkgIndex });
      $(allGoods).each(function(a, b)
      {
        let pkg = that.packages[b.PkgIndex - 1];

        if ((a == 0) || (b.PkgIndex != allGoods[a - 1].PkgIndex))
        {
          pkg.Weight.Amount = 0;
          pkg.DeclaredValueAmount.Amount = 0;
        }

        pkg.Weight.Amount += +b.PVTotalWeight;
        pkg.DeclaredValueAmount.Amount += +b.PVTotalValue;
      });
    }

  AddItemHtmlToContainer = (ListItem, ContainerIndex) =>
  {
    //need to put the items in the panels in the same order they appear in the table.
    //get all the row indexes from the panel list and sort the list descending
    const parent = $(this.H.PanelOpenContainersId).find(`.container-parent[container-index=${ContainerIndex}]`);
    let listitems = $(parent).find('ul.list-group').first().find('li:visible');
    let currentItems = $(listitems).map(function(i, val) { return $(this).data() }).get();
    let itemIndexes = $(currentItems).map((a, b) => +b.RowIndex).get().sort((c, d) => d - c);
    let ul = $(parent).find('ul.list-group').first();

    //get the last available index in the list items
    const lastIndex = itemIndexes.filter(i => +i < +$(ListItem).attr('row-index'))[0];

    //if there were no items with a lower index than the new item, add it to the top of the list
    if (!lastIndex && lastIndex !== 0)
    {
      $(ListItem).prependTo(ul);
      //$(parent).find('ul:first').prepend(ListItem);
    }
    else //find the list item that has a row-index less than the current list item and append the current list item after that
    {
      const prevItem = $(parent).find(`ul li:visible[row-index=${lastIndex}]`).first();
      //$(ul).append(ListItem).insertAfter(prevItem);
      $(ListItem).appendTo(ul).insertAfter(prevItem);
    }

    //show the list item gracefully
    $(ListItem).slideDown().css('display', 'block').show();
  }
