                        if (vm.clientContext.Site.Name == undefined)

                            var data = { CompanyId: vm.clientContext.Company.Id, SiteId: vm.clientContext.Site.Id };

                    }

this.PostLoad = function(value, shipmentRequest) {
		debugger;
		let errorMessage = shipmentRequest.PackageDefaults.ErrorMessage;
		
		if (errorMessage === undefined) errorMessage = "";
		if (errorMessage === null) errorMessage = "";

		if (shipmentRequest.PackageDefaults.Service !== undefined && shipmentRequest.PackageDefaults.Service.Symbol === "x") {
			showLoading(false);
			alert("No service selected!");
		}
		
		if (errorMessage !== undefined && errorMessage === 'validate') {
			//$("#ctl00_cphContent_left_tabs_consignee_tab_consignee_na_ab_ctrl_na_ctrl_validate_addr_btn").click();
			showLoading(false);
			return;
		}

		if (errorMessage !== undefined && errorMessage.substring(0, 4) === "pop:") {
			showLoading(false);
			alert(errorMessage.substring(4));
			return;
		}

		if (errorMessage !== undefined && errorMessage !== '') {
			showLoading(false);
			alert(errorMessage);
			let errors = ["not found", "already shipped", "too many"];
			if (errors.some(errorText => errorMessage.includes(errorText))) {
				// open Load Shipment dialog
				this.vm.newShipment();
			}
			return;
		}

		if (shipmentRequest.PackageDefaults.Consignee.Country === 'US') {

			this.vm.shipMode = 0;

		//getWeight();

		if (shipmentRequest.PackageDefaults.MiscReference5 !== null && shipmentRequest.PackageDefaults.MiscReference5 === 'autoship') {
			this.vm.checkShip();
			return;
		}

		// select all services before rate
		setTimeout(function() {
			// check Select All after delay to allow UI update
			vmTemp.isAllServicesSelected = true;
			vmTemp.selectAllServices();
			vmTemp.rate();
		}, 100);
		
	};

        var t = parseInt(timeoutInSeconds * 5);

        return new Promise((resolve, reject) =>
        {
          var checkExist = setInterval(function()
          {
            i++;

            if ($(element).length > 0)
            {
              /* Stop the timer. */
              clearInterval(checkExist);
              resolve(1);
            }

            /* The loop has timed out. */
            if (i == t)
            {
              clearInterval(checkExist);
              reject(new Error('Operation timed out waiting for ' + element + ' to attach to the DOM.'));
            }

          }, 200);

        });
      },

            function getNextMonthYearInFrench(dateStr) {
                var monthNamesFr = ['Janvier', 'Février', 'Mars', 'Avril', 'Mai', 'Juin', 'Juillet', 'Août', 'Septembre', 'Octobre', 'Novembre', 'Décembre'];
                var splitDate = dateStr.split(' ');
                var month = splitDate[0];
                var year = splitDate[1];
            
                var currentMonthIndex = monthNamesFr.indexOf(month);
                
                // Check if the current month is December.
                if (currentMonthIndex === 11) {
                    // Increment the year and reset the month to January.
                    year++;
                    month = monthNamesFr[0];
                } else {
                    // Get the next month.
                    month = monthNamesFr[currentMonthIndex + 1];
                }
                
                return month + ' ' + year;
            }


            function getPreviousMonthYearInFrench(dateStr) {
                var monthNamesFr = ['Janvier', 'Février', 'Mars', 'Avril', 'Mai', 'Juin', 'Juillet', 'Août', 'Septembre', 'Octobre', 'Novembre', 'Décembre'];

                // Check if the current month is January.
                if (currentMonthIndex === 0) {
                    // Decrement the year and set the month to December.
                    year--;
                    month = monthNamesFr[11];
                } else {
                    // Get the previous month.
                    month = monthNamesFr[currentMonthIndex - 1];
                }
                
                return month + ' ' + year;
            }

// selectCanadaForAllCountrySelects() // returns undefined
    function selectCanadaForAllCountrySelects() {
        // Check if the flag is already true, which means we no longer need to change the selects
        if(canadaSelected || !isSelectedValueInShippersArray()) {
          return; // Exit the function if Canada has been selected in all dropdowns or if the site is not from Canada
        }
        
        // Get all select elements with the name 'Country'
        var selects = document.querySelectorAll('select[name="Country"]');
        var allSelected = true; // To keep track if all selects have been set to Canada
      
        // Iterate through the NodeList
        selects.forEach(function(select) {
          // Only set the value to 'string:CA' if it's not already set
          if (select.value !== 'string:CA') {
            select.value = 'string:CA';
            allSelected = false; // Not all selects have been set to Canada
      
            // Trigger change event for AngularJS
            if ('createEvent' in document) {
              var evt = document.createEvent('HTMLEvents');
              evt.initEvent('change', false, true);
              select.dispatchEvent(evt);
            } else {
              select.fireEvent('onchange');
            }
          }
        });
      
        // If all selects have been set to Canada, set the flag to true
        if (allSelected) {
          canadaSelected = true;
        }
      }

        function loadWeight() {
            if (this.profile.ScaleConfiguration === undefined) return;
            var scale = this.profile.ScaleConfiguration;
            var url = client.config.ClientSupportServiceUrl + '/api/Scale/Read';
            var data = {
                'PortType': scale.ScaleDefinition.PortType,
                'Port': scale.Port,
                'UsbVid': scale.ScaleDefinition.UsbVid,
                'UsbPid': scale.ScaleDefinition.UsbPid,
                'RequestCommand': scale.ScaleDefinition.RequestCommand,
                'BaudRate': scale.ScaleDefinition.BaudRate,
                'StopBits': scale.ScaleDefinition.StopBits,
                'Parity': scale.ScaleDefinition.Parity,
                'DataBits': scale.ScaleDefinition.DataBits,
                'RequestTimeout': scale.ScaleDefinition.RequestTimeout,
                'HostAddress': scale.HostAddress,
                'RegexPattern': scale.ScaleDefinition.RegexPattern,
                'DeviceParser': scale.ScaleDefinition.DeviceParserType,
                'DeviceName': scale.ScaleDefinition.Name,
                'Handshake': scale.ScaleDefinition.Handshake,
                'RetryCount': scale.ScaleDefinition.RetryCount,
                'RetryDelay': scale.ScaleDefinition.RetryDelay
            };
            var ret = $.post({ url: url, data: data, async: false }).responseJSON;
            if (ret.ErrorCode === 0) {
                this.currentShipment.Packages[this.packageIndex].Weight = {
                    "Amount": ret.Weight,
                    "Units": ret.WeightUnit === null ? "LB" : ret.WeightUnit.toUpperCase()
                };

                //if (ret.HasDimensions) {

                    //};
                //}

            } else {
                client.alert.Danger(this.translations.Common.ErrorCode + " : " + ret.ErrorCode + "  " + ret.ErrorMessage);
            }            
        }
    }

    function ChromeScrollIntoView(element, value) {
        if (!isIE()) {
            var interval = setInterval(function () {
                if (element.val() == value) {
                    clearInterval(interval);
     
                    var optionTop = element.find('[value="' + value + '"]').offset().top;
                    var selectTop = element.offset().top;
     
                    element.scrollTop(element.scrollTop() + (optionTop - selectTop));
                }
            }, 50);
        }
    }
