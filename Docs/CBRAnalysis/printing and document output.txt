            function selectDownloadFile(event) {
                this.menu.guide.selectedDownloadFile = event.target.innerText;
                if (this.menu.guide.inputFile.selected) {
                    if (this.menu.guide.selectedDownloadFile.slice(-3) == this.menu.guide.inputFile.selected.name.slice(-3))
                        updateFile();
                } else {
                    var payload = { Action: 'downloadFile', Data: this.menu.guide.selectedDownloadFile };
                    var data = { UserContext: undefined, Data: JSON.stringify(payload) };
                    thinClient.httpClient('UserMethod', data).then(function (ret) {
                        if (ret.ErrorCode === 0) {
                            var data = JSON.parse(atob(ret.Data));

                            let linkData = `data:${data.fileType};base64,${data.encodedFile}`;
                            let downloadLink = document.createElement("a");
                            downloadLink.style.display = "none";
                            downloadLink.download = data.fileName;
                            downloadLink.href = linkData;
                            document.body.appendChild(downloadLink);
                            downloadLink.click();
                            document.body.removeChild(downloadLink);
                        }
                    });
                }

                function updateFile() {
                    var reader = new FileReader();
                    reader.onload = function () {
                        var request = {
                            Key: vm.shipping.menu.guide.selectedDownloadFile,
                            Value: JSON.stringify({
                                fileType: vm.shipping.menu.guide.inputFile.selected.type,
                                fileName: vm.shipping.menu.guide.inputFile.selected.name,
                                encodedFile: this.result.split(',')[1]
                            })
                        }
                        var payload = { Action: 'updateFile', Data: request };
                        var data = { UserContext: undefined, Data: JSON.stringify(payload) };
                        thinClient.httpClient('UserMethod', data).then(function (ret) {
                            if (ret.ErrorCode !== 0)
                                vm.notificationService.notifyDanger(ret.ErrorCode + " - " + ret.ErrorMessage);
                        }).catch(function (response) {
                            vm.notificationService.notifyDanger(vm.translations.Common.ErrorConnectingWCFService + ": " + response.status + " " + response.statusText);
                        })
                        vm.shipping.menu.guide.inputFile.selected = null;
                        $(vm.shipping.menu.guide.inputFile.selector).val(null);
                        vm.thinClient.loadData.show();

                    }
                    if (vm.shipping.menu.guide.inputFile.selected) reader.readAsDataURL(vm.shipping.menu.guide.inputFile.selected);
                }
            }

    function downloadPdf(pdfDocuments) {      
      let pdfArray = pdfDocuments.PdfData;
      let documentName = pdfDocuments.DocumentName;
      let file = {};
      console.log(pdfArray.length);
      for (let count = 0; count < pdfArray.length; count++) {
            if (navigator.userAgent.indexOf("MSIE ") > -1 || navigator.userAgent.indexOf("Trident/") > -1) {
              let byteCharacters = atob(pdfArray[count]);
              let byteNumbers = new Array(byteCharacters.length);
                for (let i = 0; i < byteCharacters.length; i++) {
                    byteNumbers[i] = byteCharacters.charCodeAt(i);
                }
                let byteArray = new Uint8Array(byteNumbers);
                file = new Blob([byteArray], { type: 'application/pdf' });
                window.navigator.msSaveOrOpenBlob(file, documentName + ".pdf");
            } else {
                let documentAfterConversion = base64ToArrayBuffer(pdfArray[count]);
                let a = window.document.createElement("a");
                file = new Blob([documentAfterConversion], { type: 'application/pdf' });
                let fileUrl = URL.createObjectURL(file);
                a.href = fileUrl;
                a.download = documentName + ".pdf";
                document.body.appendChild(a);
                a.click();
                document.body.removeChild(a);
            }          
      }
  }

// function downloadFile(exportFileName, encodedString) // returns undefined
function downloadFile(exportFileName, encodedString) {
    var decodedResult = atob(encodedString); // decode a base-64 encoded string
    var csvFile = new Blob([decodedResult], { type: "text/csv" });

    var ie = navigator.userAgent.match(/MSIE\s([\d.]+)/),
        ie11 = navigator.userAgent.match(/Trident\/7.0/) && navigator.userAgent.match(/rv:11/),
        ieEdge = navigator.userAgent.match(/Edge/g),
        ieVer = (ie ? ie[1] : (ie11 ? 11 : (ieEdge ? 12 : -1)));

    exportFileName = exportFileName + '.csv';

    if (ieVer > -1) {
        window.navigator.msSaveBlob(csvFile, exportFileName);

    } else {
        var downloadLink = document.createElement("a");
        downloadLink.download = exportFileName;
        downloadLink.href = window.URL.createObjectURL(csvFile);
        downloadLink.style.display = "none";
        document.body.appendChild(downloadLink);
        downloadLink.click();
        document.body.removeChild(downloadLink);
    }
}

    function ValidateCostCenterAccount() {
        
        var customData = vm.profile.UserInformation.Address.CustomData;
        
        var returnValue = true;
        if (customData.length > 0) {
          
            var costCenters = getValueByKey("CustomData1", customData)

            var costCenterCount = 0;
            if(costCenters != "")

            var accountsCount = 0;
            if(accounts != "")

    function getShippersFromCostCenter() {
        var returnData = [];
        var customData = vm.profile.UserInformation.Address.CustomData;
        var shippers = vm.profile.Shippers;
        returnData = shippers;
        if (customData.length > 0) {
            var customShippers = getValueByKey("CustomData3", customData);
            var customShipperData = customShippers.split(",");
            if (customShipperData.length > 0) {
                returnData = shippers.filter(item => customShipperData.some(cs => cs.toLowerCase()  === item.Symbol.toLowerCase()));
            }
        }

        return returnData;
    }

    function getCostCenterByShipper(selectedShipper) {
        
        var customData = vm.profile.UserInformation.Address.CustomData;
       
        var shippers = getValueByKey("CustomData3", customData);
       
        var shipperData = shippers.split(",");
        var index = shipperData.indexOf(selectedShipper);
        var costCenters = getValueByKey("CustomData1", customData);
        var costCenterData = costCenters.split(",");
        var costCenter = ''
        if (costCenterData.length > 0 && costCenterData.length >= index) {
            costCenter = costCenterData[index].trim();
        }
        return costCenter;
    }

    function getEmployeeId() {
        var customData = vm.profile.UserInformation.Address.CustomData;
        var employeeId = getValueByKey("CustomData2", customData).trim();
        return employeeId;
    }

    function checkElmLoaded(selector) {
        return new Promise(resolve => {
            if (document.querySelector(selector)) {
                return resolve(document.querySelector(selector));
            }
            const observer = new MutationObserver(mutations => {
                if (document.querySelector(selector)) {
                    resolve(document.querySelector(selector));
                    observer.disconnect();
                }
            });
            observer.observe(document.body, {
                childList: true,
                subtree: true
            });
        });
    }

    function addShipperRelease() {

                "</div>"
            $('[ng-model="proofofdeliverypane"]').children().first().after(htmlShipperRelease);

            }

    }

    function getValueByKey(key, arr) {
        for (var index in arr) {

        }
        return '';
