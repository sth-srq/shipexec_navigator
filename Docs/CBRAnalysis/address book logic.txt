// CheckForMatchingConsignee(CustomerOrderObject) // returns boolean indicating whether the customer order consignee address matches the current shipment consignee
      CheckForMatchingConsignee: function (CustomerOrderObject)
      {
        try
        {
          var con = $('name-address[nameaddress="vm.currentShipment.PackageDefaults.Consignee"]');

          let a = {
            //conCompany: $(con).find('input[name="company"]').val().trim().toUpperCase().replaceAll(' ','') ,
            conAddress1: $(con).find('input[name="address1"]').val().trim().toUpperCase(),
            conCity: $(con).find('input[name="city"]').val().trim().toUpperCase(),
            conState: $(con).find('input[name="stateProvince"]').val().trim().toUpperCase(),
            conPostalCode: $(con).find('input[name="postalCode"]').val().trim().substring(0, 5)
          };

          let b = {
            //conCompany: CustomerOrderObject.shipAddress1.CONSIGNEE_COMPANY.trim().toUpperCase().replaceAll(' ', ''),
            conAddress1: CustomerOrderObject.shipAddress1.trim().toUpperCase(),
            conCity: CustomerOrderObject.shipCity.trim().toUpperCase(),
            conState: CustomerOrderObject.shipState.trim().toUpperCase(),
            conPostalCode: CustomerOrderObject.shipZipCode.trim().substring(0, 5)
          };

          /* Remove all the spaces in the strings so we get a good comparison, not one that's thrown off by an extra space. */
          let aa = JSON.stringify(a).replaceAll(' ', '');
          let bb = JSON.stringify(b).replaceAll(' ', '');

          /* Quick comparison to see if the strings match. */
          if (aa != bb)
          {
            Logger.Log({ Source: 'CheckForMatchingConsignee()', Message: 'Could not add order ' + CustomerOrderObject.orderNumber + ' (' + CustomerOrderObject.orderId + '). Primary order data to follow.', Data: a });
            Logger.Log({ Source: 'CheckForMatchingConsignee()', Message: 'Could not add order ' + CustomerOrderObject.orderNumber + ' (' + CustomerOrderObject.orderId + '). Subsequent order data to follow.', Data: b });
          }

          return (aa == bb);
        }
        catch (error)
        {
          Logger.Log({ Source: 'CheckForMatchingConsignee()', Error: error });
        }
      }

// UpdateReferencesWithOrderNumbers(OrderNumber) // returns undefined after updating shipper reference and misc reference fields
      UpdateReferencesWithOrderNumbers: function (OrderNumber)
      {
        var inputShipperReference = $('input[type=text][ng-model="vm.currentShipment.Packages[vm.packageIndex].ShipperReference"]');
        var inputMiscReference07 = $('input[type=text][ng-model="vm.currentShipment.Packages[vm.packageIndex].MiscReference7"]');
        var shipperReferenceVal = $(inputShipperReference).val() + ',' + OrderNumber;

        var newUniqueString = Tools.GetUniqueCSVStringFromString(shipperReferenceVal);

        /* Set these input values.  ShipperReference is used by the customer, MiscReference7 is used by the SBR to update each record post-ship. */
        /* ShipperReference ONLY keeps the first order number when it goes to PostShip, before the comma, MiscReference7 keeps the entire string. */
        /* On the server-side, if MiscReference7 contains a comma, each item in the csv list gets posted back.  If MiscReference7 is empty, the ShipperReference is used instead for a single post-back. */
        $(inputShipperReference).val(newUniqueString);
        $(inputMiscReference07).val(newUniqueString);
      }

// GetCustom(fieldName, customDataType) // returns string
GetCustom(fieldName, customDataType) {
      var returnValue = "";
      var customData = null;
      if (customDataType === "User") customData = this.ShipmentRequest.PackageDefaults.OriginAddress.CustomData;
      if (customDataType === "To") customData = this.ShipmentRequest.PackageDefaults.Consignee.CustomData;

      if (customData) {
         customData.forEach(function (customField) {
            if (fieldName === customField.Key) returnValue = customField.Value;
         })
      }
      return returnValue;
   }

// this.PreShip(shipmentRequest, userParams) // returns undefined
this.PreShip = function(shipmentRequest, userParams) {    
    // We will require the actual shipment to send weight and dims in kg/cm
    // However, because US Vendors are sending the product the client gui needs to be in Lb/In.
    
    // debugger;
    
    totalShipWeight = 0;
    totalVolumetricResult = 0;
    
    vm.currentShipment.Packages.forEach(pkg => {
        totalShipWeight += Number(pkg.Weight.Amount);
        totalVolumetricResult += ((Number(pkg.Dimensions.Length).toFixed(2)*Number(pkg.Dimensions.Width).toFixed(2)*Number(pkg.Dimensions.Height)).toFixed(2))/166;
    });
              
    totalShipWeight = Number(totalShipWeight).toFixed(2);
    console.log(totalShipWeight);
    console.log(totalVolumetricResult);
    if (totalShipWeight > 150)
    {
        alert("Total Shipment Weight Exceeds Shipment Weight Limit of 150lbs")
        throw new Error("Shipping halted due to Over Maximum Weight Rule " + totalShipWeight);
    }
    
    if (totalVolumetricResult > Number(maxvolumetric))
    {
        alert("Total Shipment Weight Exceeds Total Shipment Volumetric Weight Limit of " + Number(maxvolumetric))
        throw new Error("Shipping halted due to Total Shipment Volumetric Weight Rule " + Number(totalVolumetricResult).toFixed(2));
    }
    
    //kg = lbs*0.453592    
    vm.currentShipment.Packages.forEach(pkg => {
         if (pkg.Weight.Units === "LB"){
             pkg.Weight.Amount = Number(pkg.Weight.Amount * 0.453592).toFixed(2);
             pkg.Weight.Units = "KG";
         }
         
    });
    
  
    //cm=in/0.39370
    vm.currentShipment.Packages.forEach(pkg => {
         if (pkg.Dimensions.Units === 0){
             pkg.Dimensions.Length = Number(pkg.Dimensions.Length/0.39370).toFixed(2)
             pkg.Dimensions.Width = Number(pkg.Dimensions.Width/0.39370).toFixed(2)
             pkg.Dimensions.Height = Number(pkg.Dimensions.Height/0.39370).toFixed(2)
             pkg.Dimensions.Units = 1;
       }
         
    });
     
   if(Number(shipmentRequest.Packages[0].DeclaredValueCustoms.Amount).toFixed(2) > 2500 ){
          if(shipmentRequest.Packages[0].ExportInformationCode === null){
              shipmentRequest.Packages[0].ExportInformationCode = "OS";
          }
          if(shipmentRequest.Packages[0].AesTransactionNumber === null){
              shipmentRequest.Packages[0].SedMethod = 2;
              $('input[type=text][ng-model="shipmentRequest.Packages[this.vm.packageIndex].AesTransactionNumber"]').focus();
              throw new Error('Missing AES/ITN Number');
          } 
          
          if(shipmentRequest.Packages[0].AesTransactionNumber !== null){
              shipmentRequest.Packages[0].Comments = shipmentRequest.Packages[0].Comments + "\n ITN/AES Filer " + shipmentRequest.Packages[0].AesTransactionNumber;
             
          }
      }
};

        //debugger;

        for (var item of customData3) {
            var symbol = getServiceSymbol(item.toLowerCase());
            for (var service of profile.Services) {
                if (service.Symbol === symbol) {
                    //service.Name = service.Name.replace('(UPS Adapter)','').trim();
                    filterServices.push(service);
                    break;
                }
            }
        }
        return filterServices;
        function getServiceSymbol(code) {
             if (code == '2dn') return 'UPSAPI.UPS.2DA'
            else if (code == '3dd') return 'UPSAPI.UPS.3DS'
            else if (code == 'exp') return 'UPSAPI.UPS.EXPEDITED'
            else if (code == 'ess') return 'UPSAPI.UPS.EXPRESS'
            else if (code == 'nda') return 'UPSAPI.UPS.NDA'
            else if (code == 'dae') return 'UPSAPI.UPS.NDAEAM'
            else if (code == 'grn') return 'UPSAPI.UPS.GND'
            else if (code == 'uwe') return 'UPSAPI.UPS.EXPEDITED'
            else if (code == 'wep') return 'UPSAPI.UPS.EXPRESSSP'
            else if (code == 'swe') return 'UPSAPI.UPS.EXPRESS'
            else if (code == 'uws') return 'UPSAPI.UPS.SAVER'
            else if (code == 'std') return 'UPSAPI.UPS.STANDARD'
        }
        
    }
    
    if(dims.includes(',')){
        var dimensions = dims.split(',');
        
        this.vm.currentShipment.Packages[0].Weight.Amount = Number(dimensions[0]);

    this.GetAddressBookEntries = function (addressBookName, context) {
        var searchCriteria = { WhereClauses: [{ FieldName: "Name", FieldValue: addressBookName, Operator: 0 }] };
        var data = { SearchCriteria: searchCriteria, CompanyId: context.CompanyId, SiteId: context.SiteId, UserId: context.UserId };
        var response = this.thinClientAPIRequest('GetAddressBooks', data, false).responseJSON;
        if (response.TotalRecords == 0) {
            var addressBookEntry = { Code: 'Error: AddressBook "' + addressBookName + '" Not Found' };
            return [addressBookEntry];
        } else {
            var addressBookId = response.AddressBooks[0].Id;
            searchCriteria = { OrderByClauses: [{ FieldName: "Code", Direction: "asc", FieldType: 3 }] };
            data = { AddressBookIds: [addressBookId], SearchCriteria: searchCriteria, CompanyId: context.CompanyId, SiteId: context.SiteId, UserId: context.UserId };
            response = this.thinClientAPIRequest('GetAddressBookEntries', data, false).responseJSON;
            if (response.TotalRecords > 0) {
                return response.AddressBookEntries;
            }
        }
    }

      {
        try
        {
          /* Append the shipper reference with each item in the select list in csv format. */
          /* Once this is done, change the SSBR on post load to loop through each option in the csv list. */
          var inputShipperReference = $('input[type=text][ng-model="vm.currentShipment.Packages[vm.packageIndex].ShipperReference"]');
          var shipperReferenceVal = $(inputShipperReference).val();

          /* If the user closes the dialog box, then opens it back up, we don't want to check the same orders more than once so only process what was entered into the dialog box this time. */
          var ordersToQuery = '';

          /* Loop through all the orders in the select box and determine if they are already in the list. */
          $('#selectConsolidatedOrders').find('option').each(function (i, v)
          {
            /* Remove the spaces before the commas for comparison. */
            var arrayOrders = shipperReferenceVal.split(",").map(function (item)
            {
              return item.trim();
            });

            /* If the current option value is NOT in the list within the Shipper Reference field, add it. */
            if ($.inArray(v.value.trim(), arrayOrders) == -1)
            {
              ordersToQuery += v.value.trim() + ',';
              //shipperReferenceVal += ',' + v.value;
            }

            /* Remove the option once it is added to the input box. */
            $(this).remove();

          }).promise().done(function ()
          {
            /* Done creating the list of orders to get from the server which is now held in the ordersToQuery variable. */

            /* Hide the div. */
            $('#divModalConsolidateShipments').hide();

            /* Make sure there are orders in the list. */
            if (ordersToQuery.length > 0)
            {
              PackVerify.GetOrderInformationFromCustomer(ordersToQuery);
            }
            else
            {
              PackVerify.ShowErrorAlert('ERROR: No unique orders to pull from the server.  All consolidated orders have already been added to this shipment.');
            }
          });
        }
        catch (error)
        {
          Logger.Log({ Source: 'SaveConsolidatedOrderNumbers()', Error: error });
        }
      }

    function getAddressBookEntries(addressBookName, context) {
        var searchCriteria = { WhereClauses: [{ FieldName: "Name", FieldValue: addressBookName, Operator: 0 }] };
        var data = { SearchCriteria: searchCriteria, CompanyId: context.CompanyId, SiteId: context.SiteId, UserId: context.UserId };
        debugger;
        //var response = ajaxPost('GetAddressBooks', data);
        var response = thinClientAPIRequest('GetAddressBooks_Shared', data, false);
        debugger;
        if (response.TotalRecords == 0) {
            var addressBookEntry = { Code: 'Error: AddressBook "' + addressBookName + '" Not Found' };
            return [addressBookEntry];
        } else {
            var addressBookId = response.AddressBooks[0].Id;

            searchCriteria = { OrderByClauses: [{ FieldName: "Code", Direction: "asc", FieldType: 3 }] };
            data = { AddressBookIds: [addressBookId], SearchCriteria: searchCriteria, CompanyId: context.CompanyId, SiteId: context.SiteId, UserId: context.UserId };
            //response = ajaxPost('GetAddressBookEntries', data);
            response = thinClientAPIRequest('GetAddressBookEntries_Shared', data, false);
            debugger;
            if (response.TotalRecords > 0) {
                return response.AddressBookEntries;
            }
        }
    }

  {

    let numOfItems = $(li).find('.num-of-items-in-pkg');
    let remItems = $(li).find('.goods-quantity > strong');

    if (!$(numOfItems).val()) { return; }

    $(numOfItems).attr('disabled', 'disabled');

    //to get the number of rows that have already been assigned, count the list items with the goods-key

  ApplyCommodityAssignments()
  {
    let that = this;

    //every list item will have a pkg-index attr, get that highest value to determine how many pkgs we need
    //don't use pkg-status.length because that will include multiple commodities and the end result will be wrong
    let packed = $('#listItemContainer').find('li[pkg-status=packed]');

    //first get all the pkg-index attributes, then get the highest number and add one
    let pkgs = $(packed).map(function(x, value) { return +$(this).attr('pkg-index') }).toArray();
    let pkgCount = Math.max(...pkgs) + 1;

    while (this.packages.length < pkgCount)
    {
      //the vm.addPackage isn't firing, just click the button
      $('button[ng-click="vm.addPackage();"]').trigger('click').promise().done(function(e) { });
    }

    //loop the pkg count, get the commodities for that pkg, write the objects to the vm CommodityContents object
    for (let p = 0; p < pkgCount; p++)
    {
      let w = $.grep(this.working, function(item, i) { return item.packageIndex == p });
      this.packages[p].CommodityContents = w;
    }

    //click the General tab
    $('li[heading=General]:visible a').first().click();
    this.HideAssignGoodsPane();

    //after we are done, move back to the first package in the shipment
    while ($('#spanCurrentPackageIndex').text() !== '1')
    {
      $('button[ng-click="vm.previousPackage();"]').first().click();
    }

  }

  }

        var searchCriteria = { WhereClauses: [{ FieldName: "Name", FieldValue: addressBookName, Operator: 0 }] };
        var data = { SearchCriteria: searchCriteria, CompanyId: context.CompanyId, SiteId: context.SiteId, UserId: context.UserId };
        var response = this.thinClientAPIRequest('GetAddressBooks_Shared', data, false);
        if (response.ErrorCode != 0) {
            client.alert.Danger(response.ErrorMessage);
        } else {
            if (response.TotalRecords == 0) {
                client.alert.Danger('Address Book: "' + addressBookName + '" does not exist');                
            } else {
                return response.AddressBooks[0];
            }
        }
    }

    function checkOriginConsigneeCountry() {
        var shipFromCountry = getObjectValue(vm.currentShipment.PackageDefaults.OriginAddress.Country);
        var shipToCountry = getObjectValue(vm.currentShipment.PackageDefaults.Consignee.Country);

        /* If we have no shipToCountry OR the shipFromCountry is either US or CA, disable the elements, return out. */
        if (!shipToCountry | shipFromCountry == "US" | shipFromCountry == "CA") {
            DisableThirdPartyBillingElements();
            return false;
        }

        if (shipFromCountry != shipToCountry) {
            /* If the shipFromCountry does NOT equal the shipToCountry then ship Third Party. */
            SetThirdPartyBilling(vm.currentShipment);
        }
        else {
            /* Now check to see if both the origin and destination countries are members of the EU. */
            if ((IsMemberOfEu(shipFromCountry)) && (IsMemberOfEu(shipToCountry))) {
                SetThirdPartyBilling(vm.currentShipment);
            }

            //ShipThirdPartyBilling((IsMemberOfEu(shipFromCountry)) && (IsMemberOfEu(shipToCountry)));
        }

        if (shipFromCountry != "US") {
            $('select[ng-model="vm.currentShipment.Packages[vm.packageIndex].Weight.Units"]').val('string:KG').change();
            $('select[ng-model="vm.currentShipment.Packages[vm.packageIndex].Dimensions.Units"]').val('number:1').change();
        }
    }
