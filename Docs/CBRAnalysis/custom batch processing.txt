          Logger.Log({ Source: 'SaveConsolidatedOrderNumbers()', Error: error });
        }
      }

// ProcessDataFromCustomer(jsonData) // returns undefined
      ProcessDataFromCustomer: function (jsonData)
      {
        try
        {
          $(jsonData).each(function (a, b)
          {
            /* For testing, this is set to '!== true' so it will return order data.  This needs to be set to '=== true' for any other case. */
            if (ConsolidatedOrders.CheckForMatchingConsignee(b) == true)
            {
              var currentPackage = Tools.GetCurrentPackage();
              if (!currentPackage)
              {
                Tools.ShowErrorAlert('ERROR:  Could not add commodities to package for order ' + b.orderNumber + '.  See error console for details.');
                return false;
              }
              ConsolidatedOrders.UpdateReferencesWithOrderNumbers(b.orderNumber);
            }
            else
            {
              Tools.ShowErrorAlert('ERROR 500:  Could not add commodities for order ' + b.orderNumber + '.  Address mismatch.');
            }
          });
        }
        catch (error)
        { 
          console.log(error);
        }
      }

// UpdateReferencesWithOrderNumbers(OrderNumber) // returns undefined
      UpdateReferencesWithOrderNumbers: function (OrderNumber)
      {
        var inputShipperReference = $('input[type=text][ng-model="vm.currentShipment.Packages[vm.packageIndex].ShipperReference"]');
        var inputMiscReference07 = $('input[type=text][ng-model="vm.currentShipment.Packages[vm.packageIndex].MiscReference7"]');
        var shipperReferenceVal = $(inputShipperReference).val() + ',' + OrderNumber;

        var newUniqueString = Tools.GetUniqueCSVStringFromString(shipperReferenceVal);

        /* Set these input values.  ShipperReference is used by the customer, MiscReference7 is used by the SBR to update each record post-ship. */
        /* ShipperReference ONLY keeps the first order number when it goes to PostShip, before the comma, MiscReference7 keeps the entire string. */
        /* On the server-side, if MiscReference7 contains a comma, each item in the csv list gets posted back.  If MiscReference7 is empty, the ShipperReference is used instead for a single post-back. */
        $(inputShipperReference).val(newUniqueString);
        $(inputMiscReference07).val(newUniqueString);
      }

    this.handleItemSkuTextChanged = function(event) {

        var newitemsku = this.Packages[0].MiscReference10;
		if( newitemsku.trim().length === 0 )
			return;

        //debugger;
        
		//alert("Item number: " + orderItemList[1].itemcode + "  qtyordered: " + orderItemList[1].qtyordered + " qtyshipped: " + orderItemList[1].qtyshipped);

		// Check for item code SHIPCOMPLETE indicating that this order is complete
		if( newitemsku == "SHIPCOMPLETE") {

            vm.currentShipment.Packages[vm.packageIndex].MiscReference5 = "SHIPCOMPLETE";
		    //var pkg = sox_current_shipment.packages[sox_current_package_idx];
			//pkg.misc_reference_5 = "SHIPCOMPLETE";
			//$("#item_sku_txt").val("");			
            vm.currentShipment.Packages[vm.packageIndex].MiscReference10 = "";
			vm.checkShip();
			return;
		}

		// Check for SHIP indicating that this box is complete but not the whole order
		if( newitemsku == "SHIP") {
            vm.currentShipment.Packages[vm.packageIndex].MiscReference5 = "";		    
		    //var pkg = sox_current_shipment.packages[sox_current_package_idx];
			//pkg.misc_reference_5 = "";
			//$("#item_sku_txt").val("");			
            vm.currentShipment.Packages[vm.packageIndex].MiscReference10 = "";			
			vm.checkShip();
			return;
		}

		// Pad the input sku to 14 digits with leading zeros
		var skulen = newitemsku.length;
		var newitemskupad = "";
		if( skulen == 6)
			newitemskupad = '00000000' + newitemsku;
		else if( skulen == 7)
			newitemskupad = '0000000' + newitemsku;
		else if( skulen == 8)
			newitemskupad = '000000' + newitemsku;
		else if( skulen == 9)
			newitemskupad = '00000' + newitemsku;
		else if( skulen == 10)
			newitemskupad = '0000' + newitemsku;
		else if( skulen == 11)
			newitemskupad = '000' + newitemsku;
		else if( skulen == 12)
			newitemskupad = '00' + newitemsku;
		else if( skulen == 13)
			newitemskupad = '0' + newitemsku;
		else
			newitemskupad = newitemsku;
			
		newitemsku = newitemskupad;

		//alert("newitemsku: " + newitemsku);

        //alert("orderItemList length: " + orderItemList.length);
        
        //alert("orderItemList itemsku: " + orderItemList[0].itemsku);
       
       //debugger;
       
		// Verify item sku is on this order
		var itemFound = false;
		var orderItemListCount = orderItemList.length;
		for (var i = 0; i < orderItemListCount; i++) {
		    if( itemFound === true)
		        break;
			var skuarr = orderItemList[i].itemsku.split(",");
			for(var j = 0; j < skuarr.length; j++) {
				if( skuarr[j] == newitemsku ) {
					itemFound = true;
					break;
				}
			}
		}
		
		//alert("itemFound: " + itemFound);
		
		if( itemFound === false) {
			ShowAlert("Pack Error","This SKU (" + newitemsku + ") does not belong in this shipment. Please remove it before proceeding.");
            $("#mr10").css("background-color","Red");        	    				
            vm.currentShipment.Packages[vm.packageIndex].MiscReference10 = "";
            angular.element('[ng-model="vm.currentShipment.Packages[vm.packageIndex].MiscReference10"]').focus();    	                                
            return;			
		}

		// Search for itemcode
		itemFound = false;
		var fullyPacked = false;
		var svSkuCode = "";
		var orderItemListCount = orderItemList.length;
		for (var i = 0; i < orderItemListCount; i++) {
			if( itemFound == true)
				break;
			
			var nextitem = false;
			var skuarr = orderItemList[i].itemsku.split(",");
			for(var j = 0; j < skuarr.length; j++) {
				if( skuarr[j] == newitemsku ) {
					if( orderItemList[i].qtyshipped == orderItemList[i].qtyordered ) {
						fullyPacked = true;
						nextitem = true;
						break;
					} else {
						itemFound = true;
						fullyPacked = false;
						var tmpqtyshipped = Number(orderItemList[i].qtyshipped);
						var tmptotqtyshipped = Number(orderItemList[i].totqtyshipped);
						var tmpqtyordered = Number(orderItemList[i].qtyordered);
						if( (tmpqtyshipped + tmptotqtyshipped) < tmpqtyordered) {
							var calcqty1 = tmpqtyshipped + 1 + tmptotqtyshipped;
							var calcqty2 = tmpqtyordered - tmpqtyshipped - tmptotqtyshipped;
							/*alert("Calc quantity 1: " + calcqty1 + " qtyordered: " + tmpqtyordered);*/
							/*alert("Calc quantity 2: " + calcqty2);*/
							if( calcqty1 <= tmpqtyordered) {
								var tmpnewqty = tmpqtyshipped + 1;
								orderItemList[i].qtyshipped = tmpnewqty;
								orderItemList[i].modified = true;
								svSkuCode = newitemsku;
								/*alert("Item added to box ok");*/
							} else {
							    ShowAlert("Pack Error","Scanned quantity for SKU (" + newitemsku + ") would exceed order quantity. Please remove extras from the shipment and try again.");
								$("#mr10").css("background-color","Red");        	    				
                                vm.currentShipment.Packages[vm.packageIndex].MiscReference10 = "";
                                angular.element('[ng-model="vm.currentShipment.Packages[vm.packageIndex].MiscReference10"]').focus();    	                                
								return;						
							}
						} else {
							ShowAlert("Pack Alert","This item has been fully packed");
							$("#mr10").css("background-color","Red");        	    				
                            vm.currentShipment.Packages[vm.packageIndex].MiscReference10 = "";
                            angular.element('[ng-model="vm.currentShipment.Packages[vm.packageIndex].MiscReference10"]').focus();    	                                
							return;						
						}
					}
				}
			}
			if( nextitem == true)
				continue;
		}

		if( fullyPacked == true) {
			ShowAlert("Pack Alert","This item has been fully packed");
			$("#mr10").css("background-color","Red");        	    				
			vm.currentShipment.Packages[vm.packageIndex].MiscReference10 = "";
			angular.element('[ng-model="vm.currentShipment.Packages[vm.packageIndex].MiscReference10"]').focus();    	                                
			return;					
		}

		$("#mr10").css("background-color","Green");        	    	

		//if( itemFound == false) {
		//	ShowAlert("Pack Error 2","This SKU (" + newitemsku + ") does not belong in this shipment. Please remove it before proceeding.");
		//} else {
        //    $("#mr10").css("background-color","Green");        	    	
		//}

        vm.currentShipment.Packages[vm.packageIndex].MiscReference10 = "";
        angular.element('[ng-model="vm.currentShipment.Packages[vm.packageIndex].MiscReference10"]').focus();    	                                
		}

this.PreShip = function (shipmentRequest, userParams) {

	    shipmentRequest = vmTemp.currentShipment;
		// Validate shipmentRequest and Packages array
		if (!shipmentRequest || !shipmentRequest.Packages || !Array.isArray(shipmentRequest.Packages)) {
			console.error("❌ [PreShip Error] Shipment request is invalid or missing required properties.");
			//alert("[PreShip Error] Shipment request is invalid or missing required properties.");
			return;
		}
	
		let validDimensions = [];
	
		console.log(`📦 [PreShip] Processing ${shipmentRequest.Packages.length} package(s)...`);
	
		// Retrieve current package dimensions from vmTemp
		try {
			let currentPackage = vmTemp.currentShipment.Packages[vmTemp.packageIndex];
			if (!currentPackage || !currentPackage.Dimensions) {
				console.warn("⚠️ [PreShip Warning] vmTemp.currentShipment package or dimensions not found.");
			} else {
				let dims = currentPackage.Dimensions;
				console.log(`📏 [PreShip] Current Package Dimensions from vmTemp: ${dims.Length}x${dims.Width}x${dims.Height} 0`);
			}
		} catch (error) {
			console.error("❌ [PreShip Error] Unable to retrieve package dimensions from vmTemp.");
		}
	
		// Loop through all packages to validate their dimensions
		for (let i = 0; i < shipmentRequest.Packages.length; i++) {
			console.log(`🔍 [PreShip] Checking package at index ${i}...`);
	
			let packageData = shipmentRequest.Packages[i];
	
			if (!packageData) {
				console.error(`❌ [PreShip Error] Package at index ${i} is undefined.`);
				//alert(`[PreShip Error] Package at index ${i} is undefined.`);
				return;
			}
	
			if (!packageData.Dimensions) {
				console.warn(`⚠️ [PreShip Warning] Package at index ${i} does not contain 'Dimensions'.`);
				//alert(`[PreShip Error] Package at index ${i} does not contain 'Dimensions'.`);
				return;
			}
	
			let dims = packageData.Dimensions;
	
			// Ensure all dimension values exist and are valid numbers
			if (dims.Length > 0 && dims.Width > 0 && dims.Height > 0) {
				let dimensionString = `${dims.Length}x${dims.Width}x${dims.Height} 0`; // Always append "0"
	
				// Validate format (three numeric values + "0" at the end)
				let dimParts = dimensionString.split(new RegExp("[x ]"), 4); // Split on 'x' and space
	
				if (dimParts.length === 4 && !isNaN(dimParts[0]) && !isNaN(dimParts[1]) && !isNaN(dimParts[2]) && dimParts[3] === "0") {
					validDimensions.push(dimensionString);
					console.log(`✅ [PreShip] Package ${i} Dimensions: ${dimensionString}`);
				} else {
					console.error(`❌ [PreShip Error] Package at index ${i} has incorrectly formatted dimensions: ${dimensionString}`);
					//alert(`[PreShip Error] Package at index ${i} has incorrectly formatted dimensions.`);
					return;
				}
			} else {
				console.error(`❌ [PreShip Error] Package at index ${i} has invalid dimensions.`);
				//alert(`[PreShip Error] Package at index ${i} has invalid dimensions.`);
				return;
			}
		}
	
		// Store only valid dimensions in MiscReference5
		if (validDimensions.length > 0) {
			shipmentRequest.PackageDefaults.MiscReference5 = validDimensions.join(", ");
			console.log(`✅ [PreShip] All dimensions successfully stored in MiscReference5: ${shipmentRequest.PackageDefaults.MiscReference5}`);
		} else {
			console.warn("⚠️ [PreShip Warning] No packages had valid dimensions.");
		}
	
		// Retrieve weight and markup values
		userParams.weight = $('div[caption="Weight"]').find('input[name="unitValue_txt"]').val();
		userParams.markup = $("#custom_markup_id").val();
		console.log(`📏 [PreShip] Retrieved weight: ${userParams.weight}, Markup: ${userParams.markup}`);
	
		// Custom code to hide/filter on postship errors - must be started here
		$("<style> .alert-danger { display: none; } </style>").appendTo("head");
		console.log("🚫 [PreShip] Hidden alert-danger elements to prevent UI issues.");
	
		alertInterval = setInterval(function () {
			var errAlert = $('div.alert-danger');
	
			// Filter and show alert
			if (errAlert.length > 0) {
				clearInterval(alertInterval);
				let msg = errAlert.text();
				if (msg.indexOf('-1') === -1 && msg.indexOf('CONNECTSHIP') === -1) {
					errAlert.show();
					console.warn(`⚠️ [PreShip Warning] PostShip Alert Detected: ${msg}`);
				}
			}
		}, 200);
	
		console.log("✅ [PreShip] Function completed successfully.");
	};

// function ParseCutTicket(inVal) // returns undefined
      ParseCutTicket: function (inVal)
      {
        var splitArray = inVal.split(" ", 2);
        var itemNumber = splitArray[0];
        var itemQuantity = splitArray[1];
        var remainingQuantity, shippedQuantity;
        var matchingArray;

        try
        {
          /* Now find the table row that corresponds to the item number passed into the function. */
          var rowsArray = $('#tablePackVerifyOrderDetails').find('tr.not-packed, tr.not-complete');

          matchingArray = $.grep(rowsArray, function (pVal, pIndex)
          {
            remainingQuantity = parseFloat($(pVal).find('td:eq(3)').text());
            shippedQuantity = parseFloat($(pVal).find('td:eq(2)').text());
            return (($(pVal).text().split(" ")[0] == itemNumber) && (itemQuantity == remainingQuantity));
          })[0];

          /* The matchingArray will contain rows that match the item needed with the quantity needed. */
          /* If the array is empty, find rows that have enough material to use. */
          if ($.isEmptyObject(matchingArray))
          {
            matchingArray = $.grep(rowsArray, function (pVal, pIndex)
            {
              remainingQuantity = parseFloat($(pVal).find('td:eq(3)').text());
              shippedQuantity = parseFloat($(pVal).find('td:eq(2)').text());
              return (($(pVal).text().split(" ")[0] == itemNumber) && (itemQuantity < remainingQuantity));
            })[0];
          }

          /* If the matchingArray is still empty, then the user tried to scan something into the box that wasn't part of the order.  Raise an error. */
          if ($.isEmptyObject(matchingArray))
          {
            this.ShowErrorAlert('ERROR: Item ' + itemNumber + ' not found within currently loaded order(s).');
            return false;
          }

          $(matchingArray).data().ScanQuantity = itemQuantity;
          $(matchingArray).data().ItemNumber = itemNumber;

          this.AddRowItemToCurrentBox(matchingArray);
        }
        catch (error)
        {
          Logger.Log({ Source: 'ParseCutTicket()', Error: error });
        }
      }

// function GetOrderInformationFromCustomer(OrderList, CallBack) // returns undefined
      GetOrderInformationFromCustomer: function (OrderList, CallBack)
      {
        var dataObject = {
          ServerMethod: 'GetOrderInformation',
          OrderList: OrderList
        };

        var ajaxObject = {
          UserContext: SessionData.GetCurrentUserContext(),
          Data: JSON.stringify(dataObject)
        };

        var ajaxRequest = $.ajax(
        {
          url: 'api/ShippingService/UserMethod',
          method: 'POST',
          contentType: 'application/x-www-form-urlencoded; charset=UTF-8',
          dataType: 'json',
          data: ajaxObject
        });

        ajaxRequest.done(function (data)
        {
          var jsonData = JSON.parse(data.Data);

          for (var x = 0; x < jsonData.length; x++)
          {
            if (PackVerify.CheckForMatchingConsignee(jsonData[x]) === true)
            {
              PackVerify.AddCommoditiesToTable(jsonData[x].Details);
              PackVerify.UpdateReferencesWithOrderNumbers(jsonData[x].Header.Order_Number);
            }
            else
            {
              /* Add a note to the footer of the dialog. */
              $('#spanPackVerifyFooter').text('Could not add commodities for order ' + jsonData[x].Header.Order_Number);
            }
          }
        });

        ajaxRequest.fail(function (jqXHR, textStatus)
        {
          Logger.Log({ Source: GetCallingFunctionName(), Error: { name: 'AjaxError', message: jqXHR.ResponseText } });
        });

        ajaxRequest.always(function (jqXHR, textStatus, jqXHRError)
        {
          if (CallBack && typeof (CallBack) == "function")
          {
            CallBack();
          }
        });
      }

      CheckForMatchingConsignee: function (CustomerOrderObject)
      {
        try
        {
          var con = $('name-address[nameaddress="vm.currentShipment.PackageDefaults.Consignee"]');
          var a = {
            conCompany: $(con).find('input[name="company"]').val().trim().toUpperCase(),
            conAddress1: $(con).find('input[name="address1"]').val().trim().toUpperCase(),
            conCity: $(con).find('input[name="city"]').val().trim().toUpperCase(),
            conState: $(con).find('input[name="stateProvince"]').val().trim().toUpperCase(),
            conPostalCode: $(con).find('input[name="postalCode"]').val().trim().substring(0, 5).toUpperCase()
          };

          var hdr = CustomerOrderObject.Header;
          var b = {
            conCompany: hdr.CONSIGNEE_COMPANY.trim().toUpperCase(),
            conAddress1: hdr.CONSIGNEE_ADDRESS_LINE_1.trim().toUpperCase(),
            conCity: hdr.CONSIGNEE_CITY.trim().toUpperCase(),
            conState: hdr.CONSIGNEE_STATEPROVINCE.trim().toUpperCase(),
            conPostalCode: hdr.CONSIGNEE_POSTALCODE.trim().substring(0, 5).toUpperCase()
          };

          if (JSON.stringify(a) != JSON.stringify(b))
          {
            Logger.Log({ Source: 'CheckForMatchingConsignee()', Message: 'Could not add order ' + hdr.SHIPPER_REFERENCE + '. Primary order data to follow.', Data: a });
            Logger.Log({ Source: 'CheckForMatchingConsignee()', Message: 'Could not add order ' + hdr.SHIPPER_REFERENCE + '. Subsequent order data to follow.', Data: b });
          }

          /* Quick comparison to see if the strings match. */
          return (JSON.stringify(a) == JSON.stringify(b));
        }
        catch (error)
        {
          Logger.Log({ Source: 'CheckForMatchingConsignee()', Error: error });
        }
      }

      {
        try
        {
          /* Append the shipper reference with each item in the select list in csv format. */
          /* Once this is done, change the SSBR on post load to loop through each option in the csv list. */
          var inputShipperReference = $('input[type=text][ng-model="vm.currentShipment.Packages[vm.packageIndex].ShipperReference"]');

          /* If the user closes the dialog box, then opens it back up, we don't want to check the same orders more than once so only process what was entered into the dialog box this time. */
          var ordersToQuery = '';

            });

            /* If the current option value is NOT in the list within the Shipper Reference field, add it. */
            if ($.inArray(v.value.trim(), arrayOrders) == -1)
            {
              ordersToQuery += v.value.trim() + ',';
              //shipperReferenceVal += ',' + v.value;
            }

            /* Remove the option once it is added to the input box. */
            $(this).remove();

            /* Hide the div. */
            $('#divModalConsolidateShipments').hide();

            /* Make sure there are orders in the list. */
            if (ordersToQuery.length > 0)
            {
              PackVerify.GetOrderInformationFromCustomer(ordersToQuery);
            }
            else
            {
              PackVerify.ShowErrorAlert('ERROR: No unique orders to pull from the server.  All consolidated orders have already been added to this shipment.');
            }
          });
        }
        catch (error)
        {
          Logger.Log({ Source: 'SaveConsolidatedOrderNumbers()', Error: error });

  UpdateGoodsObject(pkgCount, listItem)
  {
    const itemkey = $(listItem).attr('hash-key');

      ccobj.packageIndex = findex + y;

      this.working.push(ccobj);

      this.UpdateScreenMarkup(ccobj, listItem);
    }
  }

  UpdateListItemPackageQuantities(li)
  {
    this.UpdatePkgCountsInWorkingObject(li);

    let numOfItems = $(li).find('.num-of-items-in-pkg');
    let remItems = $(li).find('.goods-quantity > strong');

    if (!$(numOfItems).val()) { return; }

    let x = (+$(remItems).text() - $(numOfItems).val());

    //sanity check to make sure we didn't pack more items that we had
    if (x < 0)
    {
      $(remItems).addClass('bad-math');
      return;
    }

    $(remItems).removeClass('bad-math');
    $(numOfItems).attr('disabled', 'disabled');
    $(li).attr('pkg-status', 'packed');

    //move to the next list item and update then focus
    $(li).next().find('.goods-quantity > strong').text(x)

    //on the second-to-last list item, use the input value to auto-set the last item
    //to get the number of rows that have already been assigned, count the list items with the goods-key
    //that have disabled input boxes
    let rowsForThisGood = $('#listItemContainer').find('li[goods-key="' + $(li).attr('goods-key') + '"]');
    let rowsDone = $(rowsForThisGood).find('input:disabled:visible').length;

    if (rowsForThisGood.length == (rowsDone + 1))
    {
      let lastLi = $('#listItemContainer').find('li[goods-key="' + $(li).attr('goods-key') + '"]').last();

      $(lastLi).attr('pkg-status', 'packed');

      let lastVal = $(li).next().find('.goods-quantity > strong').text();
      $(lastLi).find('.num-of-items-in-pkg').val(lastVal).attr('disabled', 'disabled');

      $('#listItemContainer').find('.assign-to-pkg:visible').first().focus();

      this.UpdatePkgCountsInWorkingObject(lastLi);
    }
  }

// function ApplyCommodityAssignments() // returns undefined
  ApplyCommodityAssignments()
  {
    let that = this;

    //every list item will have a pkg-index attr, get that highest value to determine how many pkgs we need
    //don't use pkg-status.length because that will include multiple commodities and the end result will be wrong
    let packed = $('#listItemContainer').find('li[pkg-status=packed]');

    //first get all the pkg-index attributes, then get the highest number and add one
    let pkgs = $(packed).map(function(x, value) { return +$(this).attr('pkg-index') }).toArray();
    let pkgCount = Math.max(...pkgs) + 1;

    while (this.packages.length < pkgCount)
    {
      //the vm.addPackage isn't firing, just click the button
      $('button[ng-click="vm.addPackage();"]').trigger('click').promise().done(function(e) { });
    }

    //loop the pkg count, get the commodities for that pkg, write the objects to the vm CommodityContents object
    for (let p = 0; p < pkgCount; p++)
    {
      let w = $.grep(this.working, function(item, i) { return item.packageIndex == p });
      this.packages[p].CommodityContents = w;
    }

    //click the General tab
    $('li[heading=General]:visible a').first().click();
    this.HideAssignGoodsPane();

    //after we are done, move back to the first package in the shipment
    while ($('#spanCurrentPackageIndex').text() !== '1')
    {
      $('button[ng-click="vm.previousPackage();"]').first().click();
    }

  }

    {

// MoveGoodToBox(li, destBox) // returns false | undefined
MoveGoodToBox(li, destBox)
    {
      let vm = this.ClassViewModel;
      let that = this;
      let data = $(li).data();
      let destCC = this.packages[destBox - 1].CommodityContents || [];

      //find the box the item currently resides in
      //split this statement up over a few calls to make it easier to read
      let allGoods = this.packages.map(m => m.CommodityContents).flat();
      let item = allGoods.find(cc => cc.UniqueId == data.UniqueId);

      if (+item.PkgIndex == +destBox) { console.log('Item already exists in box.'); return false; }

      //remove the item from its current box
      this.packages[item.PkgIndex - 1].CommodityContents = this.packages[item.PkgIndex - 1].CommodityContents.filter(p => p.UniqueId != data.UniqueId);

      //add it to the new box
      item.PkgIndex = destBox;
      destCC.push(item);
      $(li).find('.pkg-index-text').fadeOut().promise().done(function(e) { $(this).text(destBox).fadeIn(); });

      //console.log(vm.currentShipment.Packages);
      $('li[heading="Goods"]:visible:first > a').trigger('click');
      $('#goods .btn-group').find('.btn:gt(0)').not('.active').first().trigger('click');

      //update the weights and values
      //this data will have changed by now so get a new reference to the object
      allGoods = this.packages.map(m => m.CommodityContents).flat();
      //the items array will have the CommodityContents object for every package in the shipment

      //sort by package index
      //update the weight and declared values for all the packages in the shipment
      allGoods = allGoods.sort((a, b) => { return a.PkgIndex - b.PkgIndex });
      $(allGoods).each(function(a, b)
      {
        let pkg = that.packages[b.PkgIndex - 1];

        if ((a == 0) || (b.PkgIndex != allGoods[a - 1].PkgIndex))
        {

          pkg.DeclaredValueAmount.Amount = 0;
        }

        pkg.Weight.Amount += +b.PVTotalWeight;

      });
    }
