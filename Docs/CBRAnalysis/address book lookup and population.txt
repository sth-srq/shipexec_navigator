// getAddressBookEntries(addressBookName, context) // returns array|undefined
    function getAddressBookEntries(addressBookName, context) {
        var searchCriteria = { WhereClauses: [{ FieldName: "Name", FieldValue: addressBookName, Operator: 0 }] };
        var data = { SearchCriteria: searchCriteria, CompanyId: context.CompanyId, SiteId: context.SiteId, UserId: context.UserId };
        debugger;
        //var response = ajaxPost('GetAddressBooks', data);
        var response = thinClientAPIRequest('GetAddressBooks_Shared', data, false);
        debugger;
        if (response.TotalRecords == 0) {
            var addressBookEntry = { Code: 'Error: AddressBook "' + addressBookName + '" Not Found' };
            return [addressBookEntry];
        } else {
            var addressBookId = response.AddressBooks[0].Id;

            searchCriteria = { OrderByClauses: [{ FieldName: "Code", Direction: "asc", FieldType: 3 }] };
            data = { AddressBookIds: [addressBookId], SearchCriteria: searchCriteria, CompanyId: context.CompanyId, SiteId: context.SiteId, UserId: context.UserId };
            //response = ajaxPost('GetAddressBookEntries', data);
            response = thinClientAPIRequest('GetAddressBookEntries_Shared', data, false);
            debugger;
            if (response.TotalRecords > 0) {
                return response.AddressBookEntries;
            }
        }
    }

// GetAddressBook(addressBookName, context) // returns address book object or undefined
this.GetAddressBook = function (addressBookName, context) {
        var searchCriteria = { WhereClauses: [{ FieldName: "Name", FieldValue: addressBookName, Operator: 0 }] };
        var data = { SearchCriteria: searchCriteria, CompanyId: context.CompanyId, SiteId: context.SiteId, UserId: context.UserId };
        var response = this.thinClientAPIRequest('GetAddressBooks_Shared', data, false);
        if (response.ErrorCode != 0) {
            client.alert.Danger(response.ErrorMessage);
        } else {
            if (response.TotalRecords == 0) {
                client.alert.Danger('Address Book: "' + addressBookName + '" does not exist');                
            } else {
                return response.AddressBooks[0];
            }
        }
    }

// GetAddressBookEntries(addressBook, context, codeReference) // returns address book entries array or undefined
this.GetAddressBookEntries = function (addressBook, context, codeReference) {
        if (addressBook != undefined) {
            var searchCriteria = { WhereClauses: [{ FieldName: "Code", FieldValue: codeReference, Operator: 0 }] };
            var data = { AddressBookIds: [addressBook.Id], SearchCriteria: searchCriteria, CompanyId: context.CompanyId, SiteId: context.SiteId, UserId: context.UserId };
            var response = this.thinClientAPIRequest('GetAddressBookEntries_Shared', data, false);
            if (response.ErrorCode != 0) {
                client.alert.Danger(response.ErrorMessage);
            } else {
                if (response.TotalRecords == 0) {
                    client.alert.Danger('Address Book Entries: No address found in "' + addressBookName + '" address book');
                } else {
                    return response.AddressBookEntries;
                }
            }
        }
    }

                    if (!shipToCountry | shipFromCountry == "US" | shipFromCountry == "CA") {
                        DisableThirdPartyBillingElements();
                        return false;

                    if (shipFromCountry != shipToCountry) {
                        /* If the shipFromCountry does NOT equal the shipToCountry then ship Third Party. */
                        SetThirdPartyBilling(shipmentRequest);
                    }
                    else {
                        /* Now check to see if both the origin and destination countries are members of the EU. */

                            SetThirdPartyBilling(shipmentRequest);

// function GetAddressBookEntries(addressBookName, context, shipper) // returns AddressBookEntries array or throws
function (addressBookName, context, shipper) {
        var searchCriteria = { WhereClauses: [{ FieldName: "Name", FieldValue: addressBookName, Operator: 0 }] };
        var data = { CompanyId: context.CompanyId, UserId: context.UserId };
        var response = thinClientAPIRequest('GetAddressBooks', data, false).responseJSON;
        if (response.TotalRecords == 0) {
            throw ({ message: 'Address Book:"' + addressBookName + '" does not exists'});
        } else {
            var addressBookId = response.AddressBooks[0].Id;
            
            searchCriteria = { Skip: 0, WhereClauses: [{FieldName: "Code", FieldValue: shipper, Operator: 0}], OrderByClauses: [] };
            data = { AddressBookIds: [addressBookId], SearchCriteria: searchCriteria, UserContext :{Machine:""} , CompanyId: context.CompanyId, UserId: context.UserId };
            response = thinClientAPIRequest('GetAddressBookEntries', data, false).responseJSON;
            
            if (response.TotalRecords == 0) {
                throw ({ message: 'Address Book:"' + addressBookName + '" does not have any address entry'});
            } else {
                return response.AddressBookEntries;
            }
        }
    }

        var data = { CompanyId: userContext.CompanyId };
        thinClientAPIRequest(apiMethod, data, true).done(function (result) {

                var userObject = { Action: apiMethod, Data: JSON.stringify(result) };

    this.PostalCodeLookup = function (addressName) {
        var userContext = this.getUserContext(this);
        var postalCode = $('[address-property-name="' + addressName + '"] [name="postalCode"]').val();
        var content = { AddressName: addressName, PostalCode: postalCode };
        var payload = { Action: 'CSZ', Content: JSON.stringify(content) };
        var data = { Data: JSON.stringify(payload), UserContext: userContext };
        var response = this.thinClientAPIRequest('UserMethod', data, false).responseJSON;
        if (response.ErrorCode > 0) {
            client.shipping.alertModalShow(this, 'Postal Code Search', response.ErrorMessage)
        } else {
            var responseData = JSON.parse(client.htmlText(response.Data));
            if (responseData === null) {
                client.shipping.alertModalShow(this, 'Postal Code Search', 'Invalid or Not US Postal Code');
            } else {
                if (responseData.length > 1) {
                    this.addressResults = responseData;
                    client.shipping.validateAddressModalShow();
                } else {
                    this.currentShipment.updateNameAddressCSZ(addressName, responseData[0].City, responseData[0].StateProvince)
                }
            }
        }
    }

    this.PostalCodeLookup = function (addressName) {
debugger;
        var userContext = this.getUserContext(this);
        var postalCode = $('[address-property-name="' + addressName + '"] [name="postalCode"]').val();
        var content = { AddressName: addressName, PostalCode: postalCode };
        var payload = { Action: 'CSZ', Content: JSON.stringify(content) };
        var data = { Data: JSON.stringify(payload), UserContext: userContext };
        var response = this.thinClientAPIRequest('UserMethod', data, false).responseJSON;
        if (response.ErrorCode > 0) {
            client.shipping.alertModalShow(this, 'Postal Code Search', response.ErrorMessage)
        } else {
            var responseData = JSON.parse(client.htmlText(response.Data));        
            if (responseData == null) {
                client.shipping.alertModalShow(this, 'Postal Code Search', 'Invalid or Not US Postal Code');
            } else {
                if (responseData.length > 1) {
                    this.addressResults = responseData;
                    client.shipping.validateAddressModalShow();
                } else {
                    this.currentShipment.updateNameAddressCSZ(addressName, responseData[0].City, responseData[0].StateProvince)
                }
            }
        }
    }

// function UpdateNameAddressCSZ(addressName, city, stateProvince) // returns void
    this.UpdateNameAddressCSZ = function (addressName, city, stateProvince) {
        var nameAddress = this.PackageDefaults.Consignee;
        if (addressName == 'ReturnAddress') {
            if (this.PackageDefaults.ReturnAddress == undefined)
                this.PackageDefaults.ReturnAddress = {};
            nameAddress = this.PackageDefaults.ReturnAddress;
        }
        nameAddress.City = city;
        nameAddress.StateProvince = stateProvince;
    }

    this.SelectAddress = function (address) {

    }
