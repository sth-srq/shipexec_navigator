// this.PostRate(shipmentRequest, rateResults) // returns undefined
    this.PostRate = function (shipmentRequest, rateResults) {
        debugger;
        modifiedRateResults();
        if (this.vm.shipping.IsMailroom) {
            rateResults[0].customRateResults = customRateResults;
        }

        function modifiedRateResults() {
            for(var i = 0; i< rateResults.length; i++) {
                var rateService = rateResults[i].PackageDefaults.Service;
                rateService.Name = rateService.Name.replace('(CSAdapter)', '').trim();
                rateResults[i].SortIndex = rateService.Symbol == shipmentRequest.PackageDefaults.Service.Symbol ? 0 : i + 1;
            }
        } 

        function customRateResults(vm) {
            var selectService = selectSortIndex();
            vm.selectService = [ selectService ];
            vm.selectedRow = selectService.Symbol;

            vm.customOnSelectService = customOnSelectService;

            function selectSortIndex () {
                for(var index in rateResults) {
                    if (rateResults[index].SortIndex === 0) {
                        return rateResults[index].PackageDefaults.Service;
                    }
                }
            }

            function customOnSelectService(service, serviceType) {
                if (this.profile.ProfileSetting.ServiceSelectionMode === this.serviceSelectionMode.ShipByService) {
                    this.onSelectService(service, serviceType)
                } else {
                    this.selectService = [ service ];
                    this.selectedRow = service.Symbol;
                    if (!this.profile.ProfileSetting.DisableJustificationMessage) {
                        this.selectedRow = service.Symbol;
                        if (serviceType === "ApplicableService" && selectSortIndex() !== service)
                            this.justify = true;
                        else if (serviceType === "NotApplicableService")
                            this.justify = true;
                        else
                            this.justify = false;
                    } else this.justify = false;
                }
            }
        }
    }

// this.validateClientMatter() // returns undefined
    this.validateClientMatter = function() {

        if( theVM.currentShipment.Packages[0].ShipperReference === null) {
            alert('Client Matter field is required. Please enter a valid client matter number!');
            setFocus('input[name="ShipperReference_txt"]');            
            return;
        }
        
        var cmFound = false;
        var cm = theVM.currentShipment.Packages[0].ShipperReference;
        var cmValidationList = theVM.fieldOptions.ShipperReference.FieldOptionValidations;
        for(var i = 0; i < cmValidationList.length; i++) {
            var cmValidationItem = cmValidationList[i];
            if( cmValidationItem.Value == cm) {
                cmFound = true;
                break;
            }
        }
        if( cmFound )
            theVM.saveLicensePlate();
        else {
            alert("Client Matter field is invalid. Please enter a valid client matter number!");
            setFocus('input[name="ShipperReference_txt"]');                        
        }
    }

    this.PreShip = function(shipmentRequest, userParams) {

        var freightfound = 0;
        var frieghterr = 0;

        for (var i = 0; i <= shipmentRequest.Packages.length - 1; i++) {
            if (parseInt(shipmentRequest.Packages[i].Weight.Amount) >= 70) {
                freightfound = "1";
            }
            if (freightfound == "1") {
                if (parseInt(shipmentRequest.Packages[i].Weight.Amount) < 70) {
                    frieghterr = "1";
                }
            }
        }
        if (freightfound == "0") {
            shipmentRequest.PackageDefaults.Service = {
                Symbol: "CONNECTSHIP_UPS.UPS.EXPSVR"
            }
        } else {
            if (frieghterr == "0") {
                shipmentRequest.PackageDefaults.Service = {
                    Symbol: "CONNECTSHIP_UPS.UPS.EXPFRT"
                }
            } else {
                alert("Cannot create one shipment with 2 different service levels.");
            }
        }
        for (var i = 0; i <= shipmentRequest.Packages.length - 1; i++) {
            if (shipmentRequest.PackageDefaults.Service.Symbol == "CONNECTSHIP_UPS.UPS.EXPFRT") {
                shipmentRequest.Packages[i].Packaging = "CUSTOMER_PALLET";
            }
            shipmentRequest.Packages[i].ImportDelivery = true;

        }
        if (shipmentRequest.PackageDefaults.OriginAddress.Country == "US") {
            if ($('select[name="CarrierInstructions"]').val() == '?') {
                alert("Please confirm shipment value");
                throw ("Excpetion");
            } else {
                if (shipmentRequest.Packages[0].ConsigneeReference == null || shipmentRequest.Packages[0].ConsigneeReference == "") {
                    if ($('select[name="CarrierInstructions"]').val() == 'string:Yes') {
                        alert("Please Enter ITN/ EES number in ConsigneeReference");
                        throw ("Excpetion");
                    }
                }
            }
        } else {
            if (shipmentRequest.PackageDefaults.OriginAddress.Country != "undefined") {
                alert("For EU Exports : Shipments above 1000 EUR, a formal export declaration is required. A formal electronic export declaration must be sent to customs authorities by the exporter or their representative. Customs authorities grant the release for export and issue an Export Accompanying Document (EAD) with a unique Movement Reference Number (MRN). The EAD has to accompany the shipment. UPS is responsible for the exit clearance at the EU border.")
            }
        }
    }

// this.PreShip(shipmentRequest, userParams) // returns undefined
this.PreShip = function(shipmentRequest, userParams) {    
    // We will require the actual shipment to send weight and dims in kg/cm
    // However, because US Vendors are sending the product the client gui needs to be in Lb/In.
    
    // debugger;
    
    totalShipWeight = 0;
    totalVolumetricResult = 0;
    
    vm.currentShipment.Packages.forEach(pkg => {
        totalShipWeight += Number(pkg.Weight.Amount);
        totalVolumetricResult += ((Number(pkg.Dimensions.Length).toFixed(2)*Number(pkg.Dimensions.Width).toFixed(2)*Number(pkg.Dimensions.Height)).toFixed(2))/166;
    });
              
    totalShipWeight = Number(totalShipWeight).toFixed(2);
    console.log(totalShipWeight);
    console.log(totalVolumetricResult);
    if (totalShipWeight > 150)
    {
        alert("Total Shipment Weight Exceeds Shipment Weight Limit of 150lbs")
        throw new Error("Shipping halted due to Over Maximum Weight Rule " + totalShipWeight);
    }
    
    if (totalVolumetricResult > Number(maxvolumetric))
    {
        alert("Total Shipment Weight Exceeds Total Shipment Volumetric Weight Limit of " + Number(maxvolumetric))
        throw new Error("Shipping halted due to Total Shipment Volumetric Weight Rule " + Number(totalVolumetricResult).toFixed(2));
    }
    
    //kg = lbs*0.453592    
    vm.currentShipment.Packages.forEach(pkg => {
         if (pkg.Weight.Units === "LB"){
             pkg.Weight.Amount = Number(pkg.Weight.Amount * 0.453592).toFixed(2);
             pkg.Weight.Units = "KG";
         }
         
    });
    
  
    //cm=in/0.39370
    vm.currentShipment.Packages.forEach(pkg => {
         if (pkg.Dimensions.Units === 0){
             pkg.Dimensions.Length = Number(pkg.Dimensions.Length/0.39370).toFixed(2)
             pkg.Dimensions.Width = Number(pkg.Dimensions.Width/0.39370).toFixed(2)
             pkg.Dimensions.Height = Number(pkg.Dimensions.Height/0.39370).toFixed(2)
             pkg.Dimensions.Units = 1;
       }
         
    });
     
   if(Number(shipmentRequest.Packages[0].DeclaredValueCustoms.Amount).toFixed(2) > 2500 ){
          if(shipmentRequest.Packages[0].ExportInformationCode === null){
              shipmentRequest.Packages[0].ExportInformationCode = "OS";
          }
          if(shipmentRequest.Packages[0].AesTransactionNumber === null){
              shipmentRequest.Packages[0].SedMethod = 2;
              $('input[type=text][ng-model="shipmentRequest.Packages[this.vm.packageIndex].AesTransactionNumber"]').focus();
              throw new Error('Missing AES/ITN Number');
          } 
          
          if(shipmentRequest.Packages[0].AesTransactionNumber !== null){
              shipmentRequest.Packages[0].Comments = shipmentRequest.Packages[0].Comments + "\n ITN/AES Filer " + shipmentRequest.Packages[0].AesTransactionNumber;
             
          }
      }
};

// this.ShippingLoaded() // returns undefined
this.ShippingLoaded = function(){
    this.vm.populateReturnAddress = this.PopulateReturnAddress;
    this.vm.populateShipToAddress = this.PopulateShipToAddress;
    this.vm.serviceOption = serviceOption;
    this.vm.serviceOptions = serviceOptions(this.vm.profile);
    this.vm.currentShipment.Packages[0].ShipperReference = cost_Center;
    this.vm.currentShipment.Packages[0].MiscReference1 = cost_Center;
    //this.vm.currentShipment.Packages[0].ConsigneeReference = po_number;
    //this.vm.currentShipment.Packages[0].MiscReference2 = po_Number;
    this.vm.currentShipment.Packages[0].MiscReference3 = user_partNumber;
    this.vm.currentShipment.Packages[0].Description = user_description;
  
    function serviceOption() {
        this.selectedServices = [this.currentShipment.PackageDefaults.Service];
    }
    function serviceOptions(profile) {
        //debugger;
        var filterServices = [];
        var customData3 = uPS_Service.split(',');
        for (var item of customData3) {
            var symbol = getServiceSymbol(item.toLowerCase());
            for (var service of profile.Services) {
                if (service.Symbol === symbol) {
                    //service.Name = service.Name.replace('(UPS Adapter)','').trim();
                    filterServices.push(service);
                    break;
                }
            }
        }
        return filterServices;
        function getServiceSymbol(code) {
             if (code == '2dn') return 'UPSAPI.UPS.2DA'
            else if (code == '3dd') return 'UPSAPI.UPS.3DS'
            else if (code == 'exp') return 'UPSAPI.UPS.EXPEDITED'
            else if (code == 'ess') return 'UPSAPI.UPS.EXPRESS'
            else if (code == 'nda') return 'UPSAPI.UPS.NDA'
            else if (code == 'dae') return 'UPSAPI.UPS.NDAEAM'
            else if (code == 'grn') return 'UPSAPI.UPS.GND'
            else if (code == 'uwe') return 'UPSAPI.UPS.EXPEDITED'
            else if (code == 'wep') return 'UPSAPI.UPS.EXPRESSSP'
            else if (code == 'swe') return 'UPSAPI.UPS.EXPRESS'
            else if (code == 'uws') return 'UPSAPI.UPS.SAVER'
            else if (code == 'std') return 'UPSAPI.UPS.STANDARD'
        }
        
    }
    
    if(dims.includes(',')){
        var dimensions = dims.split(',');
        
        this.vm.currentShipment.Packages[0].Weight.Amount = Number(dimensions[0]);
        maxvolumetric = Number(dimensions[4]);
        console.log("Max Volumetric: " + maxvolumetric);
        this.vm.currentShipment.Packages[0].Dimensions = {
            Length: Number(dimensions[1]),
            Height: Number(dimensions[2]),
            Width: Number(dimensions[3]),
            Units: 0
        };
    }
    vm = this.vm;
};

// PreShip(shipmentRequest, userParams) // returns void
this.PreShip = function(shipmentRequest, userParams) {    
  
    // We will require the actual shipment to send weight and dims in kg/cm
     // However, because US Vendors are sending the product the client gui needs to be in Lb/In.
    
     // debugger;
     
     totalShipWeight = 0;
    totalVolumetricResult = 0;
    
    vm.currentShipment.Packages.forEach(pkg => {
        totalShipWeight += Number(pkg.Weight.Amount);
        totalVolumetricResult += ((Number(pkg.Dimensions.Length).toFixed(2)*Number(pkg.Dimensions.Width).toFixed(2)*Number(pkg.Dimensions.Height)).toFixed(2))/166;
    });
              
    totalShipWeight = Number(totalShipWeight).toFixed(2);
    console.log(totalShipWeight);
    console.log(totalVolumetricResult);
    if (totalShipWeight > 150)
    {
        alert("Total Shipment Weight Exceeds Shipment Weight Limit of 150lbs")
        throw new Error("Shipping halted due to Over Maximum Weight Rule " + totalShipWeight);
    }
    
    if (totalVolumetricResult > Number(maxvolumetric))
    {
        alert("Total Shipment Weight Exceeds Total Shipment Volumetric Weight Limit of " + Number(maxvolumetric))
        throw new Error("Shipping halted due to Total Shipment Volumetric Weight Rule " + Number(totalVolumetricResult).toFixed(2));
    }
     
     //kg = lbs*0.453592    
     vm.currentShipment.Packages.forEach(pkg => {
         if (pkg.Weight.Units === "LB"){
             pkg.Weight.Amount = Number(pkg.Weight.Amount * 0.453592).toFixed(2);
             pkg.Weight.Units = "KG";
         }
         
     });
     
    //cm=in/0.39370
     vm.currentShipment.Packages.forEach(pkg => {
         if (pkg.Dimensions.Units === 0){
             pkg.Dimensions.Length = Number(pkg.Dimensions.Length/0.39370).toFixed(2)
             pkg.Dimensions.Width = Number(pkg.Dimensions.Width/0.39370).toFixed(2)
             pkg.Dimensions.Height = Number(pkg.Dimensions.Height/0.39370).toFixed(2)
             pkg.Dimensions.Units = 1;
         }
         
     });
     
     
    // debugger;
      if(Number(vm.currentShipment.Packages[0].DeclaredValueCustoms.Amount).toFixed(2) > 2500 ){
          if(vm.currentShipment.Packages[0].ExportInformationCode === null){
              vm.currentShipment.Packages[vm.packageIndex].ExportInformationCode = "OS";
          }
          if(vm.currentShipment.Packages[0].AesTransactionNumber === null){
              vm.currentShipment.Packages[0].SedMethod = 2;
              $('input[type=text][ng-model="vm.currentShipment.Packages[vm.packageIndex].AesTransactionNumber"]').focus();
              throw new Error('Missing AES/ITN Number');
          } 
          if(vm.currentShipment.Packages[0].AesTransactionNumber !== null){
              vm.currentShipment.Packages[0].Comments = vm.currentShipment.Packages[0].Comments + "\n ITN/AES Filer " + vm.currentShipment.Packages[0].AesTransactionNumber;
             
          }
      }
     
    console.log(shipmentRequest);
  
};

    function serviceOptions(profile) {
        //debugger;
        var filterServices = [];
        var customData3 = uPS_Service.split(',');
        for (var item of customData3) {
            var symbol = getServiceSymbol(item.toLowerCase());
            for (var service of profile.Services) {
                if (service.Symbol === symbol) {
                    service.Name = service.Name.replace('(UPS Adapter)','').trim();
                    if (service.Name.includes("Expedited")){
                        service.Name = service.Name + " (MAIN SERVICE)";
                    }
                    if (service.Name.includes("Saver")){
                        service.Name = service.Name + " (To Heredia or Only with Boston's Authorization)";
                    }
                    filterServices.push(service);
                    break;
                }
            }
        }
        return filterServices;
        function getServiceSymbol(code) {
            if (code == '2dn') return 'UPSAPI.UPS.2DA'
            else if (code == '3dd') return 'UPSAPI.UPS.3DS'
            else if (code == 'exp') return 'UPSAPI.UPS.EXPEDITED'
            else if (code == 'ess') return 'UPSAPI.UPS.EXPRESS'
            else if (code == 'nda') return 'UPSAPI.UPS.NDA'
            else if (code == 'dae') return 'UPSAPI.UPS.NDAEAM'
            else if (code == 'grn') return 'UPSAPI.UPS.GND'
            else if (code == 'uwe') return 'UPSAPI.UPS.EXPEDITED'
            else if (code == 'wep') return 'UPSAPI.UPS.EXPRESSSP'
            else if (code == 'swe') return 'UPSAPI.UPS.EXPRESS'
            else if (code == 'uws') return 'UPSAPI.UPS.SAVER'
            else if (code == 'std') return 'UPSAPI.UPS.STANDARD'
        }
        
    }
    
    if(dims.includes(',')){

            Length: Number(dimensions[1]),
            Height: Number(dimensions[2]),
            Width: Number(dimensions[3]),
            Units: 0
        };
    }
   
    vm = this.vm;
};

this.PreShip = function (shipmentRequest, userParams) {
		console.log("🚀 [PreShip] Starting PreShip function...");
	    shipmentRequest = vmTemp.currentShipment;
		// Validate shipmentRequest and Packages array
		if (!shipmentRequest || !shipmentRequest.Packages || !Array.isArray(shipmentRequest.Packages)) {
			console.error("❌ [PreShip Error] Shipment request is invalid or missing required properties.");
			//alert("[PreShip Error] Shipment request is invalid or missing required properties.");
			return;
		}
	
		let validDimensions = [];
	
		console.log(`📦 [PreShip] Processing ${shipmentRequest.Packages.length} package(s)...`);
	
		// Retrieve current package dimensions from vmTemp
		try {
			let currentPackage = vmTemp.currentShipment.Packages[vmTemp.packageIndex];
			if (!currentPackage || !currentPackage.Dimensions) {
				console.warn("⚠️ [PreShip Warning] vmTemp.currentShipment package or dimensions not found.");
			} else {
				let dims = currentPackage.Dimensions;
				console.log(`📏 [PreShip] Current Package Dimensions from vmTemp: ${dims.Length}x${dims.Width}x${dims.Height} 0`);
			}
		} catch (error) {
			console.error("❌ [PreShip Error] Unable to retrieve package dimensions from vmTemp.");
		}
	
		// Loop through all packages to validate their dimensions
		for (let i = 0; i < shipmentRequest.Packages.length; i++) {
			console.log(`🔍 [PreShip] Checking package at index ${i}...`);
	
			let packageData = shipmentRequest.Packages[i];
	
			if (!packageData) {
				console.error(`❌ [PreShip Error] Package at index ${i} is undefined.`);
				//alert(`[PreShip Error] Package at index ${i} is undefined.`);
				return;
			}
	
			if (!packageData.Dimensions) {
				console.warn(`⚠️ [PreShip Warning] Package at index ${i} does not contain 'Dimensions'.`);
				//alert(`[PreShip Error] Package at index ${i} does not contain 'Dimensions'.`);
				return;
			}
	
			let dims = packageData.Dimensions;
	
			// Ensure all dimension values exist and are valid numbers
			if (dims.Length > 0 && dims.Width > 0 && dims.Height > 0) {
				let dimensionString = `${dims.Length}x${dims.Width}x${dims.Height} 0`; // Always append "0"
	
				// Validate format (three numeric values + "0" at the end)
				let dimParts = dimensionString.split(new RegExp("[x ]"), 4); // Split on 'x' and space
	
				if (dimParts.length === 4 && !isNaN(dimParts[0]) && !isNaN(dimParts[1]) && !isNaN(dimParts[2]) && dimParts[3] === "0") {
					validDimensions.push(dimensionString);
					console.log(`✅ [PreShip] Package ${i} Dimensions: ${dimensionString}`);
				} else {
					console.error(`❌ [PreShip Error] Package at index ${i} has incorrectly formatted dimensions: ${dimensionString}`);
					//alert(`[PreShip Error] Package at index ${i} has incorrectly formatted dimensions.`);
					return;
				}
			} else {
				console.error(`❌ [PreShip Error] Package at index ${i} has invalid dimensions.`);
				//alert(`[PreShip Error] Package at index ${i} has invalid dimensions.`);
				return;
			}
		}
	
		// Store only valid dimensions in MiscReference5
		if (validDimensions.length > 0) {
			shipmentRequest.PackageDefaults.MiscReference5 = validDimensions.join(", ");
			console.log(`✅ [PreShip] All dimensions successfully stored in MiscReference5: ${shipmentRequest.PackageDefaults.MiscReference5}`);
		} else {
			console.warn("⚠️ [PreShip Warning] No packages had valid dimensions.");
		}
	
		// Retrieve weight and markup values
		userParams.weight = $('div[caption="Weight"]').find('input[name="unitValue_txt"]').val();
		userParams.markup = $("#custom_markup_id").val();
		console.log(`📏 [PreShip] Retrieved weight: ${userParams.weight}, Markup: ${userParams.markup}`);
	
		// Custom code to hide/filter on postship errors - must be started here
		$("<style> .alert-danger { display: none; } </style>").appendTo("head");
		console.log("🚫 [PreShip] Hidden alert-danger elements to prevent UI issues.");
	
		alertInterval = setInterval(function () {
			var errAlert = $('div.alert-danger');
		
			// Filter and show alert
			if (errAlert.length > 0) {
				clearInterval(alertInterval);
				let msg = errAlert.text();
				if (msg.indexOf('-1') === -1 && msg.indexOf('CONNECTSHIP') === -1) {
					errAlert.show();
					console.warn(`⚠️ [PreShip Warning] PostShip Alert Detected: ${msg}`);
				}
			}
		}, 200);
	
		console.log("✅ [PreShip] Function completed successfully.");
	};

        if (checked) {

// function SetShipNotificationAddressEmail(pkg, ngModelName, emailAddress, checked) // returns undefined
    function SetShipNotificationAddressEmail(pkg, ngModelName, emailAddress, checked) {
        if (checked) {
            if (pkg.ShipNotificationAddressEmail == undefined || pkg.ShipNotificationAddressEmail.trim() == '') {
                if (ngModelName == 'ShipFromNotificationEmail' && emailAddress.Origin) {
                    pkg.ShipNotificationAddressEmail = emailAddress.Origin;
                } else if (ngModelName == 'ShipToNotificationEmail' && emailAddress.Consignee) {
                    pkg.ShipNotificationAddressEmail = emailAddress.Consignee;
                }
            } else {
                if (ngModelName == 'ShipFromNotificationEmail' && emailAddress.Origin) {
                    var regexp = new RegExp(',?' + emailAddress.Origin, 'g');
                    if (!regexp.test(pkg.ShipNotificationAddressEmail)) {
                        pkg.ShipNotificationAddressEmail += ',' + emailAddress.Origin;
                    }
                } else if (ngModelName == 'ShipToNotificationEmail' && emailAddress.Consignee) {
                    pkg.ShipNotificationAddressEmail += ',' + emailAddress.Consignee;
                }
            }
        } else {
            var notificationAddressEmail = pkg.ShipNotificationAddressEmail;
            if (ngModelName == 'ShipToNotificationEmail' && emailAddress.Consignee) {
                var regexp = new RegExp(',?' + emailAddress.Consignee, 'g')
                notificationAddressEmail = notificationAddressEmail.replace(regexp, '');
            } else if (ngModelName == 'ShipFromNotificationEmail' && emailAddress.Origin) {
                var regexp = new RegExp(',?' + emailAddress.Origin, 'g')
                notificationAddressEmail = notificationAddressEmail.replace(regexp, '');
            }

            if (notificationAddressEmail.indexOf(',') == 0) {
                notificationAddressEmail = notificationAddressEmail.substring(1);
            }
            pkg.ShipNotificationAddressEmail = notificationAddressEmail;
        }
    }
