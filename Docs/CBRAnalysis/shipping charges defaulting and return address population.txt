    this.PreShip = function(shipmentRequest, userParams) {

        var freightfound = 0;
        var frieghterr = 0;

        for (var i = 0; i <= shipmentRequest.Packages.length - 1; i++) {
            if (parseInt(shipmentRequest.Packages[i].Weight.Amount) >= 70) {
                freightfound = "1";
            }
            if (freightfound == "1") {
                if (parseInt(shipmentRequest.Packages[i].Weight.Amount) < 70) {
                    frieghterr = "1";
                }
            }
        }
        if (freightfound == "0") {
            shipmentRequest.PackageDefaults.Service = {
                Symbol: "CONNECTSHIP_UPS.UPS.EXPSVR"
            }
        } else {
            if (frieghterr == "0") {
                shipmentRequest.PackageDefaults.Service = {
                    Symbol: "CONNECTSHIP_UPS.UPS.EXPFRT"
                }
            } else {
                alert("Cannot create one shipment with 2 different service levels.");
            }
        }
        for (var i = 0; i <= shipmentRequest.Packages.length - 1; i++) {
            if (shipmentRequest.PackageDefaults.Service.Symbol == "CONNECTSHIP_UPS.UPS.EXPFRT") {
                shipmentRequest.Packages[i].Packaging = "CUSTOMER_PALLET";
            }
            shipmentRequest.Packages[i].ImportDelivery = true;

        }
        if (shipmentRequest.PackageDefaults.OriginAddress.Country == "US") {
            if ($('select[name="CarrierInstructions"]').val() == '?') {
                alert("Please confirm shipment value");
                throw ("Excpetion");
            } else {
                if (shipmentRequest.Packages[0].ConsigneeReference == null || shipmentRequest.Packages[0].ConsigneeReference == "") {
                    if ($('select[name="CarrierInstructions"]').val() == 'string:Yes') {
                        alert("Please Enter ITN/ EES number in ConsigneeReference");
                        throw ("Excpetion");
                    }
                }
            }
        } else {
            if (shipmentRequest.PackageDefaults.OriginAddress.Country != "undefined") {
                alert("For EU Exports : Shipments above 1000 EUR, a formal export declaration is required. A formal electronic export declaration must be sent to customs authorities by the exporter or their representative. Customs authorities grant the release for export and issue an Export Accompanying Document (EAD) with a unique Movement Reference Number (MRN). The EAD has to accompany the shipment. UPS is responsible for the exit clearance at the EU border.")
            }
        }
    }

// function getKeyValue(key, arr) // returns string
    function getKeyValue(key, arr) {
        for (var index in arr) {
            if (arr[index].Key.toLowerCase() === key.toLowerCase()) return arr[index].Value;
        }
        return '';
    }

            conState: hdr.CONSIGNEE_STATEPROVINCE.trim().toUpperCase(),

          /* Quick comparison to see if the strings match. */
          return (JSON.stringify(a) == JSON.stringify(b));
        }
        catch (error)

        }
      }

// PackVerify.GetOrderInformationFromCustomer(OrderList, CallBack) // returns undefined
      GetOrderInformationFromCustomer: function (OrderList, CallBack)
      {
        var dataObject = {
          ServerMethod: 'GetOrderInformation',
          OrderList: OrderList
        };

        var ajaxObject = {
          UserContext: SessionData.GetCurrentUserContext(),
          Data: JSON.stringify(dataObject)
        };

        var ajaxRequest = $.ajax(
        {
          url: 'api/ShippingService/UserMethod',
          method: 'POST',
          contentType: 'application/x-www-form-urlencoded; charset=UTF-8',
          dataType: 'json',
          data: ajaxObject
        });

        ajaxRequest.done(function (data)
        {
          var jsonData = JSON.parse(data.Data);

          for (var x = 0; x < jsonData.length; x++)
          {
            if (PackVerify.CheckForMatchingConsignee(jsonData[x]) === true)
            {
              PackVerify.AddCommoditiesToTable(jsonData[x].Details);
              PackVerify.UpdateReferencesWithOrderNumbers(jsonData[x].Header.Order_Number);
            }
            else
            {
              /* Add a note to the footer of the dialog. */
              $('#spanPackVerifyFooter').text('Could not add commodities for order ' + jsonData[x].Header.Order_Number);
            }
          }
        });

        ajaxRequest.fail(function (jqXHR, textStatus)
        {
          Logger.Log({ Source: GetCallingFunctionName(), Error: { name: 'AjaxError', message: jqXHR.ResponseText } });
        });

        ajaxRequest.always(function (jqXHR, textStatus, jqXHRError)
        {
          if (CallBack && typeof (CallBack) == "function")
          {
            CallBack();
          }
        });
      }

      SaveConsolidatedOrderNumbers: function ()
      {
        try
        {
          /* Append the shipper reference with each item in the select list in csv format. */
          /* Once this is done, change the SSBR on post load to loop through each option in the csv list. */
          var inputShipperReference = $('input[type=text][ng-model="vm.currentShipment.Packages[vm.packageIndex].ShipperReference"]');
          var shipperReferenceVal = $(inputShipperReference).val();

          /* If the user closes the dialog box, then opens it back up, we don't want to check the same orders more than once so only process what was entered into the dialog box this time. */
          var ordersToQuery = '';

          /* Loop through all the orders in the select box and determine if they are already in the list. */
          $('#selectConsolidatedOrders').find('option').each(function (i, v)
          {
            /* Remove the spaces before the commas for comparison. */
            var arrayOrders = shipperReferenceVal.split(",").map(function (item)
            {
              return item.trim();
            });

            /* If the current option value is NOT in the list within the Shipper Reference field, add it. */
            if ($.inArray(v.value.trim(), arrayOrders) == -1)
            {
              ordersToQuery += v.value.trim() + ',';
              //shipperReferenceVal += ',' + v.value;
            }

            /* Remove the option once it is added to the input box. */
            $(this).remove();

          }).promise().done(function ()
          {
            /* Done creating the list of orders to get from the server which is now held in the ordersToQuery variable. */

            /* Hide the div. */
            $('#divModalConsolidateShipments').hide();

            /* Make sure there are orders in the list. */
            if (ordersToQuery.length > 0)
            {
              PackVerify.GetOrderInformationFromCustomer(ordersToQuery);
            }
            else
            {
              PackVerify.ShowErrorAlert('ERROR: No unique orders to pull from the server.  All consolidated orders have already been added to this shipment.');
            }
          });
        }
        catch (error)

// function checkOriginConsigneeCountry() // returns false when third-party billing is disabled; otherwise undefined
    function checkOriginConsigneeCountry() {
        var shipFromCountry = getObjectValue(vm.currentShipment.PackageDefaults.OriginAddress.Country);
        var shipToCountry = getObjectValue(vm.currentShipment.PackageDefaults.Consignee.Country);

        /* If we have no shipToCountry OR the shipFromCountry is either US or CA, disable the elements, return out. */
        if (!shipToCountry | shipFromCountry == "US" | shipFromCountry == "CA") {
            DisableThirdPartyBillingElements();
            return false;
        }

        if (shipFromCountry != shipToCountry) {
            /* If the shipFromCountry does NOT equal the shipToCountry then ship Third Party. */
            SetThirdPartyBilling(vm.currentShipment);
        }
        else {
            /* Now check to see if both the origin and destination countries are members of the EU. */
            if ((IsMemberOfEu(shipFromCountry)) && (IsMemberOfEu(shipToCountry))) {
                SetThirdPartyBilling(vm.currentShipment);
            }

            //ShipThirdPartyBilling((IsMemberOfEu(shipFromCountry)) && (IsMemberOfEu(shipToCountry)));
        }

        if (shipFromCountry != "US") {
            $('select[ng-model="vm.currentShipment.Packages[vm.packageIndex].Weight.Units"]').val('string:KG').change();
            $('select[ng-model="vm.currentShipment.Packages[vm.packageIndex].Dimensions.Units"]').val('number:1').change();
        }
    }

        var today = new Date();

        if (shipDate.getDay() == 6) {
            throw ({ message: 'Saturday shipping is not allowed. Code[CBR]"', errorCode: '115' });
        }
        
        if (shipDate.getDay() == 7) {
            throw ({ message: 'Sunday shipping is not allowed. Code[CBR]"', errorCode: '115' });
        }
        

        if (shipmentRequest.Packages[0].ShipperReference === undefined || shipmentRequest.Packages[0].ShipperReference === null || shipmentRequest.Packages[0].ShipperReference === '') {
            throw ({ message: 'Study Reference is required.  Please select a valid Study Group in order to continue. [CBR ShipperReference]', errorCode: '115' });
        }
        if (shipmentRequest.Packages[0].ConsigneeReference === undefined || shipmentRequest.Packages[0].ConsigneeReference === null || shipmentRequest.Packages[0].ConsigneeReference === '') {
            throw ({ message: 'Study Reference is required.  Please select a valid Study Group in order to continue. [CBR ConsigneeReference]', errorCode: '115' });
        }

        var diw = shipmentRequest.Packages[0].DryIceWeight;
        var pkgw = shipmentRequest.Packages[0].Weight;

        if (diw.Amount > 0) {
            if (diw.Amount < 10) {

    function ApplyRulesWithServiceRequested(shipmentRequest,rateResults,dateToday)
    {
        var requestedServiceSymbol = shipmentRequest.PackageDefaults.Service.Symbol;
        var backupNDADomesticService = shipmentRequest.Packages[0].MiscReference18; //UPSAPI.UPS.NDA        
        var backupExpressIntlService = shipmentRequest.Packages[0].MiscReference20; //UPSAPI.UPS.EXPRESS  
        var serviceToBeReturned = '';
        var domestic = shipmentRequest.PackageDefaults.Consignee.Country === "US";

        rateResults.slice()
                   .forEach(function(item, index, object) 
        {            
            if (!requestedServiceSymbol) 
            {
                requestedServiceSymbol = shipmentRequest.PackageDefaults.Service.Symbol;
            }

            serviceToBeReturned = requestedServiceSymbol;

            var requestedServiceIsFoundAsRateResult = (item.PackageDefaults.Service.Symbol === shipmentRequest.PackageDefaults.Service.Symbol);

            if (requestedServiceIsFoundAsRateResult) 
            {                                           
                console.log("Found requested service in position: " + index + "of rate results");

                var applyBackupService = BackupServiceRequired(rateResults,shipmentRequest);

                if (applyBackupService)
                {
                    if (domestic) 
                    {
                        serviceToBeReturned = backupNDADomesticService;                

                        UpdateFields(formatDate(getTomorrowDate(),true));  
                    }                
                    else
                    {
                        serviceToBeReturned = backupExpressIntlService;

                        rateResults.forEach(function(item, index, object) 
                        {
                            if (item.PackageDefaults.Service.Symbol === backupExpressIntlService) 
                            {
                                UpdateFields(formatDate(GetArriveDate(item.PackageDefaults.ArriveDate)),applyBackupService);                  
                            }
                        });                       
                    }
                }
                else
                {
                    if (rateResults) 
                    {
                        var fullArriveDate = GetArriveDate(rateResults[0].PackageDefaults.ArriveDate);

                        UpdateFields(fullArriveDate,false);                         
                    }
                }
            }
        });
    }

    this.PostRate = function(shipmentRequest, rateResults) {
        var upchargePercentage = 20;
        for (var i = 0; i < rateResults.length; i++) {
            // console.log(rateResults[i].PackageDefaults.ErrorMessage);
            if (rateResults[i].PackageDefaults.ErrorMessage == "Success") {
                //console.log(rateResults[i].PackageDefaults.Total.Amount);
                var upcharge = (rateResults[i].PackageDefaults.Total.Amount / 100.0) * upchargePercentage;

                var specialAmount = rateResults[i].PackageDefaults.Special.Amount + upcharge;

                //rounding to 2 decimal places
                rateResults[i].PackageDefaults.Special.Amount = Math.round((specialAmount + Number.EPSILON) * 100) / 100;

                if (rateResults[i].PackageDefaults.Special.Currency === null)
                    rateResults[i].PackageDefaults.Special.Currency = rateResults[i].PackageDefaults.Total.Currency;

                var totalAmount = rateResults[i].PackageDefaults.Total.Amount + upcharge;

                //rounding to 2 decimal places
                rateResults[i].PackageDefaults.Total.Amount = Math.round((totalAmount + Number.EPSILON) * 100) / 100;


                for (var j = 0; j < rateResults[i].Packages.length; j++) {

                    //console.log(rateResults[i].Packages[j].ApportionedTotal.Amount);
                    var upcharge = (rateResults[i].Packages[j].ApportionedTotal.Amount / 100.0) * upchargePercentage;

                    var apportionedSpecialAmount = rateResults[i].Packages[j].ApportionedSpecial.Amount + upcharge;

                    //rounding to 2 decimal places
                    rateResults[i].Packages[j].ApportionedSpecial.Amount = Math.round((apportionedSpecialAmount + Number.EPSILON) * 100) / 100;

                    if (rateResults[i].Packages[j].ApportionedSpecial.Currency === null)

        }

var GetAddressBookEntries = function (addressBookName, context, shipper) {
        var searchCriteria = { WhereClauses: [{ FieldName: "Name", FieldValue: addressBookName, Operator: 0 }] };
        var data = { CompanyId: context.CompanyId, UserId: context.UserId };
        var response = thinClientAPIRequest('GetAddressBooks', data, false).responseJSON;
        if (response.TotalRecords == 0) {
            throw ({ message: 'Address Book:"' + addressBookName + '" does not exists'});
        } else {
            var addressBookId = response.AddressBooks[0].Id;
            
            searchCriteria = { Skip: 0, WhereClauses: [{FieldName: "Code", FieldValue: shipper, Operator: 0}], OrderByClauses: [] };
            data = { AddressBookIds: [addressBookId], SearchCriteria: searchCriteria, UserContext :{Machine:""} , CompanyId: context.CompanyId, UserId: context.UserId };
            response = thinClientAPIRequest('GetAddressBookEntries', data, false).responseJSON;
            
            if (response.TotalRecords == 0) {
                throw ({ message: 'Address Book:"' + addressBookName + '" does not have any address entry'});
            } else {
                return response.AddressBookEntries;
            }
        }
    }

// GetNewTableMarkup(OrderData) // returns rows
  GetNewTableMarkup = (OrderData) =>
  {
    let that = this, rows = [];
    const json = new JsonData();

    //loop through the data and create the object that will be attached to each row as a .data() object
    OrderData.forEach((item, index) =>
    {
      json.TrimAllKeysAndValues(OrderData);

      //setup the row data object
      let data = {};
      //the .Count var is used when the items get placed into containers, it represents the count of items in the parent container object
      data.Count = 0;
      data.ItemNumber = item.CSpartno;
      data.ItemDescription = item.CSdescript;
      data.HarmonizedCode = (item.CSHTScode || 'Not Available');
      data.HarmonizedDescription = (item.CSHTSDESC || 'Not Available');
      data.CountryOfOrigin = (item.CSCOO || 'Not Available');
      data.OrderNumber = item.CSsono;
      data.OrderRevision = item.CSsorev;
      data.OrderString = item.CSsono + ` (${item.CSsorev})`;
      data.PurchaseOrder = item.fcustpono;
      data.QuantityOrdered = +item.CSshITMqty;
      data.QuantityPacked = 0;
      data.QuantityRemaining = +item.CSshITMqty;
      data.RowId = (Math.random() + 1).toString(36).substring(2);
      data.RowIndex = index;
      data.PackedItems = []; //.PackedItems will be {} objects with a structure of .ContainerIndex = x, .Count = y

      let row = $('tr.table-row-template').clone()
        .removeClass('table-row-template')
        .attr('row-id', data.RowId)
        .data(data);

      //add all the data to the table row
      $(row).find('td').each(function(i, td)
      {
        let key = $(td).attr('sos-key');
        $(td).find('div:first').text(data[key]);
      });

      //setup the tooltip for the description
      let title = `<div class="text-left"><span>HTS: <b>${data.HarmonizedCode}</b></span><br/>`;
      title += `<span>C/O: <b>${data.CountryOfOrigin}</b></span></div>`;
      $(row).find('td.item-description').first().attr({ 'data-toggle': "tooltip", 'data-placement': 'top', 'data-html': 'true', 'data-container': 'body', 'title': title });

      rows.push(row);
    });

    return rows;
  }

    this.UpdateNameAddressCSZ = function (addressName, city, stateProvince) {

            nameAddress = this.PackageDefaults.ReturnAddress;
        }
        nameAddress.City = city;
        nameAddress.StateProvince = stateProvince;

            nameAddress.Country = 'PR';
        }
        else {
            nameAddress.Country = 'US';
        }
    }

// this.PostalCodeLookup(addressName) // returns undefined
    this.PostalCodeLookup = function (addressName) {
        var userContext = this.getUserContext(this);
        var postalCode = $('[address-property-name="' + addressName + '"] [name="postalCode"]').val();
        var content = { AddressName: addressName, PostalCode: postalCode };
        var payload = { Action: 'CSZ', Content: JSON.stringify(content) };
        var data = { Data: JSON.stringify(payload), UserContext: userContext };
        var response = this.thinClientAPIRequest('UserMethod', data, false).responseJSON;
        if (response.ErrorCode > 0) {
            client.shipping.alertModalShow(this, 'Postal Code Search', response.ErrorMessage)
        } else {
            var responseData = JSON.parse(client.htmlText(response.Data));
            if (responseData === null) {
                client.shipping.alertModalShow(this, 'Postal Code Search', 'Invalid or Not US Postal Code');
            } else {
                if (responseData.length > 1) {
                    this.addressResults = responseData;
                    client.shipping.validateAddressModalShow();
                } else {
                    this.currentShipment.updateNameAddressCSZ(addressName, responseData[0].City, responseData[0].StateProvince)
                }
            }
        }
    }

                var DryIceWeight = {
                    "Amount": ret.replace('kgs',''),
                    "Units": "KG"
                };
				pkg.DryIceWeight = DryIceWeight;
				
				var Country = shipmentRequest.PackageDefaults.Consignee.Country;
				console.log(Country);
				if(Country =="US"){
				    pkg.DryIceRegulationSet = 1;
				}else{
				    pkg.DryIceRegulationSet = 2;
				}
                pkg.DryIcePurpose = 2;
            }
            pkg.ShipperReference = shipperRef;
            pkg.ConsigneeReference = consigneeRef;
            pkg.MiscReference20 = userContext.UserId;

            pkg.ShipNotificationAddressEmail = '';
            pkg.ShipNotificationEmail = shipToEmail != '' || shipFromEmail != '';
            if (shipToEmail != '') {
                pkg.ShipNotificationAddressEmail = shipToEmail;
            }
