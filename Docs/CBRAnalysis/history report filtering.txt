        function history() {
            vm.history = {
                searchButton: 'button:contains("Search")',

            };

            $(document).on('change', vm.history.reportSelector, selectReportChange);

            function activate() {
                if (vm.profile.Name != vm.thinClient.adminProfileName) {
                    vm.hideButtonExport = true;

                    let reports = [ 'Default', 'Detailed Report', 'Void Report', 'Summary Report' ];
                    vm.reports = vm.reports.filter(function(report) {
                        report.TemplateHtml = report.TemplateHtml.replace(/<span.*glyphicon-search.*<\/span>/g, '');
                        return reports.includes(report.Name);
                    });
                }
            }

            function selectReportChange(event) {
                if (vm.report?.Name == 'Summary Report') {
                    vm.hideButtonExport = true;
                    vm.history.carrierAcceptTableContainer = {
                        'height': $('.panel-body').height() + 'px',
                        'overflow-y': 'auto'
                    };

                    let payload = {
                        Action: 'summaryReport',
                        Data: {
                            startDate: vm.dtstart.toLocaleDateString('en-CA'),
                            endDate: vm.dtend.toLocaleDateString('en-CA'),
                            sites: getSites()
                        }
                    };
                    let data = { UserContext: undefined, Data: JSON.stringify(payload) };
                    thinClient.httpClient('UserMethod', data).then(function (ret) {
                        debugger;
                        vm.history.carrierAcceptReport = JSON.parse(atob(ret.Data));
                        vm.thinClient.loadData.show();
                    });
                } else {
                    vm.hideButtonExport = false;
                }

                function getSites() {
                    let sites = vm.UserInformation.SiteId ? vm.sites.filter(function(site) { return site.Id == vm.UserInformation.SiteId;}) : vm.sites;
                    return sites.map(function(site) { return { name: site.Name }; });
                }
            }
        }

            function getCurrentMonthYear(action) {

            function getNextMonthYearInFrench(dateStr) {
                var monthNamesFr = ['Janvier', 'Février', 'Mars', 'Avril', 'Mai', 'Juin', 'Juillet', 'Août', 'Septembre', 'Octobre', 'Novembre', 'Décembre'];
                var splitDate = dateStr.split(' ');
                var month = splitDate[0];
                var year = splitDate[1];
            
                var currentMonthIndex = monthNamesFr.indexOf(month);
                
                // Check if the current month is December.
                if (currentMonthIndex === 11) {
                    // Increment the year and reset the month to January.
                    year++;
                    month = monthNamesFr[0];
                } else {
                    // Get the next month.
                    month = monthNamesFr[currentMonthIndex + 1];
                }
                
                return month + ' ' + year;
            }


            function getPreviousMonthYearInFrench(dateStr) {
                var monthNamesFr = ['Janvier', 'Février', 'Mars', 'Avril', 'Mai', 'Juin', 'Juillet', 'Août', 'Septembre', 'Octobre', 'Novembre', 'Décembre'];
                var splitDate = dateStr.split(' ');
                var month = splitDate[0];
                var year = splitDate[1];
            
                var currentMonthIndex = monthNamesFr.indexOf(month);
                
                // Check if the current month is January.
                if (currentMonthIndex === 0) {
                    // Decrement the year and set the month to December.
                    year--;
                    month = monthNamesFr[11];
                } else {
                    // Get the previous month.
                    month = monthNamesFr[currentMonthIndex - 1];
                }
                
                return month + ' ' + year;
            }

    function selectCanadaForAllCountrySelects() {
        // Check if the flag is already true, which means we no longer need to change the selects
        if(canadaSelected || !isSelectedValueInShippersArray()) {
          return; // Exit the function if Canada has been selected in all dropdowns or if the site is not from Canada
        }
        
        // Get all select elements with the name 'Country'
        var selects = document.querySelectorAll('select[name="Country"]');
        var allSelected = true; // To keep track if all selects have been set to Canada
      
        // Iterate through the NodeList
        selects.forEach(function(select) {
          // Only set the value to 'string:CA' if it's not already set
          if (select.value !== 'string:CA') {
            select.value = 'string:CA';
            allSelected = false; // Not all selects have been set to Canada
      
            // Trigger change event for AngularJS
            if ('createEvent' in document) {
              var evt = document.createEvent('HTMLEvents');
              evt.initEvent('change', false, true);
              select.dispatchEvent(evt);
            } else {
              select.fireEvent('onchange');
            }
          }
        });
      
        // If all selects have been set to Canada, set the flag to true
        if (allSelected) {
          canadaSelected = true;
        }
      }
      

      function isSelectedValueInShippersArray() {
        // Use a more specific selector that targets the ng-model
        var selectElement = document.querySelector('select[ng-model="vm.currentShipment.PackageDefaults.Shipper"]');
        
        // Ensure the element exists
        if (!selectElement) {
          console.warn('Select element not found');
          return false;
        }
        
        // Get the index of the selected option
        var selectedIndex = selectElement.selectedIndex;
        
        // Ensure an option is selected
        if (selectedIndex < 0) {
          console.warn('No option is selected');
          return false;
        }
        
        // Get the selected option element
        var selectedOption = selectElement.options[selectedIndex];
        
        // Get the value of the selected option (and remove the "string:" prefix if present)
        var selectedValue = selectedOption.value.replace('string:', '');
        
        // Check if the selected value is in the array
        return StoreLocationsCanadaArray.includes(selectedValue);
      }

           get ToRef2() { return this.GetCustom("Custom2", "To"); }

           get ToRef4() { return this.GetCustom("Custom4", "To"); }
           get ToRef5() { return this.GetCustom("Custom5", "To"); }
           get ToRef6() { return this.GetCustom("Custom6", "To"); }
           get ToRef7() { return this.GetCustom("Custom7", "To"); }
           get ToRef8() { return this.GetCustom("Custom8", "To"); }
           get ToRef9() { return this.GetCustom("Custom9", "To"); }
           get ToRef10() { return this.GetCustom("Custom10", "To"); }

           GetCustom(fieldName, customDataType) {
              var returnValue = "";
              var customData = null;
              if (customDataType === "User") customData = this.ShipmentRequest.PackageDefaults.OriginAddress.CustomData;
              if (customDataType === "To") customData = this.ShipmentRequest.PackageDefaults.Consignee.CustomData;

              if (customData) {
                 customData.forEach(function (customField) {
                    if (fieldName === customField.Key) returnValue = customField.Value;
                 })
              }
              return returnValue;
           }
        }

        return new Shipment(shipmentRequest);
     }
