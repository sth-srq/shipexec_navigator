this.PreShip = function(shipmentRequest, userParams) {    
    // We will require the actual shipment to send weight and dims in kg/cm
    // However, because US Vendors are sending the product the client gui needs to be in Lb/In.
    
    // debugger;
    
    totalShipWeight = 0;
    totalVolumetricResult = 0;
    
    vm.currentShipment.Packages.forEach(pkg => {
        totalShipWeight += Number(pkg.Weight.Amount);
        totalVolumetricResult += ((Number(pkg.Dimensions.Length).toFixed(2)*Number(pkg.Dimensions.Width).toFixed(2)*Number(pkg.Dimensions.Height)).toFixed(2))/166;
    });
              
    totalShipWeight = Number(totalShipWeight).toFixed(2);
    console.log(totalShipWeight);
    console.log(totalVolumetricResult);
    if (totalShipWeight > 150)
    {
        alert("Total Shipment Weight Exceeds Shipment Weight Limit of 150lbs")
        throw new Error("Shipping halted due to Over Maximum Weight Rule " + totalShipWeight);
    }
    
    if (totalVolumetricResult > Number(maxvolumetric))
    {
        alert("Total Shipment Weight Exceeds Total Shipment Volumetric Weight Limit of " + Number(maxvolumetric))
        throw new Error("Shipping halted due to Total Shipment Volumetric Weight Rule " + Number(totalVolumetricResult).toFixed(2));
    }
    
    //kg = lbs*0.453592    
    vm.currentShipment.Packages.forEach(pkg => {
         if (pkg.Weight.Units === "LB"){
             pkg.Weight.Amount = Number(pkg.Weight.Amount * 0.453592).toFixed(2);
             pkg.Weight.Units = "KG";
         }
         
    });
    
  
    //cm=in/0.39370
    vm.currentShipment.Packages.forEach(pkg => {
         if (pkg.Dimensions.Units === 0){
             pkg.Dimensions.Length = Number(pkg.Dimensions.Length/0.39370).toFixed(2)
             pkg.Dimensions.Width = Number(pkg.Dimensions.Width/0.39370).toFixed(2)
             pkg.Dimensions.Height = Number(pkg.Dimensions.Height/0.39370).toFixed(2)
             pkg.Dimensions.Units = 1;
       }
         
    });
     
   if(Number(shipmentRequest.Packages[0].DeclaredValueCustoms.Amount).toFixed(2) > 2500 ){
          if(shipmentRequest.Packages[0].ExportInformationCode === null){
              shipmentRequest.Packages[0].ExportInformationCode = "OS";
          }
          if(shipmentRequest.Packages[0].AesTransactionNumber === null){
              shipmentRequest.Packages[0].SedMethod = 2;
              $('input[type=text][ng-model="shipmentRequest.Packages[this.vm.packageIndex].AesTransactionNumber"]').focus();
              throw new Error('Missing AES/ITN Number');
          } 
          
          if(shipmentRequest.Packages[0].AesTransactionNumber !== null){
              shipmentRequest.Packages[0].Comments = shipmentRequest.Packages[0].Comments + "\n ITN/AES Filer " + shipmentRequest.Packages[0].AesTransactionNumber;
             
          }
      }

// SWOGShipment() // returns void
    this.SWOGShipment = function () {
        // Custom function that fires when SWOG button is clicked
        // If Shipper is "MCD" "MCT" or "MCC" there is no validation done on SWOGS and then focus on Dimensions when completed
        
        // Get current Shipper from screen       
        var shipper = this.currentShipment.PackageDefaults.Shipper;        

        debugger;
        // Define initial SWOG array and counter
        var swogArray = ["SWOG_START"];
        var counter = 0;

        // If the masterOrderNumber has an initial SWOG value from PreLoad, push that into the swogArray before collecting new ones.  Set the counter to 1.
        if (masterOrderNumber != '') {
            swogArray.push(masterOrderNumber);
            swogArray = unique(swogArray);
            counter = 1;
        }

        do {
            // Prompt for SWOG data entry until END_SWOG is scanned
            var swog = window.prompt("Enter SWOG order number.  Scan END_SWOG Barcode when complete", "");
            debugger;
            // Make sure swog is not empty
            if (swog != '') {
                if (counter > 0 && swogArray.length > 1) {
                    // This check is only done when the Shipper is NOT MCD, MCT or MCC
                    if (shipper != "MCD" && shipper != "MCT" && shipper != "MCC") {
                        // Make sure the entry is valid (not 'END_SWOG' AND length is greater than 9, not the same as the first entry in the array)
                        if (swog != 'END_SWOG' && swog.length > 9 && swog.substring(0, 9) != swogArray[1].substring(0, 9)) {
                            alert("The order that you scanned = " + swog + " is invalid for this SWOG and will be removed.");
                            swog = '';
                        }
                    }
                }
                // If masterOrderNumber is not empty, this function was called from PreShip.  Otherwise it came from clicking SWOG
                if (masterOrderNumber != '') {
                    // function called from PreShip...validate length of swog
                    if ((swog != '' && swog.length > 9) || swog == 'END_SWOG') {
                        // Add the swog to the swogArray
                        swogArray.push(swog);
                        // Make sure that only unique entries are in the swogArray
                        swogArray = unique(swogArray);
                        counter++;
                    }
                }
                else {
                    // function called from clicking button...do not validate length of swog
                    if (swog != ''  || swog == 'END_SWOG') {
                        // Add the swog to the swogArray
                        swogArray.push(swog);
                        // Make sure that only unique entries are in the swogArray
                        swogArray = unique(swogArray);
                        counter++;
                    }
                }                
            }
        } while (swog != 'END_SWOG');        

        // Put '|' between each SWOG number
        var swogOrderNumbers = "";
        swogArray.forEach(function (item, index, array) { swogOrderNumbers = swogOrderNumbers + "|" + item; });
        swogOrderNumbers = removeFirstChar(swogOrderNumbers);

        // Store final swog data in UserData1
        //this.currentShipment.Packages[0].ShipperReference = 123;
        //this.currentShipment.Packages[0].ConsigneeReference = swogOrderNumbers;
        this.currentShipment.Packages[0].UserData1 = swogOrderNumbers;

        // If Shipper IS MCD, MCT or MCC then put focus on length field in dimensions
        if (shipper == "MCD" || shipper == "MCT" || shipper == "MCC") {
            lengthText = ('[ng-model="vm.currentShipment.Packages[vm.packageIndex].Dimensions.Length"]');
            setFocus(lengthText);
        }
        // Clear out masterOrderNumber
        masterOrderNumber = "";
        debugger;
    }

// PreLoad(loadValue, shipmentRequest, userParams) // returns void
  this.PreLoad = function (loadValue, shipmentRequest, userParams) {

    // 🔄 Reset any previous SWOG starter
    masterOrderNumber = "";

    // 🧾 First prompt: general input
    var firstInput = window.prompt(
        'Enter Package Dimensions (###x###x###)\n' +
        '— OR type PACKSIZE_CONTINUE to skip dimensions\n' +
        '— OR enter a SWOG Order Number to start the SWOG function:',
        ''
    );

    if (firstInput === null) {
        userParams.dimensions = "";
        return;  // ❌ Cancelled
    }

    firstInput = firstInput.trim();
    var dimRegex = /^[0-9]+\s*x\s*[0-9]+\s*x\s*[0-9]+$/i;

    if (firstInput.toUpperCase() === 'PACKSIZE_CONTINUE') {
        userParams.dimensions = 'PACKSIZE_CONTINUE';
        return;  // ✅ Skip dimensions
    }

    if (dimRegex.test(firstInput)) {
        userParams.dimensions = firstInput.toLowerCase();
        return;  // ✅ Valid dimensions directly
    }

    // 🌀 Else: treat as SWOG starter
    masterOrderNumber = firstInput;
    this.vm.swogShipment();

    // 🧾 Second prompt after SWOG
    var dimsAfterSwog = window.prompt(
        'Enter Package Dimensions (###x###x###), type PACKSIZE_CONTINUE to skip, or leave blank:',
        ''
    );

    if (dimsAfterSwog === null) {
        userParams.dimensions = "";  // ❌ Cancelled
        return;
    }

    dimsAfterSwog = dimsAfterSwog.trim();

    if (dimsAfterSwog.toUpperCase() === 'PACKSIZE_CONTINUE') {
        userParams.dimensions = 'PACKSIZE_CONTINUE';  // ✅ PACKSIZE_CONTINUE accepted in second prompt too
    } else if (dimRegex.test(dimsAfterSwog)) {
        userParams.dimensions = dimsAfterSwog.toLowerCase();  // ✅ Dimensions entered
    } else {
        userParams.dimensions = "";  // ❌ Nothing valid entered
    }
}

            function getShipperCustomers(shippers) {
                var shipper = shippers.find(shipper => shipper.Symbol == vm.shipping.slic);
                if (shipper == undefined)
                     return [{ name: '--- No Shipper Configured ---', disabled: true }];

                var customData = shipper.CustomData.filter(c => c.Key.toUpperCase() == "CUSTOMER");
                var customers = customData.map(function(c) {
                    var customer = c.Value.toUpperCase();

                    if (customer == 'AMAZON')
                        return { name: "Amazon", value: customer, hasLoad: true, hasReturn: true, autoShip: true };
                    else if (customer == 'UPS')
                        return { name: "UPS", value: customer, hasCameraImage: true, hasFileImage: true, autoShip: false };
                    else if (customer == 'DEMO')
                        return { name: "Demo", value: customer, hasLoad: false, hasReturn: true, hasCameraImage: true, hasFileImage: false, autoShip: false };
                    else if (customer == 'FANATICS')
                        return { name: "Fanatics", value: customer, hasReturn: true, hasCameraImage: true, autoShip: false };
                    else
                        return { name: `--- ${customer} ---`, disabled: true };
                });

                customers.sort(function(a, b) {
                    if (a.disabled && !b.disabled) return 1;
                    if (!a.disabled && b.disabled) return -1;
                    return a.name.localeCompare(b.name);
                });
                return [{ name: '--- Select Customer Option ---', selector: 'select[ng-model="vm.shipping.customer"]', disabled: true }].concat(customers);
            }

this.PostLoad = function(value, shipmentRequest) {
		debugger;
		let errorMessage = shipmentRequest.PackageDefaults.ErrorMessage;
		
		if (errorMessage === undefined) errorMessage = "";
		if (errorMessage === null) errorMessage = "";

		if (shipmentRequest.PackageDefaults.Service !== undefined && shipmentRequest.PackageDefaults.Service.Symbol === "x") {
			showLoading(false);
			alert("No service selected!");
		}
		
		if (errorMessage !== undefined && errorMessage === 'validate') {
			//$("#ctl00_cphContent_left_tabs_consignee_tab_consignee_na_ab_ctrl_na_ctrl_validate_addr_btn").click();
			showLoading(false);
			return;
		}

		if (errorMessage !== undefined && errorMessage.substring(0, 4) === "pop:") {
			showLoading(false);
			alert(errorMessage.substring(4));
			return;
		}

		if (errorMessage !== undefined && errorMessage !== '') {
			showLoading(false);
			alert(errorMessage);
			let errors = ["not found", "already shipped", "too many"];
			if (errors.some(errorText => errorMessage.includes(errorText))) {
				// open Load Shipment dialog
				this.vm.newShipment();
			}
			return;
		}

		if (shipmentRequest.PackageDefaults.Consignee.Country === 'US') {
			this.vm.shipMode = 1;
		} else {
			this.vm.shipMode = 0;
		}

		$("#custom_markup_id").val(shipmentRequest.Packages[0].MiscReference3);

		//getWeight();

		if (shipmentRequest.PackageDefaults.MiscReference5 !== null && shipmentRequest.PackageDefaults.MiscReference5 === 'autoship') {
			this.vm.checkShip();
			return;
		}

		// select all services before rate
		setTimeout(function() {
			// check Select All after delay to allow UI update
			vmTemp.isAllServicesSelected = true;
			vmTemp.selectAllServices();
			vmTemp.rate();
		}, 100);
		
	};

// this.PreShip(shipmentRequest, userParams) // returns undefined
this.PreShip = function (shipmentRequest, userParams) {
		console.log("🚀 [PreShip] Starting PreShip function...");
	    shipmentRequest = vmTemp.currentShipment;
		// Validate shipmentRequest and Packages array
		if (!shipmentRequest || !shipmentRequest.Packages || !Array.isArray(shipmentRequest.Packages)) {
			console.error("❌ [PreShip Error] Shipment request is invalid or missing required properties.");
			//alert("[PreShip Error] Shipment request is invalid or missing required properties.");
			return;
		}
	
		let validDimensions = [];
	
		console.log(`📦 [PreShip] Processing ${shipmentRequest.Packages.length} package(s)...`);
	
		// Retrieve current package dimensions from vmTemp
		try {
			let currentPackage = vmTemp.currentShipment.Packages[vmTemp.packageIndex];
			if (!currentPackage || !currentPackage.Dimensions) {
				console.warn("⚠️ [PreShip Warning] vmTemp.currentShipment package or dimensions not found.");
			} else {
				let dims = currentPackage.Dimensions;
				console.log(`📏 [PreShip] Current Package Dimensions from vmTemp: ${dims.Length}x${dims.Width}x${dims.Height} 0`);
			}
		} catch (error) {
			console.error("❌ [PreShip Error] Unable to retrieve package dimensions from vmTemp.");
		}
	
		// Loop through all packages to validate their dimensions
		for (let i = 0; i < shipmentRequest.Packages.length; i++) {
			console.log(`🔍 [PreShip] Checking package at index ${i}...`);
	
			let packageData = shipmentRequest.Packages[i];
	
			if (!packageData) {
				console.error(`❌ [PreShip Error] Package at index ${i} is undefined.`);
				//alert(`[PreShip Error] Package at index ${i} is undefined.`);
				return;
			}
	
			if (!packageData.Dimensions) {
				console.warn(`⚠️ [PreShip Warning] Package at index ${i} does not contain 'Dimensions'.`);
				//alert(`[PreShip Error] Package at index ${i} does not contain 'Dimensions'.`);
				return;
			}
	
			let dims = packageData.Dimensions;
	
			// Ensure all dimension values exist and are valid numbers
			if (dims.Length > 0 && dims.Width > 0 && dims.Height > 0) {
				let dimensionString = `${dims.Length}x${dims.Width}x${dims.Height} 0`; // Always append "0"
	
				// Validate format (three numeric values + "0" at the end)
				let dimParts = dimensionString.split(new RegExp("[x ]"), 4); // Split on 'x' and space
	
				if (dimParts.length === 4 && !isNaN(dimParts[0]) && !isNaN(dimParts[1]) && !isNaN(dimParts[2]) && dimParts[3] === "0") {
					validDimensions.push(dimensionString);
					console.log(`✅ [PreShip] Package ${i} Dimensions: ${dimensionString}`);
				} else {
					console.error(`❌ [PreShip Error] Package at index ${i} has incorrectly formatted dimensions: ${dimensionString}`);
					//alert(`[PreShip Error] Package at index ${i} has incorrectly formatted dimensions.`);
					return;
				}
			} else {
				console.error(`❌ [PreShip Error] Package at index ${i} has invalid dimensions.`);
				//alert(`[PreShip Error] Package at index ${i} has invalid dimensions.`);
				return;
			}
		}
	
		// Store only valid dimensions in MiscReference5
		if (validDimensions.length > 0) {
			shipmentRequest.PackageDefaults.MiscReference5 = validDimensions.join(", ");
			console.log(`✅ [PreShip] All dimensions successfully stored in MiscReference5: ${shipmentRequest.PackageDefaults.MiscReference5}`);
		} else {
			console.warn("⚠️ [PreShip Warning] No packages had valid dimensions.");
		}
	
		// Retrieve weight and markup values
		userParams.weight = $('div[caption="Weight"]').find('input[name="unitValue_txt"]').val();
		userParams.markup = $("#custom_markup_id").val();
		console.log(`📏 [PreShip] Retrieved weight: ${userParams.weight}, Markup: ${userParams.markup}`);
	
		// Custom code to hide/filter on postship errors - must be started here
		$("<style> .alert-danger { display: none; } </style>").appendTo("head");
		console.log("🚫 [PreShip] Hidden alert-danger elements to prevent UI issues.");
	
		alertInterval = setInterval(function () {
			var errAlert = $('div.alert-danger');
		
			// Filter and show alert
			if (errAlert.length > 0) {

        function SetShipNotificationAddressEmail(pkg, ngModelName, emailAddress, checked) {
        if (checked) {
            if (pkg.ShipNotificationAddressEmail == undefined || pkg.ShipNotificationAddressEmail.trim() == '') {
                if (ngModelName == 'ShipFromNotificationEmail' && emailAddress.Origin) {
                    pkg.ShipNotificationAddressEmail = emailAddress.Origin;
                } else if (ngModelName == 'ShipToNotificationEmail' && emailAddress.Consignee) {
                    pkg.ShipNotificationAddressEmail = emailAddress.Consignee;
                }
            } else {
                if (ngModelName == 'ShipFromNotificationEmail' && emailAddress.Origin) {
                    var regexp = new RegExp(',?' + emailAddress.Origin, 'g');
                    if (!regexp.test(pkg.ShipNotificationAddressEmail)) {
                        pkg.ShipNotificationAddressEmail += ',' + emailAddress.Origin;
                    }
                } else if (ngModelName == 'ShipToNotificationEmail' && emailAddress.Consignee) {
                    pkg.ShipNotificationAddressEmail += ',' + emailAddress.Consignee;
                }
            }
        } else {
            var notificationAddressEmail = pkg.ShipNotificationAddressEmail;
            if (ngModelName == 'ShipToNotificationEmail' && emailAddress.Consignee) {
                var regexp = new RegExp(',?' + emailAddress.Consignee, 'g')
                notificationAddressEmail = notificationAddressEmail.replace(regexp, '');
            } else if (ngModelName == 'ShipFromNotificationEmail' && emailAddress.Origin) {
                var regexp = new RegExp(',?' + emailAddress.Origin, 'g')
                notificationAddressEmail = notificationAddressEmail.replace(regexp, '');
            }

            if (notificationAddressEmail.indexOf(',') == 0) {
                notificationAddressEmail = notificationAddressEmail.substring(1);
            }
            pkg.ShipNotificationAddressEmail = notificationAddressEmail;
        }
    }

    function checkOriginConsigneeCountry() {
        var shipFromCountry = getObjectValue(vm.currentShipment.PackageDefaults.OriginAddress.Country);
        var shipToCountry = getObjectValue(vm.currentShipment.PackageDefaults.Consignee.Country);

        /* If we have no shipToCountry OR the shipFromCountry is either US or CA, disable the elements, return out. */
        if (!shipToCountry | shipFromCountry == "US" | shipFromCountry == "CA") {
            DisableThirdPartyBillingElements();
            return false;
        }

        if (shipFromCountry != shipToCountry) {
            /* If the shipFromCountry does NOT equal the shipToCountry then ship Third Party. */
            SetThirdPartyBilling(vm.currentShipment);
        }
        else {
            /* Now check to see if both the origin and destination countries are members of the EU. */
            if ((IsMemberOfEu(shipFromCountry)) && (IsMemberOfEu(shipToCountry))) {
                SetThirdPartyBilling(vm.currentShipment);
            }

            //ShipThirdPartyBilling((IsMemberOfEu(shipFromCountry)) && (IsMemberOfEu(shipToCountry)));
        }

        if (shipFromCountry != "US") {
            $('select[ng-model="vm.currentShipment.Packages[vm.packageIndex].Weight.Units"]').val('string:KG').change();
            $('select[ng-model="vm.currentShipment.Packages[vm.packageIndex].Dimensions.Units"]').val('number:1').change();
        }
    }

        var shipToCountry = getObjectValue(vm.currentShipment.PackageDefaults.Consignee.Country);

        /* If we have no shipToCountry OR the shipFromCountry is either US or CA, disable the elements, return out. */

            DisableThirdPartyBillingElements();

                SetThirdPartyBilling(vm.currentShipment);
