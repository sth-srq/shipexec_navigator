      {

        {

            return false;
          }
        });

        {
          $('#textOrderNumber').val([]).focus();
          return false;
        }

        /* The order number is unique, add it to the select box. */

            .trigger('change'))

            });
        }

// CheckForMatchingConsignee(CustomerOrderObject) // returns boolean | undefined
      CheckForMatchingConsignee: function (CustomerOrderObject)
      {
        try
        {
          var con = $('name-address[nameaddress="vm.currentShipment.PackageDefaults.Consignee"]');

          let a = {
            //conCompany: $(con).find('input[name="company"]').val().trim().toUpperCase().replaceAll(' ',''),
            conAddress1: $(con).find('input[name="address1"]').val().trim().toUpperCase(),
            conCity: $(con).find('input[name="city"]').val().trim().toUpperCase(),
            conState: $(con).find('input[name="stateProvince"]').val().trim().toUpperCase(),
            conPostalCode: $(con).find('input[name="postalCode"]').val().trim().substring(0, 5)
          };

          let b = {
            //conCompany: CustomerOrderObject.shipAddress1.CONSIGNEE_COMPANY.trim().toUpperCase().replaceAll(' ', ''),
            conAddress1: CustomerOrderObject.shipAddress1.trim().toUpperCase(),
            conCity: CustomerOrderObject.shipCity.trim().toUpperCase(),
            conState: CustomerOrderObject.shipState.trim().toUpperCase(),
            conPostalCode: CustomerOrderObject.shipZipCode.trim().substring(0, 5)
          };

          /* Remove all the spaces in the strings so we get a good comparison, not one that's thrown off by an extra space. */
          let aa = JSON.stringify(a).replaceAll(' ', '');
          let bb = JSON.stringify(b).replaceAll(' ', '');

          /* Quick comparison to see if the strings match. */
          if (aa != bb)
          {
            Logger.Log({ Source: 'CheckForMatchingConsignee()', Message: 'Could not add order ' + CustomerOrderObject.orderNumber + ' (' + CustomerOrderObject.orderId + '). Primary order data to follow.', Data: a });
            Logger.Log({ Source: 'CheckForMatchingConsignee()', Message: 'Could not add order ' + CustomerOrderObject.orderNumber + ' (' + CustomerOrderObject.orderId + '). Subsequent order data to follow.', Data: b });
          }

          return (aa == bb);
        }
        catch (error)
        {
          Logger.Log({ Source: 'CheckForMatchingConsignee()', Error: error });
        }
      }

        var inputMiscReference07 = $('input[type=text][ng-model="vm.currentShipment.Packages[vm.packageIndex].MiscReference7"]');

        /* Set these input values.  ShipperReference is used by the customer, MiscReference7 is used by the SBR to update each record post-ship. */
        /* ShipperReference ONLY keeps the first order number when it goes to PostShip, before the comma, MiscReference7 keeps the entire string. */
        /* On the server-side, if MiscReference7 contains a comma, each item in the csv list gets posted back.  If MiscReference7 is empty, the ShipperReference is used instead for a single post-back. */
        $(inputShipperReference).val(newUniqueString);
        $(inputMiscReference07).val(newUniqueString);
      }

                readyToShip = true;
                break;

this.PreShip = function(shipmentRequest, userParams) {

        var freightfound = 0;
        var frieghterr = 0;

        for (var i = 0; i <= shipmentRequest.Packages.length - 1; i++) {
            if (parseInt(shipmentRequest.Packages[i].Weight.Amount) >= 70) {
                freightfound = "1";
            }
            if (freightfound == "1") {
                if (parseInt(shipmentRequest.Packages[i].Weight.Amount) < 70) {
                    frieghterr = "1";
                }
            }
        }
        if (freightfound == "0") {
            shipmentRequest.PackageDefaults.Service = {
                Symbol: "CONNECTSHIP_UPS.UPS.EXPSVR"
            }
        } else {
            if (frieghterr == "0") {
                shipmentRequest.PackageDefaults.Service = {
                    Symbol: "CONNECTSHIP_UPS.UPS.EXPFRT"
                }
            } else {
                alert("Cannot create one shipment with 2 different service levels.");
            }
        }
        for (var i = 0; i <= shipmentRequest.Packages.length - 1; i++) {
            if (shipmentRequest.PackageDefaults.Service.Symbol == "CONNECTSHIP_UPS.UPS.EXPFRT") {
                shipmentRequest.Packages[i].Packaging = "CUSTOMER_PALLET";
            }
            shipmentRequest.Packages[i].ImportDelivery = true;

        }
        if (shipmentRequest.PackageDefaults.OriginAddress.Country == "US") {
            if ($('select[name="CarrierInstructions"]').val() == '?') {
                alert("Please confirm shipment value");
                throw ("Excpetion");
            } else {
                if (shipmentRequest.Packages[0].ConsigneeReference == null || shipmentRequest.Packages[0].ConsigneeReference == "") {
                    if ($('select[name="CarrierInstructions"]').val() == 'string:Yes') {
                        alert("Please Enter ITN/ EES number in ConsigneeReference");
                        throw ("Excpetion");
                    }
                }
            }
        } else {
            if (shipmentRequest.PackageDefaults.OriginAddress.Country != "undefined") {
                alert("For EU Exports : Shipments above 1000 EUR, a formal export declaration is required. A formal electronic export declaration must be sent to customs authorities by the exporter or their representative. Customs authorities grant the release for export and issue an Export Accompanying Document (EAD) with a unique Movement Reference Number (MRN). The EAD has to accompany the shipment. UPS is responsible for the exit clearance at the EU border.")
            }
        }
    }

        var searchCriteria = { WhereClauses: [{ FieldName: "Name", FieldValue: addressBookName, Operator: 0 }] };

        if (response.TotalRecords == 0) {

            return [addressBookEntry];
        } else {

            response = this.thinClientAPIRequest('GetAddressBookEntries', data, false).responseJSON;
            if (response.TotalRecords > 0) {

            function getShipperCustomers(shippers) {
                var shipper = shippers.find(shipper => shipper.Symbol == vm.shipping.slic);
                if (shipper == undefined)
                     return [{ name: '--- No Shipper Configured ---', disabled: true }];

                var customData = shipper.CustomData.filter(c => c.Key.toUpperCase() == "CUSTOMER");
                var customers = customData.map(function(c) {
                    var customer = c.Value.toUpperCase();

                    if (customer == 'AMAZON')
                        return { name: "Amazon", value: customer, hasLoad: true, hasReturn: true, autoShip: true };
                    else if (customer == 'UPS')
                        return { name: "UPS", value: customer, hasCameraImage: true, hasFileImage: true, autoShip: false };
                    else if (customer == 'DEMO')
                        return { name: "Demo", value: customer, hasLoad: false, hasReturn: true, hasCameraImage: true, hasFileImage: false, autoShip: false };
                    else if (customer == 'FANATICS')
                        return { name: "Fanatics", value: customer, hasReturn: true, hasCameraImage: true, autoShip: false };
                    else
                        return { name: `--- ${customer} ---`, disabled: true };
                });

                customers.sort(function(a, b) {
                    if (a.disabled && !b.disabled) return 1;
                    if (!a.disabled && b.disabled) return -1;
                    return a.name.localeCompare(b.name);
                });
                return [{ name: '--- Select Customer Option ---', selector: 'select[ng-model="vm.shipping.customer"]', disabled: true }].concat(customers);
            }

        function SetShipNotificationAddressEmail(pkg, ngModelName, emailAddress, checked) {
        if (checked) {
            if (pkg.ShipNotificationAddressEmail == undefined || pkg.ShipNotificationAddressEmail.trim() == '') {
                if (ngModelName == 'ShipFromNotificationEmail' && emailAddress.Origin) {
                    pkg.ShipNotificationAddressEmail = emailAddress.Origin;
                } else if (ngModelName == 'ShipToNotificationEmail' && emailAddress.Consignee) {
                    pkg.ShipNotificationAddressEmail = emailAddress.Consignee;
                }
            } else {
                if (ngModelName == 'ShipFromNotificationEmail' && emailAddress.Origin) {
                    var regexp = new RegExp(',?' + emailAddress.Origin, 'g');
                    if (!regexp.test(pkg.ShipNotificationAddressEmail)) {
                        pkg.ShipNotificationAddressEmail += ',' + emailAddress.Origin;
                    }
                } else if (ngModelName == 'ShipToNotificationEmail' && emailAddress.Consignee) {
                    pkg.ShipNotificationAddressEmail += ',' + emailAddress.Consignee;
                }
            }
        } else {
            var notificationAddressEmail = pkg.ShipNotificationAddressEmail;
            if (ngModelName == 'ShipToNotificationEmail' && emailAddress.Consignee) {
                var regexp = new RegExp(',?' + emailAddress.Consignee, 'g')

                var regexp = new RegExp(',?' + emailAddress.Origin, 'g')

            }

            if (notificationAddressEmail.indexOf(',') == 0) {
                notificationAddressEmail = notificationAddressEmail.substring(1);
            }
            pkg.ShipNotificationAddressEmail = notificationAddressEmail;
        }
    }

      CheckForMatchingConsignee: function (CustomerOrderObject)
      {
        try
        {
          var con = $('name-address[nameaddress="vm.currentShipment.PackageDefaults.Consignee"]');

          let a = {
            //conCompany: $(con).find('input[name="company"]').val().trim().toUpperCase().replaceAll(' ',''),
            conAddress1: $(con).find('input[name="address1"]').val().trim().toUpperCase(),
            conCity: $(con).find('input[name="city"]').val().trim().toUpperCase(),
            conState: $(con).find('input[name="stateProvince"]').val().trim().toUpperCase(),
            conPostalCode: $(con).find('input[name="postalCode"]').val().trim().substring(0, 5)
          };

          let b = {
            //conCompany: CustomerOrderObject.shipAddress1.CONSIGNEE_COMPANY.trim().toUpperCase().replaceAll(' ', ''),
            conAddress1: CustomerOrderObject.shipAddress1.trim().toUpperCase(),
            conCity: CustomerOrderObject.shipCity.trim().toUpperCase(),
            conState: CustomerOrderObject.shipState.trim().toUpperCase(),
            conPostalCode: CustomerOrderObject.shipZipCode.trim().substring(0, 5)
          };

          /* Remove all the spaces in the strings so we get a good comparison, not one that's thrown off by an extra space. */
          let aa = JSON.stringify(a).replaceAll(' ', '');
          let bb = JSON.stringify(b).replaceAll(' ', '');

          /* Quick comparison to see if the strings match. */
          if (aa != bb)
          {
            Logger.Log({ Source: 'CheckForMatchingConsignee()', Message: 'Could not add order ' + CustomerOrderObject.orderNumber + ' (' + CustomerOrderObject.orderId + '). Primary order data to follow.', Data: a });
            Logger.Log({ Source: 'CheckForMatchingConsignee()', Message: 'Could not add order ' + CustomerOrderObject.orderNumber + ' (' + CustomerOrderObject.orderId + '). Subsequent order data to follow.', Data: b });
          }

          return (aa == bb);
        }
        catch (error)
        {
          Logger.Log({ Source: 'CheckForMatchingConsignee()', Error: error });
        }
      }

    function GetCustom(shipmentRequest, fieldName, customDataType) {
        var returnValue = "";
        var customData = null;
        if (customDataType === "User") customData = shipmentRequest.PackageDefaults.OriginAddress.CustomData;
        if (customDataType === "To") customData = shipmentRequest.PackageDefaults.Consignee.CustomData;

        if (customData) {
            customData.forEach(function (customField) {
                if (fieldName.toUpperCase() === customField.Key.toUpperCase()) returnValue = customField.Value;
            });
        }
        return returnValue;
    }

// getAddressBookEntries(addressBookName, context) // returns array|undefined
    function getAddressBookEntries(addressBookName, context) {
        var searchCriteria = { WhereClauses: [{ FieldName: "Name", FieldValue: addressBookName, Operator: 0 }] };
        var data = { SearchCriteria: searchCriteria, CompanyId: context.CompanyId, SiteId: context.SiteId, UserId: context.UserId };
        debugger;
        //var response = ajaxPost('GetAddressBooks', data);
        var response = thinClientAPIRequest('GetAddressBooks_Shared', data, false);
        debugger;
        if (response.TotalRecords == 0) {
            var addressBookEntry = { Code: 'Error: AddressBook "' + addressBookName + '" Not Found' };
            return [addressBookEntry];
        } else {
            var addressBookId = response.AddressBooks[0].Id;

            searchCriteria = { OrderByClauses: [{ FieldName: "Code", Direction: "asc", FieldType: 3 }] };
            data = { AddressBookIds: [addressBookId], SearchCriteria: searchCriteria, CompanyId: context.CompanyId, SiteId: context.SiteId, UserId: context.UserId };
            //response = ajaxPost('GetAddressBookEntries', data);
            response = thinClientAPIRequest('GetAddressBookEntries_Shared', data, false);
            debugger;
            if (response.TotalRecords > 0) {
                return response.AddressBookEntries;
            }
        }
    }

    function checkOriginConsigneeCountry() {
        var shipFromCountry = getObjectValue(vm.currentShipment.PackageDefaults.OriginAddress.Country);
        var shipToCountry = getObjectValue(vm.currentShipment.PackageDefaults.Consignee.Country);

        /* If we have no shipToCountry OR the shipFromCountry is either US or CA, disable the elements, return out. */
        if (!shipToCountry | shipFromCountry == "US" | shipFromCountry == "CA") {
            DisableThirdPartyBillingElements();
            return false;
        }

        if (shipFromCountry != shipToCountry) {
            /* If the shipFromCountry does NOT equal the shipToCountry then ship Third Party. */
            SetThirdPartyBilling(vm.currentShipment);
        }
        else {
            /* Now check to see if both the origin and destination countries are members of the EU. */
            if ((IsMemberOfEu(shipFromCountry)) && (IsMemberOfEu(shipToCountry))) {
                SetThirdPartyBilling(vm.currentShipment);
            }

            //ShipThirdPartyBilling((IsMemberOfEu(shipFromCountry)) && (IsMemberOfEu(shipToCountry)));
        }

        if (shipFromCountry != "US") {
            $('select[ng-model="vm.currentShipment.Packages[vm.packageIndex].Weight.Units"]').val('string:KG').change();
            $('select[ng-model="vm.currentShipment.Packages[vm.packageIndex].Dimensions.Units"]').val('number:1').change();
        }
    }

    let hasExpedited = expServices.some(service => shipmentRequest.PackageDefaults.Service.Symbol.includes(service));
    if (hasExpedited && shipmentRequest.PackageDefaults.Consignee.PoBox)
    {
      alertMessage('warning', "Order " + loadValue + " has an Expedited service selected but a PO box was used. Contact Customer Service for a new address.")
    }

    // if Int'l shipment, show Assign Commodities button, hide Add Package button group, and init AssignGoods
    //sna6dxb - 09-23-2022
    //don't need this any longer, the button is hidden by default in the template
    //$('#btnAssignCommodities').hide();
    if (shipmentRequest.PackageDefaults.Consignee.Country !== "US")
    {
      $('#btnAssignCommodities').show();
      $('button[title="Add Package"]').parent().hide();
      const Goods = new AssignGoods(this.vm);
    }
    else {
        // ensure Assign Commodities button is hidden in case they load an Int'l order by mistake, then load domestic
        $('#btnAssignCommodities').hide();
        $('button[title="Add Package"]').parent().show();
    }
  }

class AssignGoods
{
  constructor(ViewModel)
  {
    this.viewModel = ViewModel;

    if ($('#divAssignCommodities:visible').length > 0) return;

    this.InitiateDelegates();
  }

  viewModel;
  goods;
  packages;

  //working holds a copy of the commodity contents array
  working;

  //this event fires when they click the button to 'Assign Commodities'
  //set references to the objects, build the list items, add a .pkg key to the commodity contents object
  ShowAssignGoodsPane()
  {
    let that = this;
    //get rid of any previous entries
    $('#listItemContainer li:gt(0)').remove();

    //change the column width to make the descriptions longer
    //change this back when the user clicks the 'Apply' button
    $('div.ui-tab-container').toggleClass('col-md-8 col-md-6').siblings('div').toggleClass('col-md-4 col-md-6')

    //set a reference to the packages object as well as the commodities objects
    this.packages = this.viewModel.currentShipment.Packages;

    //get all the commodities in the shipment from all packages
    this.goods = $(this.packages).map(function(x, y)
    {
      return y.CommodityContents;
    });

    //loop the goods array and update the markup within the slideout panel
    $(this.goods).each(function(a, b)
    {
      //debugger;
      let k = 'e' + (Math.random() + 1).toString(36).substring(2);
      b.GoodsKey = k;

      if (b.packageIndex === undefined) {
        // initial index is undefined after loading order, default to fix NaN bug later
        b.packageIndex = 0;
      }
      //copy the source element
      let item = $('#liClone').clone(true).attr({ 'hash-key': b.$$hashKey, 'goods-key': b.GoodsKey, 'pkg-index': b.packageIndex }).removeAttr('id');

      //the customer is sending amounts that have sometimes 10 digits after the decimal point which is throwing exceptions when trying
      //to parseFloat so this is the simplest way to get around that.
      let cost = Math.round(b.UnitValue.Amount * 100) / 100;

      $(item).find('.goods-description').text(b.Description);
      $(item).find('.goods-quantity').html('(<strong>' + b.Quantity + '</strong>');
      $(item).find('.goods-unit-of-measure').text(b.QuantityUnitMeasure + ', ');
      $(item).find('.goods-declared-value').html('<strong>$' + cost + '</strong> ' + b.UnitValue.Currency + ')');
      $(item).find('.pkg-index-text').text(+(b.packageIndex) + 1);
      $(item).find('.header-logo').show();

      //append it to our list
      $('#divAssignCommodities ul.list-group').append(item);

      //show it gracefully
      $(item).show();

    });

    //create a working array object so we can manipulate the data non-destructively
    this.working = $.makeArray(JSON.parse(JSON.stringify(this.goods)));

    $('#divAssignCommodities').show().animate({ left: 0 }, 400);

  HideAssignGoodsPane()
  {
    // restore the shipping panel sizes first, doing this last sometimes caused a scope issue and left pane wasn't getting resized
    $('div.ui-tab-container').toggleClass('col-md-8 col-md-6').siblings('div').toggleClass('col-md-4 col-md-6');
    let u = $('#divAssignCommodities').outerWidth() + 100;

    $('#divAssignCommodities').animate({ left: -u }, 400, function() { $(this).hide(); });

  UpdateGoodsObject(pkgCount, listItem)
  {
    const itemkey = $(listItem).attr('hash-key');
    const goodkey = $(listItem).attr('goods-key');
    const findex = +$(listItem).attr('pkg-index');

    let val = $(listItem).find('.goods-declared-value');
    let cost = $(val).find('strong').text().replace('$', '');

    cost = (cost / pkgCount).toFixed(2);

    //get the object that matches the hash-key from the working array
    //this will be the template to use when creating new packages after a split operation
    //working on a copy of the vm commodity object so we can abandon changes if needed
    let src = $.grep(this.working, function(value, i) { return value.$$hashKey == itemkey; })[0];

    //update the declared value for the first package
    src.UnitValue.Amount = cost;

    for (let y = 1; y < pkgCount; y++)
    {
      let ccobj = this.GetNewCommodityObject(src);
      ccobj.packageIndex = findex + y;

      this.working.push(ccobj);

      this.UpdateScreenMarkup(ccobj, listItem);
    }
  }

  //v = the numbers to split the shipment over, li is the parent list item
  UpdateScreenMarkup(ccobj, li)
  {
    //this is all setting up the first list item to be cloned in the next function
    //hide the split button
    $(li).find('.btn-show-split-form').fadeOut();

    //show the input for number of items in this pkg
    $(li).find('.num-of-items-in-pkg').fadeIn();

    //display the header
    $(li).find('.header-logo').css('display', 'block');

    $(li).find('.goods-declared-value').html('<strong>$' + ccobj.UnitValue.Amount + '</strong> ' + 'USD' + ')');

    this.CreateListViewItem(ccobj.packageIndex, li)

    let input = $('#listItemContainer').find('.num-of-items-in-pkg');
    $(input).tooltip({ container: 'body' });

    //when done creating elements, focus on the first visible non-disabled input box
    $('#listItemContainer').find($('.num-of-items-in-pkg:visible:not(:disabled)').first().focus());

    //if there was a single item (i.e a KIT), split evenly and close
    let qty = +$(li).find('.goods-quantity strong').text();
    if (qty === 1)
    {
      let these = $('#listItemContainer').find('li[goods-key=' + $(li).attr('goods-key') + ']');
      $(these).attr('pkg-status', 'packed');
      $(these).find('.num-of-items-in-pkg').val(1).attr('disabled', 'disabled');
      $(these).find('.goods-quantity > strong').text(1);
      $(these).find('.split-pkgs-modal').hide().find('input').val([]);
      //$('#btnApplyCommodityAssignments').trigger('click');
    }
  }

  CreateListViewItem(idx, li)
  {
    //the hash-key attr is the object id for the commodity in relation to the package
    //if we keep it, all packages will have the same hash-key (pkg 0)
    let item = $(li).clone(true).removeAttr('hash-key').removeAttr('id');
    $(item).attr('pkg-index', idx);

    $(item).find('.pkg-index-text').text(idx + 1);

    let lastInList = $('#listItemContainer').find('li[goods-key="' + $(li).attr('goods-key') + '"]').last();

    //anyway, set it implicitly here so the element shows
    $(item).appendTo('#listItemContainer').insertAfter(lastInList).find('.num-of-items-in-pkg').css('opacity', '1').show();
  }

  UpdatePkgCountsInWorkingObject(li)
  {
    let goodkey = $(li).attr('goods-key');
    let pkgindex = $(li).attr('pkg-index');
    let itemcount = $(li).find('.num-of-items-in-pkg').val();

    //find the object in this.working that has the matching pkg-index and goods-key, then update it
    let w = $.grep(this.working, function(pkg, i) { return pkg.GoodsKey == goodkey && pkg.packageIndex == pkgindex  })[0];

    w.Quantity = itemcount;
  }

  //event handler for user pressing <ENTER> on the input box for number of items to split across
  UpdateListItemPackageQuantities(li)
  {
    this.UpdatePkgCountsInWorkingObject(li);

    let numOfItems = $(li).find('.num-of-items-in-pkg');
    let remItems = $(li).find('.goods-quantity > strong');

    if (!$(numOfItems).val()) { return; }

    let x = (+$(remItems).text() - $(numOfItems).val());

    //sanity check to make sure we didn't pack more items that we had
    if (x < 0)
    {
      $(remItems).addClass('bad-math');
      return;
    }

    $(remItems).removeClass('bad-math');
    $(numOfItems).attr('disabled', 'disabled');
    $(li).attr('pkg-status', 'packed');

    //move to the next list item and update then focus
    $(li).next().find('.goods-quantity > strong').text(x)

    //on the second-to-last list item, use the input value to auto-set the last item
    //to get the number of rows that have already been assigned, count the list items with the goods-key
    //that have disabled input boxes
    let rowsForThisGood = $('#listItemContainer').find('li[goods-key="' + $(li).attr('goods-key') + '"]');
    let rowsDone = $(rowsForThisGood).find('input:disabled:visible').length;

    if (rowsForThisGood.length == (rowsDone + 1))
    {
      let lastLi = $('#listItemContainer').find('li[goods-key="' + $(li).attr('goods-key') + '"]').last();

      $(lastLi).attr('pkg-status', 'packed');

      let lastVal = $(li).next().find('.goods-quantity > strong').text();
      $(lastLi).find('.num-of-items-in-pkg').val(lastVal).attr('disabled', 'disabled');

      $('#listItemContainer').find('.assign-to-pkg:visible').first().focus();

      this.UpdatePkgCountsInWorkingObject(lastLi);
    }
  }

  //event handler for the 'Apply' button
  ApplyCommodityAssignments()
  {
    let that = this;

    //every list item will have a pkg-index attr, get that highest value to determine how many pkgs we need
    //don't use pkg-status.length because that will include multiple commodities and the end result will be wrong
    let packed = $('#listItemContainer').find('li[pkg-status=packed]');

    //first get all the pkg-index attributes, then get the highest number and add one
    let pkgs = $(packed).map(function(x, value) { return +$(this).attr('pkg-index') }).toArray();
    let pkgCount = Math.max(...pkgs) + 1;

    while (this.packages.length < pkgCount)
    {
      //the vm.addPackage isn't firing, just click the button
      $('button[ng-click="vm.addPackage();"]').trigger('click').promise().done(function(e) { });
    }

    //loop the pkg count, get the commodities for that pkg, write the objects to the vm CommodityContents object
    for (let p = 0; p < pkgCount; p++)
    {
      let w = $.grep(this.working, function(item, i) { return item.packageIndex == p });
      this.packages[p].CommodityContents = w;
    }

    //click the General tab
    $('li[heading=General]:visible a').first().click();
    this.HideAssignGoodsPane();

    //after we are done, move back to the first package in the shipment
    while ($('#spanCurrentPackageIndex').text() !== '1')
    {
      $('button[ng-click="vm.previousPackage();"]').first().click();
    }

  }

  GetClassViewModel()
  {
    //the goods data is going to come from vm.currentShipment.Packages[pkgIndex].CommodityContents as an array
    return this.viewModel
  }

  //passing in the commodity object for the first package and copying the non-null values to a new object, then returning
  GetNewCommodityObject(obj)
  {
    let c = {};

    Object.keys(obj).forEach(key =>
    {
      if (key != '$$hashKey' && key != 'packageIndex')
      {
        let value = obj[key];
        if (value) c[key] = value;
      }
    });

    return c;
  }

  //when hiding this modal because the user cancelled out, show the 'Scan Barcode' input and 'Split' button
  CancelSplitOfPackages(li)
  {
    //hide the split modal and clear the input value
    $('.escape-target').hide();
    $(li).find('.split-pkgs-modal').hide().find('input[type=number]').val([]).focus();
    $(li).find('.btn-show-split-form, .assign-to-pkg').fadeIn();
  }

  //delegate the event handlers for the div
  InitiateDelegates()
  {
    let that = this

    $('body').delegate('#btnApplyCommodityAssignments', 'click', function(e)
    {
      that.ApplyCommodityAssignments();
    });

    $('body').delegate('.btn-show-split-form', 'click', function(e)
    {
      let div = $(this).closest('div').find('.split-pkgs-modal').first();
      $(div).tooltip({ container: 'body' });

      $(div).css('display', 'flex').fadeIn();
      $('.split-pkgs-modal:visible').find('input').first().focus();

      //hide the split button now...and the input
      $(this).hide();
      $(this).closest('div').find('.assign-to-pkg').hide();
    });

    $('body').delegate('input.btn-go-split-pkgs', 'click', function(e)
    {
      console.log('btn-go-split-pkgs click event handler');

      let li = $(this).closest('li').first();
      let div = $(this).closest('div').first();

      $(li).find('.pkg-index-text').first().text('1');

      //val - 1 as the first entry will be the first package
      let v = $(div).find('input[type=number]').first().val();

      that.UpdateGoodsObject(v, li);

      $('.split-pkgs-modal').hide();
    });

    $('body').delegate('#btnCancelAssignGoods', 'click', function(e)
    {
      that.HideAssignGoodsPane();
    });

    //if the user escapes out of the split modal, hide the modal and show the previous forms
    $('body').delegate('.split-pkgs-modal', 'keydown', function(e)
    {
      var keycode = (e.keyCode ? e.keyCode : e.which);
      if (keycode === 27)
      {
        //need to show the previous input and button now
        let li = $(this).closest('li');

        that.CancelSplitOfPackages(li);
      }
    });

    //if the user presses enter or tab in the num-of-items number input, move to the next one
    //these are the number inputs after the items have been split across packages
    $('body').delegate('.num-of-items-in-pkg', 'keydown focusout', function(e)
    {
      console.log('.num-of-items-in-pkg keydown focusout event handler');

      $('[data-toggle="tooltip"]').tooltip('hide');

      var keycode = (e.keyCode ? e.keyCode : e.which);
      if (keycode === 13 || keycode === 9)
      {
        //disable the input, update the commodity object, focus on the next box input
        //$(this).closest('div').find('.assign-to-pkg').focus();
        that.UpdateListItemPackageQuantities($(this).closest('li'));
      }
    });

    //user hits enter in the input box, click the button for them
    $('body').delegate('.num-of-pkgs-to-use', 'keyup keydown', function(e)
    {
      console.log('.num-of-pkgs-to-use keyup keydown event handler');
      var keycode = (e.keyCode ? e.keyCode : e.which);
      let btn = $(this).closest('div:visible').find('input.btn-go-split-pkgs');

      $(btn).attr('disabled', $(this).val() < 2);

      if (keycode == 13) //enter
      {
        //let btn = $(this).closest('div:visible').find('input.btn-go-split-pkgs');
        $(btn).trigger('click');
      }
    });

    //this was formerly .scan-input.  as of 9/24/2022, removing the split functionality completely
    //with this change, only need to get the numerical value out of the input
    $('body').delegate('.assign-to-pkg', 'keydown focusout', function(e)
    {
      console.log('.assign-to-pkg keydown focusout event handler');
      var keycode = (e.keyCode ? e.keyCode : e.which);
      debugger;
      if (keycode == 13 || keycode == 9 || e.type == "focusout") //enter
      {
        e.preventDefault();
        //pull only the number from whatever was passed.  make sure it is greater than 0 since we can't assign a good a pkg zero.
        const val = $(this).val().replace(/[^1-9]/g, '');

        let li = $(this).closest('li');
        
        if (val)
        {
          //console.log('assigning this to ', val);
          //assign the integer to the line item header (not shown, but may be shown in the future)
          //this one is easy, just assign the pkg-index attr to whatever they scanned.  when the apply button is clicked it will assign the commodities
          //let li = $(this).closest('li');
          $(li).find('.pkg-index-text').first().text(val);
          $(li).find('button').attr('disabled', 'disabled');
          $(li).attr('pkg-index', val - 1);
          $(li).attr('pkg-status', 'packed');

          //find the matching item in the this.working array and update the package index
          let item = $.grep(that.working, function(good, i) { return good.$$hashKey == $(li).attr('hash-key') })[0];
          item.packageIndex = val - 1;
          //return;
        }
        // set focus to next li input box
        if (e.type !== "focusout") {
          $(li).next().find('.assign-to-pkg').focus().change();
        }
      }
    });

    $('#btnAssignCommodities').on('click', function(e)
    {
      if ($('#divAssignCommodities').is(':visible')) return;
      that.ShowAssignGoodsPane();
    });
  }
}

// this.convertFileInputToBase64(inputId, leg) // returns Promise resolving to upload payload object
    this.convertFileInputToBase64 = function (inputId, leg) {
        return new Promise(function (resolve, reject) {

            var data = {};
            const fileInput = $('#' + inputId);

            // Do not continue if there are no files
            if (!fileInput ||
                fileInput.length === 0 ||
                !fileInput[0].files ||
                fileInput[0].files.length === 0) {
                return;
            }

            // Get basic file info
            var file = fileInput[0].files[0];
            var fileName = file.name;

            // Create reader for reading files
            var reader = new FileReader();
            reader.readAsDataURL(file);
            reader.addEventListener("load", function () {

                // Create and return payload for upload call
                data = {
                    Action: "PaperlessDocUpload",
                    FileName: fileName,
                    encodedString: reader.result.split(',')[1],
                    shipperName: that.currentShipment.PackageDefaults.Shipper,
                    leg: leg
                }

                resolve(data);

            }, false);

            reader.addEventListener("error", function (error) {
                reject(error);
            }, false);

        });

    }

// function Tools.GetUniqueArrayFromString(inString, stringDelimiter) // returns array of unique, non-empty trimmed strings
            GetUniqueArrayFromString: function (inString, stringDelimiter = ',') {
               var trimArray = $.grep(inString.split(stringDelimiter), function (a) {
                  return a.trim().length > 0;
               });
               /* Get rid of any duplicate array entries. */
               trimArray = trimArray.filter((e, i, a) => a.indexOf(e) === i);
               return trimArray;
            }

// downloadFile(exportFileName, encodedString) // returns undefined
function downloadFile(exportFileName, encodedString) {
        var decodedResult = atob(encodedString); // decode a base-64 encoded string
        var csvFile = new Blob([decodedResult], { type: "text/csv" });
    
        var ie = navigator.userAgent.match(/MSIE\s([\d.]+)/),
            ie11 = navigator.userAgent.match(/Trident\/7.0/) && navigator.userAgent.match(/rv:11/),
            ieEdge = navigator.userAgent.match(/Edge/g),
            ieVer = (ie ? ie[1] : (ie11 ? 11 : (ieEdge ? 12 : -1)));
    
        exportFileName = exportFileName + '.csv';
    
        if (ieVer > -1) {
            window.navigator.msSaveBlob(csvFile, exportFileName);
    
        } else {
            var downloadLink = document.createElement("a");
            downloadLink.download = exportFileName;
            downloadLink.href = window.URL.createObjectURL(csvFile);
            downloadLink.style.display = "none";
            document.body.appendChild(downloadLink);
            downloadLink.click();
            document.body.removeChild(downloadLink);
        }
    }
