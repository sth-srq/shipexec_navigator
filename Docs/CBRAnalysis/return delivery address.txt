// CheckForMatchingConsignee(CustomerOrderObject) // returns boolean indicating whether the customer order delivery address matches the current shipment consignee address
      CheckForMatchingConsignee: function (CustomerOrderObject)
      {
        try
        {
          var con = $('name-address[nameaddress="vm.currentShipment.PackageDefaults.Consignee"]');

          let a = {
            //conCompany: $(con).find('input[name="company"]').val().trim().toUpperCase().replaceAll(' ',''),
            conAddress1: $(con).find('input[name="address1"]').val().trim().toUpperCase(),
            conCity: $(con).find('input[name="city"]').val().trim().toUpperCase(),
            conState: $(con).find('input[name="stateProvince"]').val().trim().toUpperCase(),
            conPostalCode: $(con).find('input[name="postalCode"]').val().trim().substring(0, 5)
          };

          let b = {
            //conCompany: CustomerOrderObject.shipAddress1.CONSIGNEE_COMPANY.trim().toUpperCase().replaceAll(' ', ''),
            conAddress1: CustomerOrderObject.shipAddress1.trim().toUpperCase(),
            conCity: CustomerOrderObject.shipCity.trim().toUpperCase(),
            conState: CustomerOrderObject.shipState.trim().toUpperCase(),
            conPostalCode: CustomerOrderObject.shipZipCode.trim().substring(0, 5)
          };

          /* Remove all the spaces in the strings so we get a good comparison, not one that's thrown off by an extra space. */
          let aa = JSON.stringify(a).replaceAll(' ', '');
          let bb = JSON.stringify(b).replaceAll(' ', '');

          /* Quick comparison to see if the strings match. */
          if (aa != bb)
          {
            Logger.Log({ Source: 'CheckForMatchingConsignee()', Message: 'Could not add order ' + CustomerOrderObject.orderNumber + ' (' + CustomerOrderObject.orderId + '). Primary order data to follow.', Data: a });
            Logger.Log({ Source: 'CheckForMatchingConsignee()', Message: 'Could not add order ' + CustomerOrderObject.orderNumber + ' (' + CustomerOrderObject.orderId + '). Subsequent order data to follow.', Data: b });
          }

          return (aa == bb);
        }
        catch (error)
        {
          Logger.Log({ Source: 'CheckForMatchingConsignee()', Error: error });
        }
      }

// UpdateReferencesWithOrderNumbers(OrderNumber) // returns void; updates shipper reference and misc reference fields with the consolidated order number string
      UpdateReferencesWithOrderNumbers: function (OrderNumber)
      {
        var inputShipperReference = $('input[type=text][ng-model="vm.currentShipment.Packages[vm.packageIndex].ShipperReference"]');
        var inputMiscReference07 = $('input[type=text][ng-model="vm.currentShipment.Packages[vm.packageIndex].MiscReference7"]');
        var shipperReferenceVal = $(inputShipperReference).val() + ',' + OrderNumber;

        var newUniqueString = Tools.GetUniqueCSVStringFromString(shipperReferenceVal);

        /* Set these input values.  ShipperReference is used by the customer, MiscReference7 is used by the SBR to update each record post-ship. */
        /* ShipperReference ONLY keeps the first order number when it goes to PostShip, before the comma, MiscReference7 keeps the entire string. */
        /* On the server-side, if MiscReference7 contains a comma, each item in the csv list gets posted back.  If MiscReference7 is empty, the ShipperReference is used instead for a single post-back. */
        $(inputShipperReference).val(newUniqueString);
        $(inputMiscReference07).val(newUniqueString);
      }

function SetDeliveryNotificationAddressEmail(pkg, ngModelName, emailAddress, checked) {
        if (checked) {
            if (pkg.DeliveryNotificationAddressEmail == undefined || pkg.DeliveryNotificationAddressEmail.trim() == '') {
                if (ngModelName == 'ShipFromDeliveryNotificationEmail' && emailAddress.Origin) {
                    pkg.DeliveryNotificationAddressEmail = emailAddress.Origin;
                } else if (ngModelName == 'ShipToDeliveryNotificationEmail' && emailAddress.Consignee) {
                    pkg.DeliveryNotificationAddressEmail = emailAddress.Consignee;
                }
            } else {
                if (ngModelName == 'ShipFromDeliveryNotificationEmail' && emailAddress.Origin) {
                    pkg.DeliveryNotificationAddressEmail += ',' + emailAddress.Origin;
                } else if (ngModelName == 'ShipToDeliveryNotificationEmail' && emailAddress.Consignee) {
                    pkg.DeliveryNotificationAddressEmail += ',' + emailAddress.Consignee;
                }
            }
        } else {
            var notificationAddressEmail = pkg.DeliveryNotificationAddressEmail;
            if (ngModelName == 'ShipToDeliveryNotificationEmail' && emailAddress.Consignee) {
                var regexp = new RegExp(',?' + emailAddress.Consignee, 'g')
                notificationAddressEmail = notificationAddressEmail.replace(regexp, '');
            } else if (ngModelName == 'ShipFromDeliveryNotificationEmail' && emailAddress.Origin) {
                var regexp = new RegExp(',?' + emailAddress.Origin, 'g')
                notificationAddressEmail = notificationAddressEmail.replace(regexp, '');
            }

            if (notificationAddressEmail.indexOf(',') == 0) {
                notificationAddressEmail = notificationAddressEmail.substring(1);
            }
            pkg.DeliveryNotificationAddressEmail = notificationAddressEmail;
        }
    }

    this.convertFileInputToBase64 = function (inputId, leg) {
        return new Promise(function (resolve, reject) {

            var data = {};
            const fileInput = $('#' + inputId);

            // Do not continue if there are no files
            if (!fileInput ||
                fileInput.length === 0 ||
                !fileInput[0].files ||
                fileInput[0].files.length === 0) {
                return;
            }

            // Get basic file info
            var file = fileInput[0].files[0];
            var fileName = file.name;

            // Create reader for reading files
            var reader = new FileReader();
            reader.readAsDataURL(file);
            reader.addEventListener("load", function () {

                // Create and return payload for upload call
                data = {
                    Action: "PaperlessDocUpload",
                    FileName: fileName,
                    encodedString: reader.result.split(',')[1],
                    shipperName: that.currentShipment.PackageDefaults.Shipper,
                    leg: leg
                }

                resolve(data);

            }, false);

            reader.addEventListener("error", function (error) {
                reject(error);
            }, false);

        });

    }
    
    
    var checkElmLoaded = function (selector) {
        return new Promise(resolve => {
            if (document.querySelector(selector)) {
                return resolve(document.querySelector(selector));
            }
            const observer = new MutationObserver(mutations => {
                var temp = document.querySelector(selector);
                if (document.querySelector(selector)) {
                    resolve(document.querySelector(selector));
                    observer.disconnect();
                }
            });
            observer.observe(document.body, {
                childList: true,
                subtree: true
            });
        });
    }

        var CustomData_AfterCustom4 = [];
        var item;

        customData = this.vm.profile.UserInformation.Address.CustomData;
        // Iterate over the CustomData array and map values to their respective keys
        $.each(customData, function (index, item) {
            customValues[item.Key] = item.Value;
        });

        if (this.vm.profile.Name == 'Outbound Labels Leg 2 Allowed User Profile') {
            for (var i in customData.length) {
                item = customData[i];
                customKeysArray.push(item.Key);
            }

            var startIndex = -1;
            // Find the index of the key "Custom5"
            for (var i = 0; i < customData.length; i++) {
                if (customData[i].Key === "Custom5") {
                    startIndex = i;
                    break;
                }
            }

            // If "Custom5" is found, start storing values from it onwards
            if (startIndex !== -1) {
                for (var j = startIndex; j < customData.length; j++) {
                    CustomData_AfterCustom4.push(customData[j].Value);
                }
            }

            // Now, customValues object contains all custom values accessible by their keys
            var custom1 = customValues['Custom1'];
            var custom2 = customValues['Custom2'];
            var custom3 = customValues['Custom3'];
            var custom4 = customValues['Custom4'];
            var custom5 = customValues['Custom5'];
            var custom6 = customValues['Custom6'];
        } else {
            //debugger;
            custom1 = customValues['Custom1'];
            custom2 = customValues['Custom2'];
        }

        //Checking DryIceRequired sets true via User Custom1 data.
        if (custom1.split(',').length > 1) {
            custom1_values = Tools.GetUniqueArrayFromString(custom1, ',');
        } else {
            throw new Error("Custom1 is not Comma delimited string.");
        }

        //Setting the Leg1Shipper
        shipmentRequest.PackageDefaults.Shipper = custom1_values[0];

        //Creating Dry Ice Required Checkbox
        if ($('#DryIceRequired').length == 0 && custom1_values[1] == "true") {
            //Adding DryIceRequired checkbox and its label.

                type: 'checkbox',

                name: 'DryIceRequired'
            }));
            $('div.col-md-6:last').append($('<label />', {
                for: 'DryIceRequired',
                text: 'Dry Ice Required'
            }));
        }

        //Comparing the User Custom2 with the Profile enabled Services and remove the unmatched services from the Service UI.
        var arrayValues_custom2;
        if (custom2 != null || custom2 != undefined) {
            arrayValues_custom2 = Tools.GetUniqueArrayFromString(custom2, ',');
            var matchedValues = [];
            var profileServices = this.vm.profile.Services;
            profileServices.forEach(function (value1) {
                var serviceExists = arrayValues_custom2.some(function (value) {
                    return value1.Name.indexOf(value) !== -1;
                });
                if (serviceExists) {
                    matchedValues.push(value1);
                }
            });

            //Populating the Matched services on the Service UI.

    function checkOriginConsigneeCountry() {
        var shipFromCountry = getObjectValue(vm.currentShipment.PackageDefaults.OriginAddress.Country);
        var shipToCountry = getObjectValue(vm.currentShipment.PackageDefaults.Consignee.Country);

        /* If we have no shipToCountry OR the shipFromCountry is either US or CA, disable the elements, return out. */
        if (!shipToCountry | shipFromCountry == "US" | shipFromCountry == "CA") {
            DisableThirdPartyBillingElements();
            return false;
        }

        if (shipFromCountry != shipToCountry) {
            /* If the shipFromCountry does NOT equal the shipToCountry then ship Third Party. */
            SetThirdPartyBilling(vm.currentShipment);
        }
        else {
            /* Now check to see if both the origin and destination countries are members of the EU. */
            if ((IsMemberOfEu(shipFromCountry)) && (IsMemberOfEu(shipToCountry))) {
                SetThirdPartyBilling(vm.currentShipment);
            }

            //ShipThirdPartyBilling((IsMemberOfEu(shipFromCountry)) && (IsMemberOfEu(shipToCountry)));
        }

        if (shipFromCountry != "US") {
            $('select[ng-model="vm.currentShipment.Packages[vm.packageIndex].Weight.Units"]').val('string:KG').change();
            $('select[ng-model="vm.currentShipment.Packages[vm.packageIndex].Dimensions.Units"]').val('number:1').change();
        }
    }
