// Log(LogObject) // returns undefined
Log: function (LogObject)
      {
        try
        {
          /*
          Usage:
            Errors:
              try {
                ...
              } catch (error) {
                Logger.Log({ Source: 'FunctionName()', Error: error });
                OR
                Logger.Log({ Source: 'FunctionName()', Error: { name: 'Error name', message: 'Error message' } });
                OR 
                Logger.Log({ Source: 'FunctionName()', Error: { name: 'Error name', message: 'Error message' }, Data: 'Simple or complex data object.' });
              }

            Info (default, no 'Type' needed):
              Logger.Log({ Source: 'FunctionName()', Message: 'Test of server info logging', Data: 'Simple or complex data object.' });

            All others:
              Logger.Log({ Source: 'FunctionName()', Type: 'Trace', Message: 'Test of server info logging with a complex data object.', Data: SessionData.GetCurrentUserContext() });
              Logger.Log({ Source: 'FunctionName()', Type: 'Fatal', Message: 'Test of server info logging with a simple data string.', Data: 'Simple data string literal.' });
              Logger.Log({ Source: 'FunctionName()', Type: 'Debug', Message: 'Test of server debug logging with no data object.' });
          */

          /* If the LogObject has an 'error' object, set the LogLevel here.  This makes the LogLevel variable optional for errors. */
          if (LogObject.Error)
          {
            LogObject.LogLevel = LogLevel.Error;

            /* Can't pass the error container to the server without preparing it inside a new object first. */
            var e = {};
            e.name = LogObject.Error.name;
            e.message = LogObject.Error.message;
            LogObject.Error = e;
          }
          else if (!LogObject.LogLevel)
          {
            /* If the LogObject does NOT have an 'error' object AND there is no LogLevel set, use .Info.  This makes the LogLevel variable optional for info logging. */
            LogObject.LogLevel = LogLevel.Info;
          }

          /* All logging will go to the console, output here. */
          switch (LogObject.LogLevel)
          {
            case LogLevel.Error:

              console.error('Exception encountered in', LogObject.Source);
              console.error(LogObject.Error.name);
              console.error(LogObject.Error.message);

              /* Use .trace if needed but within the Angular sandbox the .trace object can be huge and convoluted. */
              /* Be careful if passing .trace data on to the server as it will contain various data types and directives. */
              //console.error(LogObject.Error.trace);

              /* Don't use console.error for the data object, it causes readability issues. */
              if (LogObject.Data) { console.log(LogObject.Data); }
              break;

            /* Using the same nomenclature used by SBR logging. */
            case LogLevel.Info:
            case LogLevel.Debug:
            case LogLevel.Trace:
            case LogLevel.Fatal:
            default:

              console.log('Output from', LogObject.Source);
              console.log(LogObject.Message);
              if (LogObject.Data) { console.log(LogObject.Data); }
              break;
          }

          if (Tools.GetServerDebugMode() == true)
          {
            var dataObject = {
              ServerMethod: 'AddClientEntry',
              MessageObject: LogObject
            };

            var ajaxObject = {
              UserContext: Tools.GetCurrentUserContext(),
              Data: JSON.stringify(dataObject)
            };

            /* async doesn't necessarily need to be set to false here but I think it helps with data collisions and getting the error messages to appear in chronological order on the server. */
            var ajaxRequest = $.ajax({
              url: 'api/ShippingService/UserMethod',
              method: 'POST',
              contentType: 'application/x-www-form-urlencoded; charset=UTF-8',
              dataType: 'json',
              data: ajaxObject,
              async: true
            });

            ajaxRequest.done(function (json)
            {
              /* There is no return for this call, if the POST does come back with data, it's a server message/error like 'UserContext cannot be null' or more likely '{"Data":"","ErrorCode":0}' */
              //console.log(JSON.parse(json.Data));
            });

            ajaxRequest.fail(function (jqXHR, textStatus)
            {
              console.log('Unable to log message to server.');
              console.log(jqXHR);
              console.log(textStatus);
            });

            ajaxRequest.always(function (jqXHR, textStatus, jqXHRError) 
            {

            });
          }

          /* Need to hide the loader outside the decision trees. */
          Tools.HideLoader();

        }
        catch (error)
        {
          console.log(error.message);
        }
      }

// function MoveGoodToBox(li, destBox) // returns boolean|undefined
MoveGoodToBox(li, destBox)
    {
      let vm = this.ClassViewModel;
      let that = this;
      let data = $(li).data();
      let destCC = this.packages[destBox - 1].CommodityContents || [];

      //find the box the item currently resides in
      //split this statement up over a few calls to make it easier to read
      let allGoods = this.packages.map(m => m.CommodityContents).flat();
      let item = allGoods.find(cc => cc.UniqueId == data.UniqueId);

      if (+item.PkgIndex == +destBox) { console.log('Item already exists in box.'); return false; }

      //remove the item from its current box
      this.packages[item.PkgIndex - 1].CommodityContents = this.packages[item.PkgIndex - 1].CommodityContents.filter(p => p.UniqueId != data.UniqueId);

      //add it to the new box
      item.PkgIndex = destBox;
      destCC.push(item);
      $(li).find('.pkg-index-text').fadeOut().promise().done(function(e) { $(this).text(destBox).fadeIn(); });

      //console.log(vm.currentShipment.Packages);
      $('li[heading="Goods"]:visible:first > a').trigger('click');
      $('#goods .btn-group').find('.btn:gt(0)').not('.active').first().trigger('click');

      //update the weights and values
      //this data will have changed by now so get a new reference to the object
      allGoods = this.packages.map(m => m.CommodityContents).flat();
      //the items array will have the CommodityContents object for every package in the shipment

      //sort by package index
      //update the weight and declared values for all the packages in the shipment
      allGoods = allGoods.sort((a, b) => { return a.PkgIndex - b.PkgIndex });
      $(allGoods).each(function(a, b)
      {
        let pkg = that.packages[b.PkgIndex - 1];

        if ((a == 0) || (b.PkgIndex != allGoods[a - 1].PkgIndex))
        {
          pkg.Weight.Amount = 0;
          pkg.DeclaredValueAmount.Amount = 0;
        }

        pkg.Weight.Amount += +b.PVTotalWeight;
        pkg.DeclaredValueAmount.Amount += +b.PVTotalValue;
      });
    }

// PreShip(shipmentRequest, userParams) // returns undefined
this.PreShip = function(shipmentRequest, userParams) {    
  
    // We will require the actual shipment to send weight and dims in kg/cm
    // However, because US Vendors are sending the product the client gui needs to be in Lb/In.
    
    debugger;
    //kg = lbs*0.453592 
    totalShipWeight = 0;
    totalVolumetricResult = 0;
    
    vm.currentShipment.Packages.forEach(pkg => {
        totalShipWeight += Number(pkg.Weight.Amount);
        totalVolumetricResult += ((Number(pkg.Dimensions.Length).toFixed(2)*Number(pkg.Dimensions.Width).toFixed(2)*Number(pkg.Dimensions.Height)).toFixed(2))/166;
    });
              
    totalShipWeight = Number(totalShipWeight).toFixed(2);
    console.log(totalShipWeight);
    console.log(totalVolumetricResult);
    if (totalShipWeight > 150)
    {
        alert("Total Shipment Weight Exceeds Shipment Weight Limit of 150lbs")
        throw new Error("Shipping halted due to Over Maximum Weight Rule " + totalShipWeight);
    }
    
    if (totalVolumetricResult > Number(maxvolumetric))
    {
        alert("Total Shipment Weight Exceeds Total Shipment Volumetric Weight Limit of " + Number(maxvolumetric))
        throw new Error("Shipping halted due to Total Shipment Volumetric Weight Rule " + Number(totalVolumetricResult).toFixed(2));
    }
    
     vm.currentShipment.Packages.forEach(pkg => {
         if (pkg.Weight.Units === "LB"){
             pkg.Weight.Amount = Number(pkg.Weight.Amount * 0.453592).toFixed(2);
             pkg.Weight.Units = "KG";
         }
         
     });
     
    //cm=in/0.39370
     vm.currentShipment.Packages.forEach(pkg => {
         if (pkg.Dimensions.Units === 0){
             pkg.Dimensions.Length = Number(pkg.Dimensions.Length/0.39370).toFixed(2)
             pkg.Dimensions.Width = Number(pkg.Dimensions.Width/0.39370).toFixed(2)
             pkg.Dimensions.Height = Number(pkg.Dimensions.Height/0.39370).toFixed(2)
             pkg.Dimensions.Units = 1;
         }
         
     });
     
     
    // debugger;
      if(Number(vm.currentShipment.Packages[0].DeclaredValueCustoms.Amount).toFixed(2) > 2500 ){
          if(vm.currentShipment.Packages[0].ExportInformationCode === null){
              vm.currentShipment.Packages[vm.packageIndex].ExportInformationCode = "OS";
          }
          if(vm.currentShipment.Packages[0].AesTransactionNumber === null){
              vm.currentShipment.Packages[0].SedMethod = 2;
              $('input[type=text][ng-model="vm.currentShipment.Packages[vm.packageIndex].AesTransactionNumber"]').focus();
              throw new Error('Missing AES/ITN Number');
          } 
          if(shipmentRequest.Packages[0].AesTransactionNumber !== null){
              shipmentRequest.Packages[0].Comments = shipmentRequest.Packages[0].Comments + "\n ITN/AES Filer " + shipmentRequest.Packages[0].AesTransactionNumber;
             
          }
      }
     
    console.log(shipmentRequest);
  
};

// PreShip(shipmentRequest, userParams) // returns void
this.PreShip = function(shipmentRequest, userParams) {    
    // We will require the actual shipment to send weight and dims in kg/cm
    // However, because US Vendors are sending the product the client gui needs to be in Lb/In.
    
    // debugger;
    
    totalShipWeight = 0;
    totalVolumetricResult = 0;
    
    vm.currentShipment.Packages.forEach(pkg => {
        totalShipWeight += Number(pkg.Weight.Amount);
        totalVolumetricResult += ((Number(pkg.Dimensions.Length).toFixed(2)*Number(pkg.Dimensions.Width).toFixed(2)*Number(pkg.Dimensions.Height)).toFixed(2))/166;
    });
              
    totalShipWeight = Number(totalShipWeight).toFixed(2);
    console.log(totalShipWeight);
    console.log(totalVolumetricResult);
    if (totalShipWeight > 150)
    {
        alert("Total Shipment Weight Exceeds Shipment Weight Limit of 150lbs")
        throw new Error("Shipping halted due to Over Maximum Weight Rule " + totalShipWeight);
    }
    
    if (totalVolumetricResult > Number(maxvolumetric))
    {
        alert("Total Shipment Weight Exceeds Total Shipment Volumetric Weight Limit of " + Number(maxvolumetric))
        throw new Error("Shipping halted due to Total Shipment Volumetric Weight Rule " + Number(totalVolumetricResult).toFixed(2));
    }
    
    //kg = lbs*0.453592    
    vm.currentShipment.Packages.forEach(pkg => {
         if (pkg.Weight.Units === "LB"){
             pkg.Weight.Amount = Number(pkg.Weight.Amount * 0.453592).toFixed(2);
             pkg.Weight.Units = "KG";
         }
         
    });
    
  
    //cm=in/0.39370
    vm.currentShipment.Packages.forEach(pkg => {
         if (pkg.Dimensions.Units === 0){
             pkg.Dimensions.Length = Number(pkg.Dimensions.Length/0.39370).toFixed(2)
             pkg.Dimensions.Width = Number(pkg.Dimensions.Width/0.39370).toFixed(2)
             pkg.Dimensions.Height = Number(pkg.Dimensions.Height/0.39370).toFixed(2)
             pkg.Dimensions.Units = 1;
       }
         
    });
     
   if(Number(shipmentRequest.Packages[0].DeclaredValueCustoms.Amount).toFixed(2) > 2500 ){
          if(shipmentRequest.Packages[0].ExportInformationCode === null){
              shipmentRequest.Packages[0].ExportInformationCode = "OS";
          }
          if(shipmentRequest.Packages[0].AesTransactionNumber === null){
              shipmentRequest.Packages[0].SedMethod = 2;
              $('input[type=text][ng-model="shipmentRequest.Packages[this.vm.packageIndex].AesTransactionNumber"]').focus();
              throw new Error('Missing AES/ITN Number');
          } 
          
          if(shipmentRequest.Packages[0].AesTransactionNumber !== null){
              shipmentRequest.Packages[0].Comments = shipmentRequest.Packages[0].Comments + "\n ITN/AES Filer " + shipmentRequest.Packages[0].AesTransactionNumber;
             
          }
      }
};

// ShippingLoaded() // returns void
this.ShippingLoaded = function(){
    this.vm.populateReturnAddress = this.PopulateReturnAddress;
    this.vm.populateShipToAddress = this.PopulateShipToAddress;
    this.vm.serviceOption = serviceOption;
    this.vm.serviceOptions = serviceOptions(this.vm.profile);
    this.vm.currentShipment.Packages[0].ShipperReference = cost_Center;
    this.vm.currentShipment.Packages[0].MiscReference1 = cost_Center;
    //this.vm.currentShipment.Packages[0].ConsigneeReference = po_number;
    //this.vm.currentShipment.Packages[0].MiscReference2 = po_Number;
    this.vm.currentShipment.Packages[0].MiscReference3 = user_partNumber;
    this.vm.currentShipment.Packages[0].Description = user_description;
  
    function serviceOption() {
        this.selectedServices = [this.currentShipment.PackageDefaults.Service];
    }
    function serviceOptions(profile) {
        //debugger;
        var filterServices = [];
        var customData3 = uPS_Service.split(',');
        for (var item of customData3) {
            var symbol = getServiceSymbol(item.toLowerCase());
            for (var service of profile.Services) {
                if (service.Symbol === symbol) {
                    //service.Name = service.Name.replace('(UPS Adapter)','').trim();
                    filterServices.push(service);
                    break;
                }
            }
        }
        return filterServices;
        function getServiceSymbol(code) {
             if (code == '2dn') return 'UPSAPI.UPS.2DA'
            else if (code == '3dd') return 'UPSAPI.UPS.3DS'
            else if (code == 'exp') return 'UPSAPI.UPS.EXPEDITED'
            else if (code == 'ess') return 'UPSAPI.UPS.EXPRESS'
            else if (code == 'nda') return 'UPSAPI.UPS.NDA'
            else if (code == 'dae') return 'UPSAPI.UPS.NDAEAM'
            else if (code == 'grn') return 'UPSAPI.UPS.GND'
            else if (code == 'uwe') return 'UPSAPI.UPS.EXPEDITED'
            else if (code == 'wep') return 'UPSAPI.UPS.EXPRESSSP'
            else if (code == 'swe') return 'UPSAPI.UPS.EXPRESS'
            else if (code == 'uws') return 'UPSAPI.UPS.SAVER'
            else if (code == 'std') return 'UPSAPI.UPS.STANDARD'
        }
        
    }
    
    if(dims.includes(',')){
        var dimensions = dims.split(',');
        
        this.vm.currentShipment.Packages[0].Weight.Amount = Number(dimensions[0]);
        maxvolumetric = Number(dimensions[4]);
        console.log("Max Volumetric: " + maxvolumetric);
        this.vm.currentShipment.Packages[0].Dimensions = {
            Length: Number(dimensions[1]),
            Height: Number(dimensions[2]),
            Width: Number(dimensions[3]),
            Units: 0
        };
    }
    vm = this.vm;
};

// getMegapixelsFromFileReader(fileData) // returns Promise resolving to megapixels
function getMegapixelsFromFileReader(fileData) {
                return new Promise(function(resolve, reject) {
                    var img = new Image();
                    img.src = fileData;
                    img.onload = function () {
                        var width = this.naturalWidth;
                        var height = this.naturalHeight;
                        console.log(`Image dimensions: ${width}x${height}`);
                        var megapixels = (width * height) / 1000000;
                        resolve(megapixels);
                    }
                })
            }

		}

		if (errorMessage !== undefined && errorMessage !== '') {

			alert(errorMessage);

this.PreShip = function (shipmentRequest, userParams) {
		console.log("🚀 [PreShip] Starting PreShip function...");
	    shipmentRequest = vmTemp.currentShipment;
		// Validate shipmentRequest and Packages array
		if (!shipmentRequest || !shipmentRequest.Packages || !Array.isArray(shipmentRequest.Packages)) {
			console.error("❌ [PreShip Error] Shipment request is invalid or missing required properties.");
			//alert("[PreShip Error] Shipment request is invalid or missing required properties.");
			return;
		}
	
		let validDimensions = [];
	
		console.log(`📦 [PreShip] Processing ${shipmentRequest.Packages.length} package(s)...`);
	
		// Retrieve current package dimensions from vmTemp
		try {
			let currentPackage = vmTemp.currentShipment.Packages[vmTemp.packageIndex];
			if (!currentPackage || !currentPackage.Dimensions) {
				console.warn("⚠️ [PreShip Warning] vmTemp.currentShipment package or dimensions not found.");
			} else {
				let dims = currentPackage.Dimensions;
				console.log(`📏 [PreShip] Current Package Dimensions from vmTemp: ${dims.Length}x${dims.Width}x${dims.Height} 0`);
			}
		} catch (error) {
			console.error("❌ [PreShip Error] Unable to retrieve package dimensions from vmTemp.");
		}
	
		// Loop through all packages to validate their dimensions
		for (let i = 0; i < shipmentRequest.Packages.length; i++) {
			console.log(`🔍 [PreShip] Checking package at index ${i}...`);
	
			let packageData = shipmentRequest.Packages[i];
	
			if (!packageData) {
				console.error(`❌ [PreShip Error] Package at index ${i} is undefined.`);
				//alert(`[PreShip Error] Package at index ${i} is undefined.`);
				return;
			}
	
			if (!packageData.Dimensions) {
				console.warn(`⚠️ [PreShip Warning] Package at index ${i} does not contain 'Dimensions'.`);
				//alert(`[PreShip Error] Package at index ${i} does not contain 'Dimensions'.`);
				return;
			}
	
			let dims = packageData.Dimensions;
	
			// Ensure all dimension values exist and are valid numbers
			if (dims.Length > 0 && dims.Width > 0 && dims.Height > 0) {
				let dimensionString = `${dims.Length}x${dims.Width}x${dims.Height} 0`; // Always append "0"
	
				// Validate format (three numeric values + "0" at the end)
				let dimParts = dimensionString.split(new RegExp("[x ]"), 4); // Split on 'x' and space
	
				if (dimParts.length === 4 && !isNaN(dimParts[0]) && !isNaN(dimParts[1]) && !isNaN(dimParts[2]) && dimParts[3] === "0") {
					validDimensions.push(dimensionString);
					console.log(`✅ [PreShip] Package ${i} Dimensions: ${dimensionString}`);
				} else {
					console.error(`❌ [PreShip Error] Package at index ${i} has incorrectly formatted dimensions: ${dimensionString}`);
					//alert(`[PreShip Error] Package at index ${i} has incorrectly formatted dimensions.`);
					return;
				}
			} else {
				console.error(`❌ [PreShip Error] Package at index ${i} has invalid dimensions.`);
				//alert(`[PreShip Error] Package at index ${i} has invalid dimensions.`);
				return;
			}
		}
	
		// Store only valid dimensions in MiscReference5
		if (validDimensions.length > 0) {
			shipmentRequest.PackageDefaults.MiscReference5 = validDimensions.join(", ");
			console.log(`✅ [PreShip] All dimensions successfully stored in MiscReference5: ${shipmentRequest.PackageDefaults.MiscReference5}`);
		} else {
			console.warn("⚠️ [PreShip Warning] No packages had valid dimensions.");
		}
	
		// Retrieve weight and markup values
		userParams.weight = $('div[caption="Weight"]').find('input[name="unitValue_txt"]').val();
		userParams.markup = $("#custom_markup_id").val();
		console.log(`📏 [PreShip] Retrieved weight: ${userParams.weight}, Markup: ${userParams.markup}`);

					errAlert.show();
					console.warn(`⚠️ [PreShip Warning] PostShip Alert Detected: ${msg}`);
				}
			}
		}, 200);
	
		console.log("✅ [PreShip] Function completed successfully.");
	};

// ajaxGet(url) // returns the synchronous AJAX response value
function ajaxGet(url) {
        var result;
        $.ajax({
            type: 'GET',
            url: url,
            async: false,
            success: function(response) {
                result = response;
            }
        });
        return result;
    }

// ajaxPost(url, data) // returns the synchronous AJAX response or error text
function ajaxPost(url, data) {
        var results;
        $.ajax({
            type: 'POST',
            url: url,
            async: false,
            cache: false,
            dataType: 'json',
            data: data,
            success: function(response) {
                results = response;
            },
            error: function (error) {
                results = error.responseText;
            }
        });
        return results;
    }

  function GetCallingFunctionName ()
  {
    try
    {
      throw new Error();
    }
    catch (e)
    {
      try
      {
        return e.stack.split('at ')[3].split(' ')[0] + '()';
      }
      catch (e)
      {
        return '';
      }
    }
  }

      Log: function (LogObject)
      {
        try
        {
          /*
          Usage:
            Errors:
              try {
                ...
              } catch (error) {
                Logger.Log({ Source: 'FunctionName()', Error: error });
                OR
                Logger.Log({ Source: 'FunctionName()', Error: { name: 'Error name', message: 'Error message' } });
                OR
                Logger.Log({ Source: 'FunctionName()', Error: { name: 'Error name', message: 'Error message' }, Data: 'Simple or complex data object.' });
              }

            Info (default, no 'Type' needed):
              Logger.Log({ Source: 'FunctionName()', Message: 'Test of server info logging', Data: 'Simple or complex data object.' });

            All others:
              Logger.Log({ Source: 'FunctionName()', Type: 'Trace', Message: 'Test of server info logging with a complex data object.', Data: SessionData.GetCurrentUserContext() });
              Logger.Log({ Source: 'FunctionName()', Type: 'Fatal', Message: 'Test of server info logging with a simple data string.', Data: 'Simple data string literal.' });
              Logger.Log({ Source: 'FunctionName()', Type: 'Debug', Message: 'Test of server debug logging with no data object.' });
          */

          /* If the LogObject has an 'error' object, set the LogLevel here.  This makes the LogLevel variable optional for errors. */
          if (LogObject.Error)
          {
            LogObject.LogLevel = LogLevel.Error;

            /* Can't pass the error container to the server without preparing it inside a new object first. */
            var e = {};
            e.name = LogObject.Error.name;
            e.message = LogObject.Error.message;
            LogObject.Error = e;
          }
          else if (!LogObject.LogLevel)
          {
            /* If the LogObject does NOT have an 'error' object AND there is no LogLevel set, use .Info.  This makes the LogLevel variable optional for info logging. */
            LogObject.LogLevel = LogLevel.Info;
          }

          /* All logging will go to the console, output here. */
          switch (LogObject.LogLevel)
          {
            case LogLevel.Error:

              console.error('Exception encountered in', LogObject.Source);
              console.error(LogObject.Error.name);
              console.error(LogObject.Error.message);

              /* Use .trace if needed but within the Angular sandbox the .trace object can be huge and convoluted. */
              /* Be careful if passing .trace data on to the server as it will contain various data types and directives. */
              //console.error(LogObject.Error.trace);

              /* Don't use console.error for the data object, it causes readability issues. */
              if (LogObject.Data) { console.log(LogObject.Data); }
              break;

            /* Using the same nomenclature used by SBR logging. */
            case LogLevel.Info:
            case LogLevel.Debug:
            case LogLevel.Trace:
            case LogLevel.Fatal:
            default:

              console.log('Output from', LogObject.Source);
              console.log(LogObject.Message);
              if (LogObject.Data) { console.log(LogObject.Data); }
              break;
          }

          if (SessionData.GetServerDebugMode() == true)
          {
            var dataObject = {
              ServerMethod: 'AddClientEntry',
              MessageObject: LogObject
            };

            var ajaxObject = {
              UserContext: SessionData.GetCurrentUserContext(),
              Data: JSON.stringify(dataObject)
            };

            /* async doesn't necessarily need to be set to false here but I think it helps with data collisions and getting the error messages to appear in chronological order on the server. */

              url: 'api/ShippingService/UserMethod',

              data: ajaxObject,
              async: false
            });

              console.log(jqXHR);
              console.log(textStatus);

            });

          HideLoader();

// downloadPdf(pdfDocuments) // returns void
    function downloadPdf(pdfDocuments) {      
      let pdfArray = pdfDocuments.PdfData;
      let documentName = pdfDocuments.DocumentName;
      let file = {};
      console.log(pdfArray.length);
      for (let count = 0; count < pdfArray.length; count++) {
          if (navigator.userAgent.indexOf("MSIE ") > -1 || navigator.userAgent.indexOf("Trident/") > -1) {
            let byteCharacters = atob(pdfArray[count]);
            let byteNumbers = new Array(byteCharacters.length);
              for (let i = 0; i < byteCharacters.length; i++) {
                  byteNumbers[i] = byteCharacters.charCodeAt(i);
              }
              let byteArray = new Uint8Array(byteNumbers);
              file = new Blob([byteArray], { type: 'application/pdf' });
              window.navigator.msSaveOrOpenBlob(file, documentName + ".pdf");
          } else {
              let documentAfterConversion = base64ToArrayBuffer(pdfArray[count]);
              let a = window.document.createElement("a");
              file = new Blob([documentAfterConversion], { type: 'application/pdf' });
              let fileUrl = URL.createObjectURL(file);
              a.href = fileUrl;
              a.download = documentName + ".pdf";
              document.body.appendChild(a);
              a.click();
              document.body.removeChild(a);
          }          
    }
  }

    {
      console.log('btn-go-split-pkgs click event handler');

      let li = $(this).closest('li').first();
      let div = $(this).closest('div').first();

      $(li).find('.pkg-index-text').first().text('1');

      //val - 1 as the first entry will be the first package

      that.UpdateGoodsObject(v, li);

      $('.split-pkgs-modal').hide();
    });

    $('body').delegate('#btnCancelAssignGoods', 'click', function(e)
    {
      that.HideAssignGoodsPane();
    });

    //if the user escapes out of the split modal, hide the modal and show the previous forms
    $('body').delegate('.split-pkgs-modal', 'keydown', function(e)
    {
      var keycode = (e.keyCode ? e.keyCode : e.which);
      if (keycode === 27)
      {
        //need to show the previous input and button now
        let li = $(this).closest('li');

        that.CancelSplitOfPackages(li);
      }
    });

    //if the user presses enter or tab in the num-of-items number input, move to the next one
    //these are the number inputs after the items have been split across packages
    $('body').delegate('.num-of-items-in-pkg', 'keydown focusout', function(e)
    {
      console.log('.num-of-items-in-pkg keydown focusout event handler');

      $('[data-toggle="tooltip"]').tooltip('hide');

      var keycode = (e.keyCode ? e.keyCode : e.which);
      if (keycode === 13 || keycode === 9)
      {
        //disable the input, update the commodity object, focus on the next box input
        //$(this).closest('div').find('.assign-to-pkg').focus();
        that.UpdateListItemPackageQuantities($(this).closest('li'));
      }
    });

    //user hits enter in the input box, click the button for them
    $('body').delegate('.num-of-pkgs-to-use', 'keyup keydown', function(e)
    {
      console.log('.num-of-pkgs-to-use keyup keydown event handler');
      var keycode = (e.keyCode ? e.keyCode : e.which);
      let btn = $(this).closest('div:visible').find('input.btn-go-split-pkgs');

      $(btn).attr('disabled', $(this).val() < 2);

      if (keycode == 13) //enter
      {
        //let btn = $(this).closest('div:visible').find('input.btn-go-split-pkgs');
        $(btn).trigger('click');
      }
    });

    //this was formerly .scan-input.  as of 9/24/2022, removing the split functionality completely
    //with this change, only need to get the numerical value out of the input
    $('body').delegate('.assign-to-pkg', 'keydown focusout', function(e)
    {
      console.log('.assign-to-pkg keydown focusout event handler');
      var keycode = (e.keyCode ? e.keyCode : e.which);
      debugger;
      if (keycode == 13 || keycode == 9 || e.type == "focusout") //enter
      {
        e.preventDefault();
        //pull only the number from whatever was passed.  make sure it is greater than 0 since we can't assign a good a pkg zero.
        const val = $(this).val().replace(/[^1-9]/g, '');

        let li = $(this).closest('li');
        
        if (val)
        {
          //console.log('assigning this to ', val);
          //assign the integer to the line item header (not shown, but may be shown in the future)
          //this one is easy, just assign the pkg-index attr to whatever they scanned.  when the apply button is clicked it will assign the commodities
          //let li = $(this).closest('li');
          $(li).find('.pkg-index-text').first().text(val);
          $(li).find('button').attr('disabled', 'disabled');
          $(li).attr('pkg-index', val - 1);
          $(li).attr('pkg-status', 'packed');

          //find the matching item in the this.working array and update the package index
          let item = $.grep(that.working, function(good, i) { return good.$$hashKey == $(li).attr('hash-key') })[0];
          item.packageIndex = val - 1;
          //return;
        }
        // set focus to next li input box
        if (e.type !== "focusout") {
          $(li).next().find('.assign-to-pkg').focus().change();
        }
      }
    });

    $('#btnAssignCommodities').on('click', function(e)
    {
      if ($('#divAssignCommodities').is(':visible')) return;
      that.ShowAssignGoodsPane();
    });
  }

		const blob = new Blob([byte], {
			type: "application/vnd.ms-excel"
		});

		const ie = navigator.userAgent.match(/MSIE\s([\d.]+)/),
			ie11 = navigator.userAgent.match(/Trident\/7.0/) && navigator.userAgent.match(/rv:11/),

		if (ieVer > -1) {
			window.navigator.msSaveBlob(blob, fileName);

			document.body.appendChild(downloadLink);
			downloadLink.click();
			document.body.removeChild(downloadLink);

      let that = this;
      let data = $(li).data();
      let destCC = this.packages[destBox - 1].CommodityContents || [];

      //find the box the item currently resides in
      //split this statement up over a few calls to make it easier to read
      let allGoods = this.packages.map(m => m.CommodityContents).flat();
      let item = allGoods.find(cc => cc.UniqueId == data.UniqueId);

      if (+item.PkgIndex == +destBox) { console.log('Item already exists in box.'); return false; }

      //remove the item from its current box
      this.packages[item.PkgIndex - 1].CommodityContents = this.packages[item.PkgIndex - 1].CommodityContents.filter(p => p.UniqueId != data.UniqueId);

      //add it to the new box
      item.PkgIndex = destBox;
      destCC.push(item);
      $(li).find('.pkg-index-text').fadeOut().promise().done(function(e) { $(this).text(destBox).fadeIn(); });

      //console.log(vm.currentShipment.Packages);
      $('li[heading="Goods"]:visible:first > a').trigger('click');
      $('#goods .btn-group').find('.btn:gt(0)').not('.active').first().trigger('click');

      //update the weights and values
      //this data will have changed by now so get a new reference to the object
      allGoods = this.packages.map(m => m.CommodityContents).flat();
      //the items array will have the CommodityContents object for every package in the shipment

      //sort by package index
      //update the weight and declared values for all the packages in the shipment
      allGoods = allGoods.sort((a, b) => { return a.PkgIndex - b.PkgIndex });
      $(allGoods).each(function(a, b)
      {

        }

        pkg.DeclaredValueAmount.Amount += +b.PVTotalValue;
      });
    }

  RemoveContainerData = (ContainerIndex) =>
  {
    let msg = {};
    try
    {
      let that = this;
      ContainerIndex = +ContainerIndex;
      this.Containers.Count--;

      this.UpdateContainerArrays('Remove', ContainerIndex);

      //update the data objects in all the rows to remove the passed container, then decrement any containers with a higher index than the deleted,
      //then update the attributes and repaint the screen
      let data = this.DataStore.TableRowPackedItems();
      this.Items.Packed -= +this.DataStore.ContainerPackedItemCount(ContainerIndex);

      //the below line will remove all items with the removed container index from the table data
      data.map(f => f.PackedItems = f.PackedItems.filter(g => g.ContainerIndex != ContainerIndex));

      //this will decrement the container indexes for containers above the removed one
      data.map(f => f.PackedItems.map(g => g.ContainerIndex = (g.ContainerIndex > ContainerIndex ? g.ContainerIndex - 1 : g.ContainerIndex)));

      this.RemoveContainerHtml(ContainerIndex);

      msg.Action = 'RemoveContainer';
      msg.ContainerIndex = ContainerIndex;

      this.UpdateToastNotification(msg);
    }
    catch (error)
    {
      console.log(error);
      msg.Action = 'Error';
      msg.ErrorMessage = error.message;
      this.UpdateToastNotification(msg);
    }
  }

  {

    //setup the event handlers
    let that = this;

    //event handler for selecting a container
    let ul = $('#ulMultipleItemContainerDropdown');
    $(ul).delegate('li', 'click', function(e)
    {
      e.preventDefault();
      e.stopPropagation();
      e.stopImmediatePropagation();

      $(that.multipleOperationsDialog).find('span.ci').first().text($(this).text());

    });

    $('#textQuantityToMove').on('focusout', function(e)
    {
      const i = +$(this).val();
      const m = +$(this).attr('max');

      $('#btnOkMultipleOperations').attr('disabled', (i > m));

      if (i > m)
      {
        $(this).addClass('has-error');
        $(this).focus();
        $(this).tooltip('show');
        return;
      }

      $(this).removeClass('has-error');
    });

    //clicking on the dialog buttons
    $(this.multipleOperationsDialog).find('button[op]').on('click', function(e)
    {
      const op = $(this).attr('op');

      switch (op)
      {
        case 'ok':
          let [rowid, qty, ci, action] = [$(this).data('RowId'), +$('#textQuantityToMove').val(), $('span.ci').first().text(), $(this).data('Action')];
          let row = that.packVerify.DataStore.TableRowDataObject(rowid);

          switch (action)
          {
            case 'add':
              that.packVerify.AddItemDataToContainer(row, qty, ci);
              break;
            case 'remove':
              that.packVerify.RemoveItemDataFromContainer(rowid, qty, ci);
              break;
          }
          break;

        case 'cancel':
          break;

        default:
          break;
      }

      that.bRemove = false;

      if (that.packVerify.Items.Ordered !== that.packVerify.Items.Packed) that.packVerify.Overlay.Hide();
      $(that.multipleOperationsDialog).fadeOut().find('input').val([]).find('[data-toggle="tooltip"]').tooltip('destroy');
      $(that.multipleOperationsDialog).find('span.footer-max').first().fadeOut();
    });

    //fired when the user selects an item from the add/remove multiple items autocomplete dialog
    //need to filter out from the container dropdown here so when a user is removing items, they don't have the option
    //to remove from a container that doesn't have the item
    $(that.multipleOperationsDialog).delegate('[event="SelectItemDropdown"]', 'click', function(e)
    {
      e.preventDefault();
      e.stopPropagation();
      e.stopImmediatePropagation();

      const data = $(this).data();
      let maxValue = 0;

      if (that.bRemove)
      {
        const list = that.packVerify.DataStore.GetContainersHavingItem(data.RowId);
        let ul = $('#ulMultipleItemContainerDropdown'), li = '<li><a href="#"></a></li>';

        $(list).each(function(a, b)
        {
          let tt = `data-max="${b.Count}" data-toggle="tooltip" title="Maximum: ${b.Count}" data-container="body" data-html="true"`;
          li += `<li ${tt}><a href="#">${b.ContainerIndex}</a></li>`;
        });

        $('#btnContainerDropDown').removeAttr('disabled');

        $(ul).empty().append(li);

        if (list.length == 1)
        {
          //if there's only a single container, set the variable
          $(that.multipleOperationsDialog).find('span.ci').first().text(list[0].ContainerIndex);
          maxValue = list[0].Count;
        }

        $(ul).find('[data-toggle="tooltip"]').tooltip();
      }

      //store the row information with the button data object so when the user clicks 'ok', the data can be read and processed
      $('#btnOkMultipleOperations').data(data);
      $('input[event="ShowMatchingItems"]').val(data.ItemNumber);

      //change the strings depending on whether adding/removing
      let header = (that.bRemove ? 'remove.' : 'add.');
      let max = (that.bRemove ? maxValue : data.QuantityRemaining);

      $('#btnOkMultipleOperations').data('Action', header.slice(0, -1));

      $('#textQuantityToMove').attr({ 'max': max });
      $(that.multipleOperationsDialog).find('span.max-value').first().text(max);
      $(that.multipleOperationsDialog).find('span.footer-max').first().fadeIn();

      $('#ulItemAutoComplete').find('li').slideUp(250).closest('div').hide().promise().done(function(t)
      {
        if (that.bRemove)
        {
          $('#btnContainerDropDown').focus();
          if ($('#ulMultipleItemContainerDropdown').find('li').length > 2) { $('#ulMultipleItemContainerDropdown').slideDown(); }
        }
        else
        {
          $('#textQuantityToMove').focus();
        }
      });

      if ($(e.target).hasClass('all-avail')) { $('#textQuantityToMove').val(max); $('#btnContainerDropDown').focus(); }
    });

    $('[event="ShowMatchingItems"]').on('keyup search', function(e)
    {
      const ul = $('#ulItemAutoComplete');
      const [val, items, bVisible] = [$(this).val(), $(ul).find('li:visible'), $(ul).find('li:visible').length > 0];

      //need to deal with a few different scenarios
      //did the user backspace and now there are <= 2 chars?  if so, slide up all the visible list items and then remove them.  then return out after updating the tooltip.
      if (val.length < 1)
      {
        if (bVisible) $(ul).find('li').slideUp(200).promise().done((e) => { $(e).remove(); });

        //hide the menu container
        $('#divItemAutoComplete').fadeOut('fast');

        //update the tooltip
        //$('span.tooltip-search.tooltip-results').text('enter at least 3 letters');
        return;
      }

      //the menu is already visible. check the entries that are visible to see if they match the search pattern
      //then need to run the query again as we won't know if the user backspaced or edited the middle of the string
      if (bVisible)
      {
        //get rid of list items that no longer match after the text was changed.  make sure the return stays outside the .slide
        const listitems = $(items).filter((a, b) => $(b).text().toLocaleUpperCase().indexOf(val.toLocaleUpperCase()) === -1);
        $(listitems).slideUp(200).promise().done((e) => { $(e).remove(); that.ShowAutoComplete(val); });
      }
      else
      {
        //user cleared the box or clicked the 'x'
        $(ul).empty();
        that.ShowAutoComplete(val);
      }
    });
  }
