      CheckForMatchingConsignee: function (CustomerOrderObject)
      {
        try
        {
          var con = $('name-address[nameaddress="vm.currentShipment.PackageDefaults.Consignee"]');

          let a = {
            //conCompany: $(con).find('input[name="company"]').val().trim().toUpperCase().replaceAll(' ',''),
            conAddress1: $(con).find('input[name="address1"]').val().trim().toUpperCase(),
            conCity: $(con).find('input[name="city"]').val().trim().toUpperCase(),
            conState: $(con).find('input[name="stateProvince"]').val().trim().toUpperCase(),
            conPostalCode: $(con).find('input[name="postalCode"]').val().trim().substring(0, 5)
          };

          let b = {
            //conCompany: CustomerOrderObject.shipAddress1.CONSIGNEE_COMPANY.trim().toUpperCase().replaceAll(' ', ''),
            conAddress1: CustomerOrderObject.shipAddress1.trim().toUpperCase(),
            conCity: CustomerOrderObject.shipCity.trim().toUpperCase(),
            conState: CustomerOrderObject.shipState.trim().toUpperCase(),
            conPostalCode: CustomerOrderObject.shipZipCode.trim().substring(0, 5)
          };

          /* Remove all the spaces in the strings so we get a good comparison, not one that's thrown off by an extra space. */
          let aa = JSON.stringify(a).replaceAll(' ', '');
          let bb = JSON.stringify(b).replaceAll(' ', '');

          /* Quick comparison to see if the strings match. */
          if (aa != bb)
          {
            Logger.Log({ Source: 'CheckForMatchingConsignee()', Message: 'Could not add order ' + CustomerOrderObject.orderNumber + ' (' + CustomerOrderObject.orderId + '). Primary order data to follow.', Data: a });
            Logger.Log({ Source: 'CheckForMatchingConsignee()', Message: 'Could not add order ' + CustomerOrderObject.orderNumber + ' (' + CustomerOrderObject.orderId + '). Subsequent order data to follow.', Data: b });
          }

          return (aa == bb);
        }
        catch (error)
        {
          Logger.Log({ Source: 'CheckForMatchingConsignee()', Error: error });
        }
      }

// CheckValidation(referenceValue, referenceList) // returns void; may throw an error when validation fails
    this.CheckValidation = function(referenceValue, referenceList) {
        var readyToShip = false;
        for (var item in referenceList) {
            //alert(referenceList[item]);
            if (referenceList[item].Value == referenceValue) {
                readyToShip = true;
                break;
            }
        }
        if (readyToShip === false) throw { message: "Unable to validate shipment", errorCode: "001" };
    };

// this.PreShip(shipmentRequest, userParams) // returns undefined
this.PreShip = function(shipmentRequest, userParams) {    
  
    // We will require the actual shipment to send weight and dims in kg/cm
    // However, because US Vendors are sending the product the client gui needs to be in Lb/In.
    
    debugger;
    //kg = lbs*0.453592 
    totalShipWeight = 0;
    totalVolumetricResult = 0;
    
    vm.currentShipment.Packages.forEach(pkg => {
        totalShipWeight += Number(pkg.Weight.Amount);
        totalVolumetricResult += ((Number(pkg.Dimensions.Length).toFixed(2)*Number(pkg.Dimensions.Width).toFixed(2)*Number(pkg.Dimensions.Height)).toFixed(2))/166;
    });
              
    totalShipWeight = Number(totalShipWeight).toFixed(2);
    console.log(totalShipWeight);
    console.log(totalVolumetricResult);
    if (totalShipWeight > 150)
    {
        alert("Total Shipment Weight Exceeds Shipment Weight Limit of 150lbs")
        throw new Error("Shipping halted due to Over Maximum Weight Rule " + totalShipWeight);
    }
    
    if (totalVolumetricResult > Number(maxvolumetric))
    {
        alert("Total Shipment Weight Exceeds Total Shipment Volumetric Weight Limit of " + Number(maxvolumetric))
        throw new Error("Shipping halted due to Total Shipment Volumetric Weight Rule " + Number(totalVolumetricResult).toFixed(2));
    }
    
     vm.currentShipment.Packages.forEach(pkg => {
         if (pkg.Weight.Units === "LB"){
             pkg.Weight.Amount = Number(pkg.Weight.Amount * 0.453592).toFixed(2);
             pkg.Weight.Units = "KG";
         }
         
     });
     
    //cm=in/0.39370
     vm.currentShipment.Packages.forEach(pkg => {
         if (pkg.Dimensions.Units === 0){
             pkg.Dimensions.Length = Number(pkg.Dimensions.Length/0.39370).toFixed(2)
             pkg.Dimensions.Width = Number(pkg.Dimensions.Width/0.39370).toFixed(2)
             pkg.Dimensions.Height = Number(pkg.Dimensions.Height/0.39370).toFixed(2)
             pkg.Dimensions.Units = 1;
         }
         
     });
     
     
    // debugger;
      if(Number(vm.currentShipment.Packages[0].DeclaredValueCustoms.Amount).toFixed(2) > 2500 ){
          if(vm.currentShipment.Packages[0].ExportInformationCode === null){
              vm.currentShipment.Packages[vm.packageIndex].ExportInformationCode = "OS";
          }
          if(vm.currentShipment.Packages[0].AesTransactionNumber === null){
              vm.currentShipment.Packages[0].SedMethod = 2;
              $('input[type=text][ng-model="vm.currentShipment.Packages[vm.packageIndex].AesTransactionNumber"]').focus();
              throw new Error('Missing AES/ITN Number');
          } 
          if(shipmentRequest.Packages[0].AesTransactionNumber !== null){
              shipmentRequest.Packages[0].Comments = shipmentRequest.Packages[0].Comments + "\n ITN/AES Filer " + shipmentRequest.Packages[0].AesTransactionNumber;
             
          }
      }
     
    console.log(shipmentRequest);
  
};

// this.ShippingLoaded() // returns undefined
this.ShippingLoaded = function(){
    this.vm.populateReturnAddress = this.PopulateReturnAddress;
    this.vm.populateShipToAddress = this.PopulateShipToAddress;
    this.vm.serviceOption = serviceOption;
    this.vm.serviceOptions = serviceOptions(this.vm.profile);
    this.vm.currentShipment.Packages[0].ShipperReference = user_orderNumber;
    this.vm.currentShipment.Packages[0].MiscReference1 = user_orderNumber;
    this.vm.currentShipment.Packages[0].ConsigneeReference = cost_Center;
    this.vm.currentShipment.Packages[0].MiscReference2 = cost_Center;
    this.vm.currentShipment.Packages[0].MiscReference3 = user_partNumber;
    this.vm.currentShipment.Packages[0].Description = user_description;
   

    function serviceOption() {
        this.selectedServices = [this.currentShipment.PackageDefaults.Service];
    }
    function serviceOptions(profile) {
        //debugger;
        var filterServices = [];
        var customData3 = uPS_Service.split(',');
        for (var item of customData3) {
            var symbol = getServiceSymbol(item.toLowerCase());
            for (var service of profile.Services) {
                if (service.Symbol === symbol) {
                    service.Name = service.Name.replace('(UPS Adapter)','').trim();
                    if (service.Name.includes("Expedited")){
                        service.Name = service.Name + " (MAIN SERVICE)";
                    }
                    if (service.Name.includes("Saver")){
                        service.Name = service.Name + " (Only with Boston's Authorization)";
                    }
                    filterServices.push(service);
                    break;
                }
            }
        }
        return filterServices;
        function getServiceSymbol(code) {
            if (code == '2dn') return 'UPSAPI.UPS.2DA'
            else if (code == '3dd') return 'UPSAPI.UPS.3DS'
            else if (code == 'exp') return 'UPSAPI.UPS.EXPEDITED'
            else if (code == 'ess') return 'UPSAPI.UPS.EXPRESS'
            else if (code == 'nda') return 'UPSAPI.UPS.NDA'
            else if (code == 'dae') return 'UPSAPI.UPS.NDAEAM'
            else if (code == 'grn') return 'UPSAPI.UPS.GND'
            else if (code == 'uwe') return 'UPSAPI.UPS.EXPEDITED'
            else if (code == 'wep') return 'UPSAPI.UPS.EXPRESSSP'
            else if (code == 'swe') return 'UPSAPI.UPS.EXPRESS'
            else if (code == 'uws') return 'UPSAPI.UPS.SAVER'
            else if (code == 'std') return 'UPSAPI.UPS.STANDARD'
        }
        
    }
    
    if(dims.includes(',')){
        var dimensions = dims.split(',');
        this.vm.currentShipment.Packages[0].Weight.Amount = Number(dimensions[0]);
        maxvolumetric = Number(dimensions[4]);
        console.log("Max Volumetric: " + maxvolumetric);
        this.vm.currentShipment.Packages[0].Dimensions = {
            Length: Number(dimensions[1]),
            Height: Number(dimensions[2]),
            Width: Number(dimensions[3]),
            Units: 0
        };
    }
   
    vm = this.vm;
};

  this.PreLoad = function (loadValue, shipmentRequest, userParams) {

    // 🔄 Reset any previous SWOG starter
    masterOrderNumber = "";

    // 🧾 First prompt: general input
    var firstInput = window.prompt(
        'Enter Package Dimensions (###x###x###)\n' +
        '— OR type PACKSIZE_CONTINUE to skip dimensions\n' +
        '— OR enter a SWOG Order Number to start the SWOG function:',
        ''
    );

    if (firstInput === null) {
        userParams.dimensions = "";
        return;  // ❌ Cancelled
    }

    firstInput = firstInput.trim();
    var dimRegex = /^[0-9]+\s*x\s*[0-9]+\s*x\s*[0-9]+$/i;

    if (firstInput.toUpperCase() === 'PACKSIZE_CONTINUE') {
        userParams.dimensions = 'PACKSIZE_CONTINUE';
        return;  // ✅ Skip dimensions
    }

    if (dimRegex.test(firstInput)) {
        userParams.dimensions = firstInput.toLowerCase();
        return;  // ✅ Valid dimensions directly
    }

    // 🌀 Else: treat as SWOG starter
    masterOrderNumber = firstInput;
    this.vm.swogShipment();

    // 🧾 Second prompt after SWOG
    var dimsAfterSwog = window.prompt(
        'Enter Package Dimensions (###x###x###), type PACKSIZE_CONTINUE to skip, or leave blank:',
        ''
    );

    if (dimsAfterSwog === null) {
        userParams.dimensions = "";  // ❌ Cancelled
        return;
    }

    dimsAfterSwog = dimsAfterSwog.trim();

    if (dimsAfterSwog.toUpperCase() === 'PACKSIZE_CONTINUE') {
        userParams.dimensions = 'PACKSIZE_CONTINUE';  // ✅ PACKSIZE_CONTINUE accepted in second prompt too
    } else if (dimRegex.test(dimsAfterSwog)) {
        userParams.dimensions = dimsAfterSwog.toLowerCase();  // ✅ Dimensions entered
    } else {
        userParams.dimensions = "";  // ❌ Nothing valid entered
    }
}

		for (let i = 0; i < shipmentRequest.Packages.length; i++) {
			console.log(`🔍 [PreShip] Checking package at index ${i}...`);
	
			let packageData = shipmentRequest.Packages[i];

				//alert(`[PreShip Error] Package at index ${i} is undefined.`);
				return;
			}
	
			if (!packageData.Dimensions) {
				console.warn(`⚠️ [PreShip Warning] Package at index ${i} does not contain 'Dimensions'.`);
				//alert(`[PreShip Error] Package at index ${i} does not contain 'Dimensions'.`);
				return;
			}
	
			let dims = packageData.Dimensions;
	
			// Ensure all dimension values exist and are valid numbers
			if (dims.Length > 0 && dims.Width > 0 && dims.Height > 0) {
				let dimensionString = `${dims.Length}x${dims.Width}x${dims.Height} 0`; // Always append "0"

				let dimParts = dimensionString.split(new RegExp("[x ]"), 4); // Split on 'x' and space
	
				if (dimParts.length === 4 && !isNaN(dimParts[0]) && !isNaN(dimParts[1]) && !isNaN(dimParts[2]) && dimParts[3] === "0") {
					validDimensions.push(dimensionString);
					console.log(`✅ [PreShip] Package ${i} Dimensions: ${dimensionString}`);
				} else {
					console.error(`❌ [PreShip Error] Package at index ${i} has incorrectly formatted dimensions: ${dimensionString}`);
					//alert(`[PreShip Error] Package at index ${i} has incorrectly formatted dimensions.`);
					return;
				}
			} else {
				console.error(`❌ [PreShip Error] Package at index ${i} has invalid dimensions.`);
				//alert(`[PreShip Error] Package at index ${i} has invalid dimensions.`);
				return;
			}
		}
	
		// Store only valid dimensions in MiscReference5
		if (validDimensions.length > 0) {
			shipmentRequest.PackageDefaults.MiscReference5 = validDimensions.join(", ");
			console.log(`✅ [PreShip] All dimensions successfully stored in MiscReference5: ${shipmentRequest.PackageDefaults.MiscReference5}`);
		} else {
			console.warn("⚠️ [PreShip Warning] No packages had valid dimensions.");
		}
	
		// Retrieve weight and markup values
		userParams.weight = $('div[caption="Weight"]').find('input[name="unitValue_txt"]').val();
		userParams.markup = $("#custom_markup_id").val();
		console.log(`📏 [PreShip] Retrieved weight: ${userParams.weight}, Markup: ${userParams.markup}`);

// AddOrderNumbersToList() // returns false when a duplicate order number is detected, otherwise undefined
      AddOrderNumbersToList: function()
      {
        var inVal = $('#textOrderNumber').val();
        var bExists;

        /* Do not add duplicate entries to the select box. */
        $('#selectConsolidatedOrders').find('option').each(function()
        {
          if (this.text == inVal)
          {
            bExists = true;
            /* Break out of the .each loop. */
            return false;
          }
        });

        if (bExists == true)
        {
          $('#textOrderNumber').val([]).focus();
          return false;
        }

        /* The order number is unique, add it to the select box. */
        if (inVal.length > 0)
        {
          $('#selectConsolidatedOrders')
            .append($('<option />')
              .val(inVal)
              .text(inVal)
              .addClass('not-processed')
              .trigger('change'))
            .promise().done(function()
            {
              $('#textOrderNumber').val([]).focus();
            });
        }
      },

            }

function selectCanadaForAllCountrySelects() {
        // Check if the flag is already true, which means we no longer need to change the selects
        if(canadaSelected || !isSelectedValueInShippersArray()) {
          return; // Exit the function if Canada has been selected in all dropdowns or if the site is not from Canada
        }
        
        // Get all select elements with the name 'Country'
        var selects = document.querySelectorAll('select[name="Country"]');
        var allSelected = true; // To keep track if all selects have been set to Canada
      
        // Iterate through the NodeList
        selects.forEach(function(select) {
          // Only set the value to 'string:CA' if it's not already set
          if (select.value !== 'string:CA') {
            select.value = 'string:CA';
            allSelected = false; // Not all selects have been set to Canada
      
            // Trigger change event for AngularJS
            if ('createEvent' in document) {
              var evt = document.createEvent('HTMLEvents');
              evt.initEvent('change', false, true);
              select.dispatchEvent(evt);
            } else {
              select.fireEvent('onchange');
            }
          }
        });
      
        // If all selects have been set to Canada, set the flag to true
        if (allSelected) {
          canadaSelected = true;
        }
      }

  UpdateListItemPackageQuantities(li)
  {
    this.UpdatePkgCountsInWorkingObject(li);

    let numOfItems = $(li).find('.num-of-items-in-pkg');
    let remItems = $(li).find('.goods-quantity > strong');

    if (!$(numOfItems).val()) { return; }

    let x = (+$(remItems).text() - $(numOfItems).val());

    //sanity check to make sure we didn't pack more items that we had
    if (x < 0)
    {
      $(remItems).addClass('bad-math');
      return;
    }

    $(remItems).removeClass('bad-math');
    $(numOfItems).attr('disabled', 'disabled');
    $(li).attr('pkg-status', 'packed');

    //move to the next list item and update then focus
    $(li).next().find('.goods-quantity > strong').text(x)

    //on the second-to-last list item, use the input value to auto-set the last item
    //to get the number of rows that have already been assigned, count the list items with the goods-key
    //that have disabled input boxes
    let rowsForThisGood = $('#listItemContainer').find('li[goods-key="' + $(li).attr('goods-key') + '"]');
    let rowsDone = $(rowsForThisGood).find('input:disabled:visible').length;

    if (rowsForThisGood.length == (rowsDone + 1))
    {
      let lastLi = $('#listItemContainer').find('li[goods-key="' + $(li).attr('goods-key') + '"]').last();

      $(lastLi).attr('pkg-status', 'packed');

      let lastVal = $(li).next().find('.goods-quantity > strong').text();
      $(lastLi).find('.num-of-items-in-pkg').val(lastVal).attr('disabled', 'disabled');

      $('#listItemContainer').find('.assign-to-pkg:visible').first().focus();

      this.UpdatePkgCountsInWorkingObject(lastLi);
    }
  }

  UpdatePkgCountsInWorkingObject(li)

// ApplyCommodityAssignments() // returns void
  ApplyCommodityAssignments()
  {
    let that = this;

    //every list item will have a pkg-index attr, get that highest value to determine how many pkgs we need
    //don't use pkg-status.length because that will include multiple commodities and the end result will be wrong
    let packed = $('#listItemContainer').find('li[pkg-status=packed]');

    //first get all the pkg-index attributes, then get the highest number and add one
    let pkgs = $(packed).map(function(x, value) { return +$(this).attr('pkg-index') }).toArray();
    let pkgCount = Math.max(...pkgs) + 1;

    while (this.packages.length < pkgCount)
    {
      //the vm.addPackage isn't firing, just click the button
      $('button[ng-click="vm.addPackage();"]').trigger('click').promise().done(function(e) { });
    }

    //loop the pkg count, get the commodities for that pkg, write the objects to the vm CommodityContents object
    for (let p = 0; p < pkgCount; p++)
    {
      let w = $.grep(this.working, function(item, i) { return item.packageIndex == p });
      this.packages[p].CommodityContents = w;
    }

    //click the General tab
    $('li[heading=General]:visible a').first().click();
    this.HideAssignGoodsPane();

    //after we are done, move back to the first package in the shipment
    while ($('#spanCurrentPackageIndex').text() !== '1')
    {
      $('button[ng-click="vm.previousPackage();"]').first().click();
    }

  }

// UpdateGoodsObject(pkgCount, listItem) // returns void
  UpdateGoodsObject(pkgCount, listItem)
  {
    const itemkey = $(listItem).attr('hash-key');
    const goodkey = $(listItem).attr('goods-key');
    const findex = +$(listItem).attr('pkg-index');

    let val = $(listItem).find('.goods-declared-value');
    let cost = $(val).find('strong').text().replace('$', '');

    cost = (cost / pkgCount).toFixed(2);

    //get the object that matches the hash-key from the working array
    //this will be the template to use when creating new packages after a split operation
    //working on a copy of the vm commodity object so we can abandon changes if needed
    let src = $.grep(this.working, function(value, i) { return value.$$hashKey == itemkey; })[0];

    //update the declared value for the first package
    src.UnitValue.Amount = cost;

    for (let y = 1; y < pkgCount; y++)
    {
      let ccobj = this.GetNewCommodityObject(src);
      ccobj.packageIndex = findex + y;

      this.working.push(ccobj);

      this.UpdateScreenMarkup(ccobj, listItem);
    }
  }

// GetNewCommodityObject(obj) // returns object
  GetNewCommodityObject(obj)
  {
    let c = {};

    Object.keys(obj).forEach(key =>
    {
      if (key != '$$hashKey' && key != 'packageIndex')
      {
        let value = obj[key];
        if (value) c[key] = value;
      }
    });

    return c;
  }

    this.convertFileInputToBase64 = function (inputId, leg) {
        return new Promise(function (resolve, reject) {

            var data = {};
            const fileInput = $('#' + inputId);

            // Do not continue if there are no files
            if (!fileInput ||
                fileInput.length === 0 ||
                !fileInput[0].files ||
                fileInput[0].files.length === 0) {
                return;
            }

            // Get basic file info
            var file = fileInput[0].files[0];
            var fileName = file.name;

            // Create reader for reading files
            var reader = new FileReader();
            reader.readAsDataURL(file);
            reader.addEventListener("load", function () {

                // Create and return payload for upload call
                data = {
                    Action: "PaperlessDocUpload",
                    FileName: fileName,
                    encodedString: reader.result.split(',')[1],
                    shipperName: that.currentShipment.PackageDefaults.Shipper,
                    leg: leg
                }

                resolve(data);

            }, false);

            reader.addEventListener("error", function (error) {
                reject(error);
            }, false);

        });

    }

// stringToDate(_date, _format, _delimiter) // returns Date
function stringToDate(_date, _format, _delimiter)
    {
        var formatLowerCase = _format.toLowerCase();
        var formatItems = formatLowerCase.split(_delimiter);
        var dateItems = _date.split(_delimiter);
        var monthIndex = formatItems.indexOf("mm");
        var dayIndex = formatItems.indexOf("dd");
        var yearIndex = formatItems.indexOf("yyyy");
        var month = parseInt(dateItems[monthIndex]);
        month -= 1;
        var formatedDate = new Date(dateItems[yearIndex], month, dateItems[dayIndex]);
        return formatedDate;
    }

// formatDate(date) // returns formatted date string
function formatDate(date) 
    {
        var d = new Date(date),
            month = '' + (d.getMonth() + 1),
            day = '' + d.getDate(),
            year = d.getFullYear();

        if (month.length < 2) 
        {
            month = '0' + month;
        }
        if (day.length < 2)
        {
            day = '0' + day;
        }

        month = month.slice(-2);

        day = day.slice(-2);

        return [year, month, day].join('-');
    }

    }

            weightLimit = 150;
    }else{
        weightLimit = 68;
    }
    var i;
    for(i = 0; i < shipmentRequest.Packages.length; i++)

    }
}
