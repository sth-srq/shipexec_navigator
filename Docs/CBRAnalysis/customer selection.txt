// AddOrderNumbersToList() // returns false|void
      AddOrderNumbersToList: function ()
      {
        var inVal = $('#textOrderNumber').val();
        var bExists;

        /* Do not add duplicate entries to the select box. */
        $('#selectConsolidatedOrders').find('option').each(function ()
        {
          if (this.text == inVal) 
          {
            bExists = true;
            /* Break out of the .each loop. */
            return false;
          }
        });

        if (bExists == true)
        {
          $('#textOrderNumber').val([]).focus();
          return false;
        }

        /* The order number is unique, add it to the select box. */
        if (inVal.length > 0)
        {
          $('#selectConsolidatedOrders').append($('<option />')
            .val(inVal)
            .text(inVal)
            .addClass('not-processed')
            .trigger('change'))
            .promise().done(function ()
            {
              $('#textOrderNumber').val([]).focus();
            });
        }
      }

// ProcessDataFromCustomer(jsonData) // returns void
      ProcessDataFromCustomer: function (jsonData)
      {
        try
        {
          $(jsonData).each(function (a, b)
          {
            /* For testing, this is set to '!== true' so it will return order data.  This needs to be set to '=== true' for any other case. */
            if (ConsolidatedOrders.CheckForMatchingConsignee(b) == true)
            {
              var currentPackage = Tools.GetCurrentPackage();
              if (!currentPackage)
              {
                Tools.ShowErrorAlert('ERROR:  Could not add commodities to package for order ' + b.orderNumber + '.  See error console for details.');
                return false;
              }
              ConsolidatedOrders.UpdateReferencesWithOrderNumbers(b.orderNumber);
            }
            else
            {
              Tools.ShowErrorAlert('ERROR 500:  Could not add commodities for order ' + b.orderNumber + '.  Address mismatch.');
            }
          });
        }
        catch (error)
        { 
          console.log(error);
        }
      }

// CheckForMatchingConsignee(CustomerOrderObject) // returns boolean|undefined
      CheckForMatchingConsignee: function (CustomerOrderObject)
      {
        try
        {
          var con = $('name-address[nameaddress="vm.currentShipment.PackageDefaults.Consignee"]');

          let a = {
            //conCompany: $(con).find('input[name="company"]').val().trim().toUpperCase().replaceAll(' ','') ,
            conAddress1: $(con).find('input[name="address1"]').val().trim().toUpperCase(),
            conCity: $(con).find('input[name="city"]').val().trim().toUpperCase(),
            conState: $(con).find('input[name="stateProvince"]').val().trim().toUpperCase(),
            conPostalCode: $(con).find('input[name="postalCode"]').val().trim().substring(0, 5)
          };

          let b = {
            //conCompany: CustomerOrderObject.shipAddress1.CONSIGNEE_COMPANY.trim().toUpperCase().replaceAll(' ', ''),
            conAddress1: CustomerOrderObject.shipAddress1.trim().toUpperCase(),
            conCity: CustomerOrderObject.shipCity.trim().toUpperCase(),
            conState: CustomerOrderObject.shipState.trim().toUpperCase(),
            conPostalCode: CustomerOrderObject.shipZipCode.trim().substring(0, 5)
          };

          /* Remove all the spaces in the strings so we get a good comparison, not one that's thrown off by an extra space. */
          let aa = JSON.stringify(a).replaceAll(' ', '');
          let bb = JSON.stringify(b).replaceAll(' ', '');

          /* Quick comparison to see if the strings match. */
          if (aa != bb)
          {
            Logger.Log({ Source: 'CheckForMatchingConsignee()', Message: 'Could not add order ' + CustomerOrderObject.orderNumber + ' (' + CustomerOrderObject.orderId + '). Primary order data to follow.', Data: a });
            Logger.Log({ Source: 'CheckForMatchingConsignee()', Message: 'Could not add order ' + CustomerOrderObject.orderNumber + ' (' + CustomerOrderObject.orderId + '). Subsequent order data to follow.', Data: b });
          }

          return (aa == bb);
        }
        catch (error)
        {
          Logger.Log({ Source: 'CheckForMatchingConsignee()', Error: error });
        }
      }

// UpdateReferencesWithOrderNumbers(OrderNumber) // returns void
      UpdateReferencesWithOrderNumbers: function (OrderNumber)
      {
        var inputShipperReference = $('input[type=text][ng-model="vm.currentShipment.Packages[vm.packageIndex].ShipperReference"]');
        var inputMiscReference07 = $('input[type=text][ng-model="vm.currentShipment.Packages[vm.packageIndex].MiscReference7"]');
        var shipperReferenceVal = $(inputShipperReference).val() + ',' + OrderNumber;

        var newUniqueString = Tools.GetUniqueCSVStringFromString(shipperReferenceVal);

        /* Set these input values.  ShipperReference is used by the customer, MiscReference7 is used by the SBR to update each record post-ship. */
        /* ShipperReference ONLY keeps the first order number when it goes to PostShip, before the comma, MiscReference7 keeps the entire string. */
        /* On the server-side, if MiscReference7 contains a comma, each item in the csv list gets posted back.  If MiscReference7 is empty, the ShipperReference is used instead for a single post-back. */
        $(inputShipperReference).val(newUniqueString);
        $(inputMiscReference07).val(newUniqueString);
      }

// CurrentShipmentIsD2M() // returns boolean indicating whether the current shipment is Direct to Mobile eligible and enabled
function CurrentShipmentIsD2M()
	{	
		
		var cond1 = CurrentUserIsD2MEnabled();
		var cond2 = $("#"+D2M_TEMPLATE_CHECKBOX_SHIPMENT_ID).is(":checked");
		var cond3 = CurrentShipperIsD2MEnabled();
		return cond1 && cond2  && cond3;
	}

// CurrentSessionIsCapableOfD2MShipments() // returns boolean indicating whether the current session can support Direct to Mobile shipments
function CurrentSessionIsCapableOfD2MShipments()
	{		
		return CurrentUserIsD2MEnabled() && CurrentShipperIsD2MEnabled();
	}

// CurrentShipperIsD2MEnabled() // returns boolean indicating whether the current shipper is authorized for Direct to Mobile
function CurrentShipperIsD2MEnabled(){ 
	
		LogStartMethod("CurrentShipperIsD2MEnabled()");
		
		//var currentShipperSymbol = vm?.currentShipment?.PackageDefaults?.Shipper;
		
		var currentShipperSymbol =  $('select[ng-change="vm.shipperChange()"').val().substring(7).toUpperCase();
		
		var authorizedD2MShipperList = [{"ShipperName": "Alight Solutions (Lincolnshire_D2M)","ShipperSymbol":"LINCOLNSHIRE_D2M"}];
		
		var infoMessage = "D2M shipping for current shipper symbol " + currentShipperSymbol + " is ";
		
		var result =  false;
		
		var isCurrentShipperEnabledForD2M = authorizedD2MShipperList.find(x => x.ShipperSymbol === currentShipperSymbol);
		
		result = (isCurrentShipperEnabledForD2M != undefined) ? true : false;
		
		infoMessage = result? infoMessage + "authorized" : "WARNING " + infoMessage + "NOT authorized";
		
		LogMethodInfo(infoMessage);
		
		return result;		 
    }

// CurrentUserIsD2MEnabled() // returns boolean indicating whether the current user is enabled for Direct to Mobile
function CurrentUserIsD2MEnabled(){
		
		LogStartMethod("CurrentUserIsD2MEnabled()");
		
		var currentUserData = vm?.profile?.UserInformation?.Address.CustomData;
		
		var isD2MEnableForCurrentUser = GetUserCustomData(currentUserData, D2M_USER_CUSTOM_DATA_KEY);
		
		var infoMessage ="";
		
		var result =  false;
		
		if ($.isEmptyObject(isD2MEnableForCurrentUser) === false)
		{
			 var d2mCustomEntry = JSON.stringify(isD2MEnableForCurrentUser);
			 
			 var d2mEnabledValue = JSON.parse(d2mCustomEntry).Value;
			 
			 if (d2mEnabledValue.toUpperCase() === D2M_USER_ENABLED_VALUE)
			 {
				 infoMessage = "D2M User ENABLED";
				 result = true;
			 }
			 else
			 {
				infoMessage = "D2M User DISABLED";
				result = false;
			 }
		}
		else
		{
			 infoMessage = "WARNING:" + D2M_USER_CUSTOM_DATA_KEY + " NOT DETECTED";
		}
		
		LogMethodInfo(infoMessage);
		
		return result;
    }

    MoveGoodToBox(li, destBox)
    {
      let vm = this.ClassViewModel;
      let that = this;
      let data = $(li).data();
      let destCC = this.packages[destBox - 1].CommodityContents || [];

      //find the box the item currently resides in
      //split this statement up over a few calls to make it easier to read
      let allGoods = this.packages.map(m => m.CommodityContents).flat();
      let item = allGoods.find(cc => cc.UniqueId == data.UniqueId);

      if (+item.PkgIndex == +destBox) { console.log('Item already exists in box.'); return false; }

      //remove the item from its current box
      this.packages[item.PkgIndex - 1].CommodityContents = this.packages[item.PkgIndex - 1].CommodityContents.filter(p => p.UniqueId != data.UniqueId);

      //add it to the new box
      item.PkgIndex = destBox;
      destCC.push(item);
      $(li).find('.pkg-index-text').fadeOut().promise().done(function(e) { $(this).text(destBox).fadeIn(); });

      //console.log(vm.currentShipment.Packages);
      $('li[heading="Goods"]:visible:first > a').trigger('click');
      $('#goods .btn-group').find('.btn:gt(0)').not('.active').first().trigger('click');

      //update the weights and values
      //this data will have changed by now so get a new reference to the object
      allGoods = this.packages.map(m => m.CommodityContents).flat();
      //the items array will have the CommodityContents object for every package in the shipment

      //sort by package index
      //update the weight and declared values for all the packages in the shipment
      allGoods = allGoods.sort((a, b) => { return a.PkgIndex - b.PkgIndex });
      $(allGoods).each(function(a, b)
      {
        let pkg = that.packages[b.PkgIndex - 1];

        if ((a == 0) || (b.PkgIndex != allGoods[a - 1].PkgIndex))

        }

        pkg.DeclaredValueAmount.Amount += +b.PVTotalValue;
      });

function getShipperCustomers(shippers) {
                var shipper = shippers.find(shipper => shipper.Symbol == vm.shipping.slic);
                if (shipper == undefined)
                     return [{ name: '--- No Shipper Configured ---', disabled: true }];

                var customData = shipper.CustomData.filter(c => c.Key.toUpperCase() == "CUSTOMER");
                var customers = customData.map(function(c) {
                    var customer = c.Value.toUpperCase();

                    if (customer == 'AMAZON')
                        return { name: "Amazon", value: customer, hasLoad: true, hasReturn: true, autoShip: true };
                    else if (customer == 'UPS')
                        return { name: "UPS", value: customer, hasCameraImage: true, hasFileImage: true, autoShip: false };
                    else if (customer == 'DEMO')
                        return { name: "Demo", value: customer, hasLoad: false, hasReturn: true, hasCameraImage: true, hasFileImage: false, autoShip: false };
                    else if (customer == 'FANATICS')
                        return { name: "Fanatics", value: customer, hasReturn: true, hasCameraImage: true, autoShip: false };
                    else
                        return { name: `--- ${customer} ---`, disabled: true };
                });

                customers.sort(function(a, b) {
                    if (a.disabled && !b.disabled) return 1;
                    if (!a.disabled && b.disabled) return -1;
                    return a.name.localeCompare(b.name);
                });
                return [{ name: '--- Select Customer Option ---', selector: 'select[ng-model="vm.shipping.customer"]', disabled: true }].concat(customers);
            }

this.PostSelectAddressBook = function (shipmentRequest, nameaddress) {
        var stName;
        var vmExists = false;

        var customData1 = PullCustomData(nameaddress.CustomData, "Custom1");
        var customData2 = PullCustomData(nameaddress.CustomData, "Custom2");
        var customData3 = PullCustomData(nameaddress.CustomData, "Custom3");

        if (typeof objVM === "undefined") {
            stName = shipmentRequest.PackageDefaults.OriginAddress.Company;
            shipmentRequest.Packages[0].ConsigneeReference = customData2 + "~" + stName;
        } else {
            vmExists = true;
            stName = objVM.currentShipment.PackageDefaults.OriginAddress.Company;
            objVM.currentShipment.Packages[0].ConsigneeReference = isDefined(customData2) + "~" + stName;
        }
        // alert("customData2 = " + customData2);


        if (isDefined(customData3) == "FC") {
            if (vmExists === true) {
                objVM.currentShipment.PackageDefaults.Terms = "CONSIGNEE";
            } else {
                shipmentRequest.PackageDefaults.Terms = "CONSIGNEE";
            }
            nameaddress.Account = isDefined(customData1);
        }

        function PullCustomData(consignee, field) {
            //alert(JSON.stringify(consignee));

            for (var item in consignee) {
                if (consignee[item].Key == field) return consignee[item].Value;
            }
        }

        function isDefined(referenceValue) {
            var goodValue = null;
            if (typeof referenceValue == "undefined" ||
                referenceValue === null) {
                goodValue = "";
            } else {
                goodValue = referenceValue;
            }
            return goodValue;
        }
    };

    function CheckForRequiredValues()
    {
      var bValid = true;

      if ($('#inputclient').val().trim().length == 0)
      {
        $('#inputclient').addClass('has-error');
        bValid = false;
        UIMethods.SwitchToTab(0);
      }

      /* If the 3rd party billing checkbox is checked, all the text fields need to be filled out. */
      if (($('#checkboxthirdparty').prop('checked') == true) && (bValid == true))
      {
        var textBoxes = $('#tabThirdPartyBilling input[type=text]');
        $(textBoxes).each(function(x, y)
        {
          if ($(this).val().trim().length == 0)
          {
            $(this).addClass('has-error');
            bValid = false;
            UIMethods.SwitchToTab(2);
          }
        });
      }

      return bValid;
    }

// GetOrderInformationFromCustomer(OrderList, CallBack) // returns no explicit return value; performs AJAX lookup of customer order information
      GetOrderInformationFromCustomer: function (OrderList, CallBack)
      {
        var dataObject = {
          ServerMethod: 'GetOrderInformation',
          OrderList: OrderList
        };

        var ajaxObject = {
          UserContext: SessionData.GetCurrentUserContext(),
          Data: JSON.stringify(dataObject)
        };

        var ajaxRequest = $.ajax(
        {
          url: 'api/ShippingService/UserMethod',
          method: 'POST',
          contentType: 'application/x-www-form-urlencoded; charset=UTF-8',
          dataType: 'json',
          data: ajaxObject
        });

        ajaxRequest.done(function (data)
        {
          var jsonData = JSON.parse(data.Data);

          for (var x = 0; x < jsonData.length; x++)
          {
            if (PackVerify.CheckForMatchingConsignee(jsonData[x]) === true)
            {
              PackVerify.AddCommoditiesToTable(jsonData[x].Details);
              PackVerify.UpdateReferencesWithOrderNumbers(jsonData[x].Header.Order_Number);
            }
            else
            {
              /* Add a note to the footer of the dialog. */
              $('#spanPackVerifyFooter').text('Could not add commodities for order ' + jsonData[x].Header.Order_Number);
            }
          }
        });

        ajaxRequest.fail(function (jqXHR, textStatus)
        {
          Logger.Log({ Source: GetCallingFunctionName(), Error: { name: 'AjaxError', message: jqXHR.ResponseText } });
        });

        ajaxRequest.always(function (jqXHR, textStatus, jqXHRError)
        {
          if (CallBack && typeof (CallBack) == "function")
          {
            CallBack();
          }
        });
      }

      SaveConsolidatedOrderNumbers: function ()
      {
        try
        {
          /* Append the shipper reference with each item in the select list in csv format. */
          /* Once this is done, change the SSBR on post load to loop through each option in the csv list. */
          var inputShipperReference = $('input[type=text][ng-model="vm.currentShipment.Packages[vm.packageIndex].ShipperReference"]');
          var shipperReferenceVal = $(inputShipperReference).val();

          /* If the user closes the dialog box, then opens it back up, we don't want to check the same orders more than once so only process what was entered into the dialog box this time. */
          var ordersToQuery = '';

          /* Loop through all the orders in the select box and determine if they are already in the list. */
          $('#selectConsolidatedOrders').find('option').each(function (i, v)
          {
            /* Remove the spaces before the commas for comparison. */
            var arrayOrders = shipperReferenceVal.split(",").map(function (item)
            {
              return item.trim();
            });

            /* If the current option value is NOT in the list within the Shipper Reference field, add it. */
            if ($.inArray(v.value.trim(), arrayOrders) == -1)
            {
              ordersToQuery += v.value.trim() + ',';
              //shipperReferenceVal += ',' + v.value;
            }

            /* Remove the option once it is added to the input box. */
            $(this).remove();

          }).promise().done(function ()
          {
            /* Done creating the list of orders to get from the server which is now held in the ordersToQuery variable. */

            /* Hide the div. */
            $('#divModalConsolidateShipments').hide();

            /* Make sure there are orders in the list. */
            if (ordersToQuery.length > 0)
            {
              PackVerify.GetOrderInformationFromCustomer(ordersToQuery);
            }
            else
            {
              PackVerify.ShowErrorAlert('ERROR: No unique orders to pull from the server.  All consolidated orders have already been added to this shipment.');
            }
          });
        }
        catch (error)
        {
          Logger.Log({ Source: 'SaveConsolidatedOrderNumbers()', Error: error });
        }
      }

        if(canadaSelected || !isSelectedValueInShippersArray()) {
          return; // Exit the function if Canada has been selected in all dropdowns or if the site is not from Canada
        }

        var allSelected = true; // To keep track if all selects have been set to Canada
      
        // Iterate through the NodeList
        selects.forEach(function(select) {
          // Only set the value to 'string:CA' if it's not already set
          if (select.value !== 'string:CA') {
            select.value = 'string:CA';
            allSelected = false; // Not all selects have been set to Canada
      
            // Trigger change event for AngularJS
            if ('createEvent' in document) {
              var evt = document.createEvent('HTMLEvents');
              evt.initEvent('change', false, true);
              select.dispatchEvent(evt);
            } else {
              select.fireEvent('onchange');
            }
          }
        });
      
        // If all selects have been set to Canada, set the flag to true
        if (allSelected) {
          canadaSelected = true;
        }
      }

  UpdateGoodsObject(pkgCount, listItem)
  {
    const itemkey = $(listItem).attr('hash-key');
    const goodkey = $(listItem).attr('goods-key');
    const findex = +$(listItem).attr('pkg-index');

    let val = $(listItem).find('.goods-declared-value');
    let cost = $(val).find('strong').text().replace('$', '');

    cost = (cost / pkgCount).toFixed(2);

    //get the object that matches the hash-key from the working array
    //this will be the template to use when creating new packages after a split operation
    //working on a copy of the vm commodity object so we can abandon changes if needed
    let src = $.grep(this.working, function(value, i) { return value.$$hashKey == itemkey; })[0];

    //update the declared value for the first package
    src.UnitValue.Amount = cost;

    for (let y = 1; y < pkgCount; y++)
    {
      let ccobj = this.GetNewCommodityObject(src);
      ccobj.packageIndex = findex + y;

      this.working.push(ccobj);

      this.UpdateScreenMarkup(ccobj, listItem);
    }
  }

// UpdateListItemPackageQuantities(li) // returns undefined
  UpdateListItemPackageQuantities(li)
  {
    this.UpdatePkgCountsInWorkingObject(li);

    let numOfItems = $(li).find('.num-of-items-in-pkg');
    let remItems = $(li).find('.goods-quantity > strong');

    if (!$(numOfItems).val()) { return; }

    let x = (+$(remItems).text() - $(numOfItems).val());

    //sanity check to make sure we didn't pack more items that we had
    if (x < 0)
    {
      $(remItems).addClass('bad-math');
      return;
    }

    $(remItems).removeClass('bad-math');
    $(numOfItems).attr('disabled', 'disabled');
    $(li).attr('pkg-status', 'packed');

    //move to the next list item and update then focus
    $(li).next().find('.goods-quantity > strong').text(x)

    //on the second-to-last list item, use the input value to auto-set the last item
    //to get the number of rows that have already been assigned, count the list items with the goods-key
    //that have disabled input boxes
    let rowsForThisGood = $('#listItemContainer').find('li[goods-key="' + $(li).attr('goods-key') + '"]');
    let rowsDone = $(rowsForThisGood).find('input:disabled:visible').length;

    if (rowsForThisGood.length == (rowsDone + 1))
    {
      let lastLi = $('#listItemContainer').find('li[goods-key="' + $(li).attr('goods-key') + '"]').last();

      $(lastLi).attr('pkg-status', 'packed');

      let lastVal = $(li).next().find('.goods-quantity > strong').text();
      $(lastLi).find('.num-of-items-in-pkg').val(lastVal).attr('disabled', 'disabled');

      $('#listItemContainer').find('.assign-to-pkg:visible').first().focus();

      this.UpdatePkgCountsInWorkingObject(lastLi);
    }
  }

// ApplyCommodityAssignments() // returns undefined
  ApplyCommodityAssignments()
  {
    let that = this;

    //every list item will have a pkg-index attr, get that highest value to determine how many pkgs we need
    //don't use pkg-status.length because that will include multiple commodities and the end result will be wrong
    let packed = $('#listItemContainer').find('li[pkg-status=packed]');

    //first get all the pkg-index attributes, then get the highest number and add one
    let pkgs = $(packed).map(function(x, value) { return +$(this).attr('pkg-index') }).toArray();
    let pkgCount = Math.max(...pkgs) + 1;

    while (this.packages.length < pkgCount)
    {
      //the vm.addPackage isn't firing, just click the button
      $('button[ng-click="vm.addPackage();"]').trigger('click').promise().done(function(e) { });
    }

    //loop the pkg count, get the commodities for that pkg, write the objects to the vm CommodityContents object
    for (let p = 0; p < pkgCount; p++)
    {
      let w = $.grep(this.working, function(item, i) { return item.packageIndex == p });
      this.packages[p].CommodityContents = w;
    }

    //click the General tab
    $('li[heading=General]:visible a').first().click();
    this.HideAssignGoodsPane();

    //after we are done, move back to the first package in the shipment
    while ($('#spanCurrentPackageIndex').text() !== '1')
    {
      $('button[ng-click="vm.previousPackage();"]').first().click();
    }

  }

// GetNewCommodityObject(obj) // returns object
  GetNewCommodityObject(obj)
  {
    let c = {};

    Object.keys(obj).forEach(key =>
    {
      if (key != '$$hashKey' && key != 'packageIndex')
      {
        let value = obj[key];
        if (value) c[key] = value;
      }
    });

    return c;
  }

                        SetThirdPartyBilling(shipmentRequest);

                break;
            }

        }

        var selectedService = {};
        var allServices = false;
        for (var i = 0; i < this.vm.selectedServices.length; i++) {
            if (shipNextDay) {
                //Next day air

                    selectedService = this.vm.selectedServices[i];
                    break;
                }
            } else {
                //ground
                if (this.vm.selectedServices[i].Symbol == "UPSAPI.UPS.GND") {
                    allServices = true;
                    break;
                    //selectedService = this.vm.selectedServices;   
                }
            }

            //create new variable to assign original list in post rate function
            this.vm.selectedServicesOriginalList = this.vm.selectedServices;
            this.vm.selectedServices = [];
            this.vm.selectedServices = this.vm.selectedServicesOriginalList;
        } else {
            //create new variable to assign original list in post rate function
            this.vm.selectedServicesOriginalList = this.vm.selectedServices;
            this.vm.selectedServices = [];
            this.vm.selectedServices.push(selectedService);
        }

        // console.log(this.vm);
    }
