        {
            const currentDate = new Date();
            currentDate.setDate(currentDate.getDate());

            }

function ReplaceAccountNumberWithXXXXXX() {
		const observer = new MutationObserver(() => {
			// Select all child tracking numbers
			const trackingNumberElements = document.querySelectorAll('div.ng-binding');
			trackingNumberElements.forEach(element => {
				const text = element.textContent.trim();
				if (text.length === 18) {
					element.textContent = text.substring(0, 3) + 'XXXXXX' + text.substring(9);
				}
			});

			// Select all lead tracking numbers
			const tableCells = document.querySelectorAll('td.ng-binding');
			tableCells.forEach(cell => {
				const text = cell.textContent.trim();
				if (text.length === 18) {
					cell.textContent = text.substring(0, 3) + 'XXXXXX' + text.substring(9);
				}
			});
		});

		// Start observing the body for changes
		observer.observe(document.body, { childList: true, subtree: true });
	}
